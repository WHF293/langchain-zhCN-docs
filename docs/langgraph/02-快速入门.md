## 安装

### 使用 npm 安装

```bash
npm install @langchain/langgraph @langchain/core
```

### 使用 yarn 安装

```bash
yarn add @langchain/langgraph @langchain/core
```

### 使用 pnpm 安装

```bash
pnpm add @langchain/langgraph @langchain/core
```

> **注意**：`@langchain/core` 是必需的对等依赖（peer dependency），必须同时安装。

## 创建第一个图

让我们创建一个简单的图，模拟一个基本的对话流程。

### 步骤 1：定义状态

首先，我们需要定义图的状态结构。状态是节点之间传递的数据。

```typescript
import { Annotation } from "@langchain/langgraph";
import { BaseMessage } from "@langchain/core/messages";

// 定义状态注解
const StateAnnotation = Annotation.Root({
  // 消息列表，使用 reducer 函数来追加新消息
  messages: Annotation<BaseMessage[]>({
    reducer: (current, update) => current.concat(update),
    default: () => [],
  }),
  // 当前步骤
  currentStep: Annotation<string>,
});
```

### 步骤 2：创建节点

节点是执行具体任务的函数。每个节点接收当前状态，返回更新后的状态。

```typescript
import { HumanMessage, AIMessage } from "@langchain/core/messages";

// 节点 1：处理用户输入
const processInput = async (state: typeof StateAnnotation.State) => {
  const lastMessage = state.messages[state.messages.length - 1];
  console.log("收到用户消息:", lastMessage.content);

  return {
    currentStep: "processing",
  };
};

// 节点 2：生成回复
const generateResponse = async (state: typeof StateAnnotation.State) => {
  const response = new AIMessage("你好！我是你的 AI 助手。");

  return {
    messages: [response],
    currentStep: "completed",
  };
};
```

### 步骤 3：构建图

使用 `StateGraph` 类来构建图，添加节点和边。

```typescript
import { StateGraph, START, END } from "@langchain/langgraph";

// 创建状态图
const workflow = new StateGraph(StateAnnotation)
  // 添加节点
  .addNode("processInput", processInput)
  .addNode("generateResponse", generateResponse)
  // 定义边
  .addEdge(START, "processInput")           // 从开始到处理输入
  .addEdge("processInput", "generateResponse")  // 从处理输入到生成回复
  .addEdge("generateResponse", END);        // 从生成回复到结束
```

### 步骤 4：编译并运行

编译图并执行。

```typescript
// 编译图
const app = workflow.compile();

// 运行图
const result = await app.invoke({
  messages: [new HumanMessage("你好！")],
  currentStep: "start",
});

console.log("执行结果:", result);
```

## 完整示例

下面是一个完整的、可运行的示例：

```typescript
import { Annotation, StateGraph, START, END } from "@langchain/langgraph";
import { HumanMessage, AIMessage, BaseMessage } from "@langchain/core/messages";

// 1. 定义状态
const StateAnnotation = Annotation.Root({
  messages: Annotation<BaseMessage[]>({
    reducer: (current, update) => current.concat(update),
    default: () => [],
  }),
  currentStep: Annotation<string>,
});

// 2. 定义节点函数
const processInput = async (state: typeof StateAnnotation.State) => {
  const lastMessage = state.messages[state.messages.length - 1];
  console.log("📥 收到消息:", lastMessage.content);
  return { currentStep: "processing" };
};

const generateResponse = async (state: typeof StateAnnotation.State) => {
  const response = new AIMessage("你好！我是 LangGraph 助手，很高兴为你服务！");
  console.log("📤 生成回复:", response.content);
  return {
    messages: [response],
    currentStep: "completed",
  };
};

// 3. 构建图
const workflow = new StateGraph(StateAnnotation)
  .addNode("processInput", processInput)
  .addNode("generateResponse", generateResponse)
  .addEdge(START, "processInput")
  .addEdge("processInput", "generateResponse")
  .addEdge("generateResponse", END);

// 4. 编译图
const app = workflow.compile();

// 5. 运行图
async function main() {
  const result = await app.invoke({
    messages: [new HumanMessage("你好，LangGraph！")],
    currentStep: "start",
  });

  console.log("\n✅ 执行完成！");
  console.log("最终状态:", result);
}

main().catch(console.error);
```

## 添加条件分支

让我们扩展上面的示例，添加条件分支逻辑。

```typescript
import { Annotation, StateGraph, START, END } from "@langchain/langgraph";
import { HumanMessage, AIMessage, BaseMessage } from "@langchain/core/messages";

// 定义状态
const StateAnnotation = Annotation.Root({
  messages: Annotation<BaseMessage[]>({
    reducer: (current, update) => current.concat(update),
    default: () => [],
  }),
  intent: Annotation<string>,  // 用户意图
});

// 节点：分析意图
const analyzeIntent = async (state: typeof StateAnnotation.State) => {
  const lastMessage = state.messages[state.messages.length - 1];
  const content = lastMessage.content as string;

  // 简单的意图识别逻辑
  let intent = "general";
  if (content.includes("天气")) {
    intent = "weather";
  } else if (content.includes("时间")) {
    intent = "time";
  }

  console.log("🔍 识别意图:", intent);
  return { intent };
};

// 节点：处理天气查询
const handleWeather = async (state: typeof StateAnnotation.State) => {
  const response = new AIMessage("今天天气晴朗，温度 25°C，适合外出！");
  return { messages: [response] };
};

// 节点：处理时间查询
const handleTime = async (state: typeof StateAnnotation.State) => {
  const now = new Date().toLocaleString("zh-CN");
  const response = new AIMessage(`当前时间是：${now}`);
  return { messages: [response] };
};

// 节点：处理通用查询
const handleGeneral = async (state: typeof StateAnnotation.State) => {
  const response = new AIMessage("你好！我可以帮你查询天气和时间。请问有什么需要帮助的？");
  return { messages: [response] };
};

// 条件路由函数
const routeByIntent = (state: typeof StateAnnotation.State) => {
  switch (state.intent) {
    case "weather":
      return "handleWeather";
    case "time":
      return "handleTime";
    default:
      return "handleGeneral";
  }
};

// 构建图
const workflow = new StateGraph(StateAnnotation)
  .addNode("analyzeIntent", analyzeIntent)
  .addNode("handleWeather", handleWeather)
  .addNode("handleTime", handleTime)
  .addNode("handleGeneral", handleGeneral)
  .addEdge(START, "analyzeIntent")
  .addConditionalEdges("analyzeIntent", routeByIntent)
  .addEdge("handleWeather", END)
  .addEdge("handleTime", END)
  .addEdge("handleGeneral", END);

// 编译并运行
const app = workflow.compile();

async function main() {
  // 测试天气查询
  console.log("\n--- 测试天气查询 ---");
  await app.invoke({
    messages: [new HumanMessage("今天天气怎么样？")],
    intent: "",
  });

  // 测试时间查询
  console.log("\n--- 测试时间查询 ---");
  await app.invoke({
    messages: [new HumanMessage("现在几点了？")],
    intent: "",
  });
}

main().catch(console.error);
```
