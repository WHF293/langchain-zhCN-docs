# 第十一章 流式输出

> 来源: [LangGraph 官方文档](https://langchain-ai.github.io/langgraph/)

---

## 流式模式

LangGraph 支持多种流式输出模式，允许你实时获取图执行过程中的数据，提升用户体验。

### 流式模式类型

| 模式 | 说明 | 适用场景 |
|------|------|---------|
| **values** | 流式输出每个步骤的完整状态 | 需要查看完整状态变化 |
| **updates** | 流式输出每个步骤的状态更新 | 只关心变化部分 |
| **messages** | 流式输出 LLM 消息 | 聊天应用 |
| **events** | 流式输出所有事件 | 需要详细的执行信息 |
| **debug** | 流式输出调试信息 | 开发调试 |

## streamEvents

### 基本流式输出

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

const State = Annotation.Root({
  input: Annotation<string>,
  output: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    // 模拟耗时处理
    await new Promise((resolve) => setTimeout(resolve, 100));
    return { output: `处理: ${state.input}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();

// 使用 streamEvents 获取流式输出
const stream = graph.streamEvents(
  { input: "测试数据" },
  { version: "v2" }
);

for await (const event of stream) {
  if (event.event === "on_chain_start") {
    console.log("节点开始:", event.name);
  } else if (event.event === "on_chain_end") {
    console.log("节点结束:", event.name, event.data?.output);
  }
}
```

### 流式获取状态更新

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

const State = Annotation.Root({
  messages: Annotation<string[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  step: Annotation<number>({
    default: () => 0,
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("step1", async (state) => {
    await new Promise((resolve) => setTimeout(resolve, 100));
    return {
      messages: ["步骤 1 完成"],
      step: 1,
    };
  })
  .addNode("step2", async (state) => {
    await new Promise((resolve) => setTimeout(resolve, 100));
    return {
      messages: ["步骤 2 完成"],
      step: 2,
    };
  })
  .addEdge("__start__", "step1")
  .addEdge("step1", "step2")
  .addEdge("step2", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "stream-thread" } };

// 流式获取状态更新
const stream = await graph.stream(
  { messages: ["开始"] },
  { ...config, streamMode: "updates" }
);

for await (const update of stream) {
  console.log("状态更新:", update);
}
```

## 自定义流式通道

### 创建自定义通道

```typescript
import {
  StateGraph,
  Annotation,
  BinaryOperatorAggregate,
} from "@langchain/langgraph";

// 自定义消息通道
const messagesChannel = new BinaryOperatorAggregate<string[], string>(
  [],
  (existing, update) => [...existing, update]
);

// 自定义进度通道
const progressChannel = new BinaryOperatorAggregate<number, number>(
  0,
  (_, update) => update
);

const State = Annotation.Root({
  data: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    // 写入自定义通道
    messagesChannel.update("处理开始");
    progressChannel.update(50);

    await new Promise((resolve) => setTimeout(resolve, 100));

    messagesChannel.update("处理完成");
    progressChannel.update(100);

    return { data: `已处理: ${state.data}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();
```

### 流式工具调用结果

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

// 工具调用结果类型
interface ToolCallResult {
  toolName: string;
  args: any;
  result: any;
  timestamp: number;
}

const State = Annotation.Root({
  query: Annotation<string>,
  toolResults: Annotation<ToolCallResult[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  answer: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

// 模拟工具
const tools: Record<string, (args: any) => Promise<string>> = {
  search: async (args) => `搜索结果: ${args.query}`,
  calculate: async (args) => `计算结果: ${eval(args.expression)}`,
};

const graph = new StateGraph(State)
  .addNode("callTools", async (state) => {
    const results: ToolCallResult[] = [];

    // 并行调用工具
    const toolCalls = [
      { name: "search", args: { query: state.query } },
      { name: "calculate", args: { expression: "2 + 2" } },
    ];

    for (const call of toolCalls) {
      const result = await tools[call.name](call.args);
      results.push({
        toolName: call.name,
        args: call.args,
        result,
        timestamp: Date.now(),
      });
    }

    return { toolResults: results };
  })
  .addNode("synthesize", async (state) => {
    const summary = state.toolResults
      .map((r) => `${r.toolName}: ${r.result}`)
      .join("\n");
    return { answer: `综合结果:\n${summary}` };
  })
  .addEdge("__start__", "callTools")
  .addEdge("callTools", "synthesize")
  .addEdge("synthesize", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "tool-stream" } };

// 流式获取工具调用结果
const stream = await graph.stream(
  { query: "搜索查询" },
  { ...config, streamMode: "updates" }
);

for await (const update of stream) {
  if (update.toolResults) {
    console.log("工具调用结果:", update.toolResults);
  }
  if (update.answer) {
    console.log("最终答案:", update.answer);
  }
}
```

## 流式 token

### LLM Token 流式输出

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";
import { ChatOpenAI } from "@langchain/openai";

const State = Annotation.Root({
  prompt: Annotation<string>,
  response: Annotation<string>({
    default: () => "",
    reducer: (prev, curr) => prev + curr,
  }),
});

const model = new ChatOpenAI({
  modelName: "gpt-4",
  streaming: true,
});

const graph = new StateGraph(State)
  .addNode("generate", async (state) => {
    const stream = await model.stream(state.prompt);

    let fullResponse = "";
    for await (const chunk of stream) {
      fullResponse += chunk.content;
      // 这里可以通过自定义通道流式输出每个 token
    }

    return { response: fullResponse };
  })
  .addEdge("__start__", "generate")
  .addEdge("generate", "__end__")
  .compile();

// 流式获取 token
const eventStream = graph.streamEvents(
  { prompt: "请用中文写一首短诗" },
  { version: "v2" }
);

for await (const event of eventStream) {
  if (event.event === "on_chat_model_stream") {
    const content = event.data?.chunk?.content;
    if (content) {
      process.stdout.write(content);  // 实时输出每个 token
    }
  }
}
```

### 配合 LangChain 模型使用

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";
import { ChatOpenAI } from "@langchain/openai";

const State = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  streamingContent: Annotation<string>({
    default: () => "",
    reducer: (prev, curr) => prev + curr,
  }),
});

const model = new ChatOpenAI({
  modelName: "gpt-4",
  streaming: true,
});

const graph = new StateGraph(State)
  .addNode("chat", async (state) => {
    const response = await model.invoke(state.messages);
    return {
      messages: [response],
      streamingContent: response.content as string,
    };
  })
  .addEdge("__start__", "chat")
  .addEdge("chat", "__end__")
  .compile();

// 使用 streamEvents 流式获取模型输出
const stream = graph.streamEvents(
  {
    messages: [{ role: "user", content: "你好，请介绍一下 LangGraph" }],
  },
  { version: "v2" }
);

for await (const event of stream) {
  // 处理不同类型的流式事件
  switch (event.event) {
    case "on_chat_model_stream":
      // LLM token 流
      process.stdout.write(event.data?.chunk?.content || "");
      break;
    case "on_chain_start":
      console.log("\n[节点开始]", event.name);
      break;
    case "on_chain_end":
      console.log("\n[节点结束]", event.name);
      break;
  }
}
```

## 流式工具调用

### 工具调用流式处理

```typescript
import { StateGraph, Annotation, tool } from "@langchain/langgraph";
import { ChatOpenAI } from "@langchain/openai";
import { z } from "zod";

// 定义工具
const searchTool = tool(
  async ({ query }) => {
    // 模拟搜索
    await new Promise((resolve) => setTimeout(resolve, 100));
    return `搜索 "${query}" 的结果`;
  },
  {
    name: "search",
    description: "搜索信息",
    schema: z.object({
      query: z.string().describe("搜索查询"),
    }),
  }
);

const calculatorTool = tool(
  async ({ expression }) => {
    return `计算结果: ${eval(expression)}`;
  },
  {
    name: "calculator",
    description: "数学计算",
    schema: z.object({
      expression: z.string().describe("数学表达式"),
    }),
  }
);

const tools = [searchTool, calculatorTool];

const State = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  toolCalls: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  finalAnswer: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

const model = new ChatOpenAI({
  modelName: "gpt-4",
  streaming: true,
}).bindTools(tools);

const graph = new StateGraph(State)
  .addNode("agent", async (state) => {
    const response = await model.invoke(state.messages);

    // 检查是否有工具调用
    if (response.tool_calls && response.tool_calls.length > 0) {
      return {
        messages: [response],
        toolCalls: response.tool_calls,
      };
    }

    return {
      messages: [response],
      finalAnswer: response.content as string,
    };
  })
  .addNode("tools", async (state) => {
    const results = [];

    for (const call of state.toolCalls) {
      const tool = tools.find((t) => t.name === call.name);
      if (tool) {
        const result = await tool.invoke(call.args);
        results.push({
          role: "tool",
          content: result,
          tool_call_id: call.id,
        });
      }
    }

    return { messages: results, toolCalls: [] };
  })
  .addEdge("__start__", "agent")
  .addConditionalEdge("agent", async (state) => {
    return state.toolCalls.length > 0 ? "tools" : "__end__";
  })
  .addEdge("tools", "agent")
  .compile();

// 流式执行，观察工具调用过程
const stream = graph.streamEvents(
  {
    messages: [{ role: "user", content: "搜索 LangGraph 并计算 2+2" }],
  },
  { version: "v2" }
);

for await (const event of stream) {
  switch (event.event) {
    case "on_chat_model_stream":
      // LLM 流式输出
      if (event.data?.chunk?.tool_calls) {
        console.log("\n[工具调用]", event.data.chunk.tool_calls);
      }
      break;
    case "on_tool_start":
      console.log("[工具开始]", event.name);
      break;
    case "on_tool_end":
      console.log("[工具结束]", event.name, event.data?.output);
      break;
  }
}
```

### 流式工具结果收集

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

interface StreamingToolResult {
  toolName: string;
  status: "pending" | "running" | "completed" | "error";
  result?: string;
  error?: string;
  startTime: number;
  endTime?: number;
}

const State = Annotation.Root({
  task: Annotation<string>,
  toolStatuses: Annotation<Record<string, StreamingToolResult>>({
    reducer: (prev, curr) => ({ ...prev, ...curr }),
    default: () => ({}),
  }),
  finalResult: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

// 模拟异步工具
async function simulateTool(name: string, duration: number): Promise<string> {
  await new Promise((resolve) => setTimeout(resolve, duration));
  return `${name} 完成`;
}

const graph = new StateGraph(State)
  .addNode("executeTools", async (state) => {
    const toolNames = ["search", "analyze", "summarize"];
    const statuses: Record<string, StreamingToolResult> = {};

    // 初始化所有工具状态
    for (const name of toolNames) {
      statuses[name] = {
        toolName: name,
        status: "running",
        startTime: Date.now(),
      };
    }

    // 并行执行工具
    const promises = toolNames.map(async (name) => {
      try {
        const result = await simulateTool(name, Math.random() * 1000);
        statuses[name] = {
          ...statuses[name],
          status: "completed",
          result,
          endTime: Date.now(),
        };
      } catch (e) {
        statuses[name] = {
          ...statuses[name],
          status: "error",
          error: e instanceof Error ? e.message : "未知错误",
          endTime: Date.now(),
        };
      }
    });

    await Promise.all(promises);

    return {
      toolStatuses: statuses,
      finalResult: "所有工具执行完成",
    };
  })
  .addEdge("__start__", "executeTools")
  .addEdge("executeTools", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "tool-status" } };

// 流式监控工具执行状态
const stream = await graph.stream(
  { task: "执行多工具任务" },
  { ...config, streamMode: "updates" }
);

for await (const update of stream) {
  if (update.toolStatuses) {
    for (const [name, status] of Object.entries(update.toolStatuses)) {
      console.log(`[${name}] ${status.status} - ${status.result || status.error || "进行中"}`);
    }
  }
  if (update.finalResult) {
    console.log("\n最终结果:", update.finalResult);
  }
}
```

> **注意**：流式输出需要图编译时支持，确保检查点配置正确。

> **建议**：对于聊天应用，使用 `messages` 流式模式可以提供最佳的用户体验。

> **提示**：使用 `streamEvents` 时，注意 `version` 参数，推荐使用 `"v2"` 版本获取更丰富的事件信息。
