## 错误类型

在 LangGraph 应用中，可能遇到多种类型的错误。

### 错误分类

| 错误类型 | 描述 | 示例 |
|----------|------|------|
| **节点执行错误** | 节点函数抛出异常 | API 调用失败、数据处理错误 |
| **状态错误** | 状态数据异常或不一致 | 无效的状态转换、缺失必需字段 |
| **外部依赖错误** | 外部服务不可用 | 数据库连接失败、第三方 API 超时 |
| **配置错误** | 配置参数错误 | 无效的模型名称、错误的 API 密钥 |
| **资源错误** | 系统资源不足 | 内存溢出、磁盘空间不足 |
| **超时错误** | 执行时间超过限制 | 长时间运行的节点、网络延迟 |

### 错误处理策略

```typescript
import { Annotation } from "@langchain/langgraph";

// 定义带错误处理的状态
const StateWithErrors = Annotation.Root({
  // 正常数据
  input: Annotation<string>,
  output: Annotation<string>,

  // 错误信息
  error: Annotation<string | null>,
  errorType: Annotation<"node" | "state" | "external" | "config" | "resource" | "timeout" | null>,

  // 重试信息
  retryCount: Annotation<number>({
    reducer: (prev, next) => next,
    default: () => 0,
  }),

  // 错误历史
  errorHistory: Annotation<string[]>({
    reducer: (prev, next) => [...prev, ...next],
    default: () => [],
  }),
});
```

## 重试机制

### 基本重试

```typescript
// 带重试的节点
async function retryableNode(
  state: typeof StateWithErrors.State,
  maxRetries: number = 3
) {
  let retryCount = state.retryCount;
  let lastError: Error | null = null;

  while (retryCount < maxRetries) {
    try {
      // 执行节点逻辑
      const result = await processNode(state);
      return {
        ...result,
        retryCount: 0, // 成功后重置重试计数
      };
    } catch (error) {
      lastError = error as Error;
      retryCount++;

      // 记录错误
      console.error(`尝试 ${retryCount}/${maxRetries} 失败:`, error.message);

      if (retryCount < maxRetries) {
        // 指数退避
        const delay = Math.pow(2, retryCount) * 1000;
        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }
  }

  // 所有重试都失败
  return {
    error: `节点执行失败，已重试 ${maxRetries} 次: ${lastError?.message}`,
    errorType: "node" as const,
    retryCount,
    errorHistory: [`重试失败: ${lastError?.message}`],
  };
}
```

### 指数退避重试

```typescript
// 指数退避重试策略
async function exponentialBackoffRetry<T>(
  operation: () => Promise<T>,
  options: {
    maxRetries?: number;
    baseDelay?: number;
    maxDelay?: number;
    backoffMultiplier?: number;
  } = {}
): Promise<T> {
  const {
    maxRetries = 3,
    baseDelay = 1000,
    maxDelay = 30000,
    backoffMultiplier = 2,
  } = options;

  let retryCount = 0;
  let lastError: Error | null = null;

  while (retryCount < maxRetries) {
    try {
      return await operation();
    } catch (error) {
      lastError = error as Error;
      retryCount++;

      if (retryCount < maxRetries) {
        // 计算延迟时间
        const delay = Math.min(
          baseDelay * Math.pow(backoffMultiplier, retryCount - 1),
          maxDelay
        );

        console.warn(`操作失败，${delay}ms 后重试 (${retryCount}/${maxRetries})`);

        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }
  }

  throw new Error(`操作失败，已重试 ${maxRetries} 次: ${lastError?.message}`);
}

// 使用示例
const result = await exponentialBackoffRetry(
  async () => {
    const response = await fetch("https://api.example.com/data");
    if (!response.ok) throw new Error("请求失败");
    return response.json();
  },
  {
    maxRetries: 5,
    baseDelay: 1000,
    maxDelay: 60000,
  }
);
```

### 条件重试

```typescript
// 条件重试：只对特定错误重试
async function conditionalRetry<T>(
  operation: () => Promise<T>,
  shouldRetry: (error: Error) => boolean,
  maxRetries: number = 3
): Promise<T> {
  let retryCount = 0;
  let lastError: Error | null = null;

  while (retryCount < maxRetries) {
    try {
      return await operation();
    } catch (error) {
      lastError = error as Error;

      // 检查是否应该重试
      if (!shouldRetry(lastError)) {
        throw lastError;
      }

      retryCount++;

      if (retryCount < maxRetries) {
        const delay = Math.pow(2, retryCount) * 1000;
        await new Promise((resolve) => setTimeout(resolve, delay));
      }
    }
  }

  throw new Error(`操作失败，已重试 ${maxRetries} 次: ${lastError?.message}`);
}

// 使用示例：只对网络错误重试
const result = await conditionalRetry(
  async () => await fetchData(),
  (error) => {
    // 只对网络错误重试
    return error.message.includes("network") || 
           error.message.includes("timeout");
  }
);
```

## 回滚策略

### 状态回滚

```typescript
import { MemorySaver } from "@langchain/langgraph";

// 带回滚的节点执行
async function rollbackableNode(
  state: typeof StateWithErrors.State,
  checkpointer: MemorySaver
) {
  // 保存当前状态快照
  const snapshot = { ...state };

  try {
    // 执行可能失败的操作
    const result = await processNode(state);
    return result;
  } catch (error) {
    console.error("节点执行失败，回滚到之前的状态:", error);

    // 回滚到快照状态
    return {
      ...snapshot,
      error: (error as Error).message,
      errorType: "node" as const,
    };
  }
}
```

### 事务性回滚

```typescript
// 事务性操作：要么全部成功，要么全部回滚
async function transactionalNode(state: typeof StateWithErrors.State) {
  const operations: Array<() => Promise<any>> = [
    () => updateDatabase(state),
    () => updateCache(state),
    () => sendNotification(state),
  ];

  const completedOps: Array<{ index: number; result: any }> = [];

  try {
    // 执行所有操作
    for (let i = 0; i < operations.length; i++) {
      const result = await operations[i]();
      completedOps.push({ index: i, result });
    }

    return { success: true };
  } catch (error) {
    console.error("事务失败，开始回滚:", error);

    // 回滚已完成的操作（逆序）
    for (const op of completedOps.reverse()) {
      try {
        await rollbackOperation(op.index, op.result);
      } catch (rollbackError) {
        console.error(`回滚操作 ${op.index} 失败:`, rollbackError);
      }
    }

    return {
      error: `事务失败，已回滚: ${(error as Error).message}`,
      errorType: "node" as const,
    };
  }
}
```

### 优雅降级

```typescript
// 优雅降级：失败时使用备用方案
async function gracefulDegradationNode(state: typeof StateWithErrors.State) {
  try {
    // 尝试主要方案
    return await primarySolution(state);
  } catch (primaryError) {
    console.warn("主要方案失败，尝试备用方案:", primaryError);

    try {
      // 尝试备用方案
      return await fallbackSolution(state);
    } catch (fallbackError) {
      console.warn("备用方案也失败，使用默认方案:", fallbackError);

      // 使用默认方案
      return {
        output: "由于技术问题，使用默认回答。",
        error: (primaryError as Error).message,
        errorType: "node" as const,
      };
    }
  }
}
```

## 状态恢复

### 检查点恢复

```typescript
import { MemorySaver } from "@langchain/langgraph";

// 创建检查点
const checkpointer = new MemorySaver();

// 编译图时添加检查点
const app = workflow.compile({ checkpointer });

// 运行图
const config = { configurable: { thread_id: "thread-1" } };
const result = await app.invoke(input, config);

// 恢复到检查点
const checkpoint = await checkpointer.get(config);
console.log("检查点状态:", checkpoint);

// 从检查点继续执行
const continuedResult = await app.invoke(null, config);
```

### 手动状态恢复

```typescript
// 手动恢复状态
async function recoverState(
  threadId: string,
  checkpointer: MemorySaver
) {
  const config = { configurable: { thread_id: threadId } };

  // 获取检查点
  const checkpoint = await checkpointer.get(config);

  if (!checkpoint) {
    throw new Error(`未找到线程 ${threadId} 的检查点`);
  }

  // 恢复状态
  const recoveredState = checkpoint.state;

  // 验证状态完整性
  if (!isValidState(recoveredState)) {
    throw new Error("恢复的状态无效");
  }

  return recoveredState;
}

// 状态验证
function isValidState(state: any): boolean {
  return (
    state &&
    typeof state.input === "string" &&
    typeof state.status === "string"
  );
}
```

### 状态迁移

```typescript
// 状态迁移：处理版本变化
async function migrateState(
  oldState: any,
  fromVersion: number,
  toVersion: number
) {
  let currentState = { ...oldState };

  // 逐步迁移
  for (let v = fromVersion; v < toVersion; v++) {
    currentState = await applyMigration(currentState, v, v + 1);
  }

  return currentState;
}

// 应用迁移
async function applyMigration(state: any, from: number, to: number): Promise<any> {
  switch (`${from}-${to}`) {
    case "1-2":
      // 版本 1 到 2 的迁移
      return {
        ...state,
        newField: state.oldField, // 重命名字段
        version: 2,
      };
    case "2-3":
      // 版本 2 到 3 的迁移
      return {
        ...state,
        nested: {
          value: state.value, // 重构嵌套结构
        },
        version: 3,
      };
    default:
      return state;
  }
}
```

### 错误监控

```typescript
// 错误监控节点
async function monitoredNode(state: typeof StateWithErrors.State) {
  const startTime = Date.now();

  try {
    const result = await processNode(state);

    // 记录成功指标
    await recordMetrics({
      node: "processNode",
      status: "success",
      duration: Date.now() - startTime,
    });

    // 记录失败指标

      level: "error",
      message: `节点执行失败: ${(error as Error).message}`,

    });

    throw error;
  }
}
```

### 性能监控

```typescript
// 性能监控
async function performanceMonitoredNode(state: typeof StateWithErrors.State) {
  const metrics = {
    startTime: Date.now(),
    memoryUsage: process.memoryUsage(),
    cpuUsage: process.cpuUsage(),
  };

  try {
    const result = await processNode(state);

    // 计算性能指标
    const endTime = Date.now();
    const duration = endTime - metrics.startTime;
    const memoryDelta = process.memoryUsage().heapUsed - metrics.memoryUsage.heapUsed;

    // 检查性能阈值
    if (duration > 5000) {
      await sendAlert({
        level: "warning",
        message: `节点执行时间过长: ${duration}ms`,
        node: "processNode",
      });
    }

    if (memoryDelta > 100 * 1024 * 1024) { // 100MB
      await sendAlert({
        level: "warning",
        message: `节点内存使用过高: ${memoryDelta / 1024 / 1024}MB`,
        node: "processNode",
      });
    }

    return result;
  } catch (error) {
    throw error;
  }
}
```

### 健康检查

```typescript
// 健康检查节点
async function healthCheckNode(state: typeof StateWithErrors.State) {
  const checks = {
    database: await checkDatabase(),
    cache: await checkCache(),
    externalApi: await checkExternalApi(),
  };

  const isHealthy = Object.values(checks).every((check) => check.status === "ok");

  if (!isHealthy) {
    const failedChecks = Object.entries(checks)
      .filter(([_, check]) => check.status !== "ok")
      .map(([name, check]) => `${name}: ${check.error}`);

    await sendAlert({
      level: "error",
      message: `健康检查失败: ${failedChecks.join(", ")}`,
      node: "healthCheck",
    });
  }

  return {
    isHealthy,
    checks,
    timestamp: new Date().toISOString(),
  };
}

// 检查数据库
async function checkDatabase() {
  try {
    await database.query("SELECT 1");
    return { status: "ok" };
  } catch (error) {
    return { status: "error", error: (error as Error).message };
  }
}

// 检查缓存
async function checkCache() {
  try {
    await cache.set("health-check", "ok");
    const value = await cache.get("health-check");
    return { status: value === "ok" ? "ok" : "error" };
  } catch (error) {
    return { status: "error", error: (error as Error).message };
  }
}

// 检查外部 API
async function checkExternalApi() {
  try {
    const response = await fetch("https://api.example.com/health");
    return { status: response.ok ? "ok" : "error" };
  } catch (error) {
    return { status: "error", error: (error as Error).message };
  }
}
```

### 监控最佳实践

| 实践 | 说明 |
|------|------|
| **记录所有错误** | 即使是可恢复的错误也要记录 |
| **设置合理的阈值** | 避免告警疲劳 |
| **监控关键指标** | 延迟、错误率、资源使用 |
| **建立基线** | 了解正常行为模式 |
| **定期审查** | 定期审查监控数据和告警 |

### 容错架构图

```typescript
// 完整的容错架构
const faultTolerantGraph = new StateGraph(StateAnnotation)
  // 健康检查
  .addNode("healthCheck", healthCheckNode)

  // 带重试的主处理
  .addNode("process", retryableNode)

  // 回滚处理
  .addNode("rollback", rollbackableNode)

  // 优雅降级
  .addNode("fallback", gracefulDegradationNode)

  // 错误处理
  .addNode("errorHandler", errorHandlerNode)

  // 定义流程
  .addEdge("__start__", "healthCheck")
  .addConditionalEdges("healthCheck", (state) => {
    if (state.isHealthy) return "process";
    return "errorHandler";
  })
  .addConditionalEdges("process", (state) => {
    if (state.error) return "rollback";
    return "__end__";
  })
  .addConditionalEdges("rollback", (state) => {
    if (state.retryCount < 3) return "process";
    return "fallback";
  })
  .addEdge("fallback", "__end__")
  .addEdge("errorHandler", "__end__");
```

> **注意**：容错机制会增加系统的复杂性，需要在可靠性和复杂性之间找到平衡。

> **提示**：在生产环境中，建议结合使用重试、回滚和优雅降级等多种容错策略。

> **建议**：建立完善的监控告警系统，及时发现和处理问题。
