## 单元测试

单元测试用于验证单个节点或函数的正确性。

### 节点单元测试

```typescript
import { StateGraph, END } from "@langchain/langgraph";

// 被测试的节点函数
const processData = (input: string): string => {
  return input.trim().toUpperCase();
};

// 单元测试
describe("processData 节点", () => {
  test("应该正确处理输入", () => {
    const result = processData("  hello  ");
    expect(result).toBe("HELLO");
  });

  test("应该处理空字符串", () => {
    const result = processData("");
    expect(result).toBe("");
  });

  test("应该去除首尾空格", () => {
    const result = processData("  test  ");
    expect(result).toBe("TEST");
  });
});
```

### 工作流单元测试

```typescript
import { StateGraph, END } from "@langchain/langgraph";

interface State {
  input: string;
  output: string;
}

// 创建测试用的工作流
function createTestWorkflow() {
  const workflow = new StateGraph<State>({
    channels: {
      input: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
      output: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
    },
  });

  const processNode = async (state: State) => {
    return {
      output: `处理: ${state.input}`,
    };
  };

  workflow.addNode("process", processNode);
  workflow.addEdge("process", END);
  workflow.setEntryPoint("process");

  return workflow.compile();
}

// 测试工作流
describe("工作流测试", () => {
  let app: ReturnType<typeof createTestWorkflow>;

  beforeEach(() => {
    app = createTestWorkflow();
  });

  test("应该正确执行工作流", async () => {
    const result = await app.invoke({
      input: "测试数据",
      output: "",
    });

    expect(result.output).toBe("处理: 测试数据");
  });

  test("应该处理空输入", async () => {
    const result = await app.invoke({
      input: "",
      output: "",
    });

    expect(result.output).toBe("处理: ");
  });
});
```

## 集成测试

集成测试验证多个组件协同工作的正确性。

### 多节点集成测试

```typescript
import { StateGraph, END } from "@langchain/langgraph";

interface State {
  data: string;
  validated: boolean;
  processed: string;
}

// 创建集成测试用的工作流
function createIntegrationWorkflow() {
  const workflow = new StateGraph<State>({
    channels: {
      data: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
      validated: {
        value: (prev: boolean, next: boolean) => next,
        default: () => false,
      },
      processed: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
    },
  });

  // 验证节点
  const validate = async (state: State) => {
    if (!state.data) {
      throw new Error("数据不能为空");
    }
    return { validated: true };
  };

  // 处理节点
  const process = async (state: State) => {
    if (!state.validated) {
      throw new Error("数据未验证");
    }
    return { processed: `已处理: ${state.data}` };
  };

  workflow.addNode("validate", validate);
  workflow.addNode("process", process);
  workflow.addEdge("validate", "process");
  workflow.addEdge("process", END);
  workflow.setEntryPoint("validate");

  return workflow.compile();
}

// 集成测试
describe("集成测试", () => {
  let app: ReturnType<typeof createIntegrationWorkflow>;

  beforeEach(() => {
    app = createIntegrationWorkflow();
  });

  test("完整的验证-处理流程", async () => {
    const result = await app.invoke({
      data: "测试数据",
      validated: false,
      processed: "",
    });

    expect(result.validated).toBe(true);
    expect(result.processed).toBe("已处理: 测试数据");
  });

  test("空数据应该抛出错误", async () => {
    await expect(
      app.invoke({
        data: "",
        validated: false,
        processed: "",
      })
    ).rejects.toThrow("数据不能为空");
  });
});
```

### 外部依赖集成测试

```typescript
import { StateGraph, END } from "@langchain/langgraph";

interface State {
  query: string;
  response: string;
}

// Mock 外部 API
const mockExternalAPI = {
  search: jest.fn().mockResolvedValue({
    results: ["结果1", "结果2"],
  }),
};

// 使用 Mock 的工作流
function createWorkflowWithMock() {
  const workflow = new StateGraph<State>({
    channels: {
      query: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
      response: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
    },
  });

  const searchNode = async (state: State) => {
    const results = await mockExternalAPI.search(state.query);
    return {
      response: `找到 ${results.results.length} 个结果`,
    };
  };

  workflow.addNode("search", searchNode);
  workflow.addEdge("search", END);
  workflow.setEntryPoint("search");

  return workflow.compile();
}

// 外部依赖测试
describe("外部依赖集成测试", () => {
  test("应该正确调用外部 API", async () => {
    const app = createWorkflowWithMock();

    await app.invoke({
      query: "测试查询",
      response: "",
    });

    expect(mockExternalAPI.search).toHaveBeenCalledWith("测试查询");
  });
});
```

## 端到端测试

端到端测试验证整个系统的完整功能。

### 完整流程测试

```typescript
import { StateGraph, END } from "@langchain/langgraph";

interface AgentState {
  messages: Array<{ role: string; content: string }>;
  currentStep: string;
  finalResponse: string;
}

// 端到端测试工作流
function createE2EWorkflow() {
  const workflow = new StateGraph<AgentState>({
    channels: {
      messages: {
        value: (
          prev: Array<{ role: string; content: string }>,
          next: Array<{ role: string; content: string }>
        ) => [...prev, ...next],
        default: () => [],
      },
      currentStep: {
        value: (prev: string, next: string) => next,
        default: () => "start",
      },
      finalResponse: {
        value: (prev: string, next: string) => next,
        default: () => "",
      },
    },
  });

  // 理解意图节点
  const understandIntent = async (state: AgentState) => {
    const lastMessage = state.messages[state.messages.length - 1];
    return {
      currentStep: "understood",
      messages: [
        {
          role: "system",
          content: `理解意图: ${lastMessage.content}`,
        },
      ],
    };
  };

  // 生成响应节点
  const generateResponse = async (state: AgentState) => {
    return {
      currentStep: "completed",
      finalResponse: `回答: ${state.messages[state.messages.length - 1].content}`,
    };
  };

  workflow.addNode("understand", understandIntent);
  workflow.addNode("respond", generateResponse);
  workflow.addEdge("understand", "respond");
  workflow.addEdge("respond", END);
  workflow.setEntryPoint("understand");

  return workflow.compile();
}

// 端到端测试
describe("端到端测试", () => {
  let app: ReturnType<typeof createE2EWorkflow>;

  beforeEach(() => {
    app = createE2EWorkflow();
  });

  test("完整的对话流程", async () => {
    const result = await app.invoke({
      messages: [{ role: "user", content: "你好" }],
      currentStep: "start",
      finalResponse: "",
    });

    expect(result.currentStep).toBe("completed");
    expect(result.finalResponse).toContain("回答");
  });

  test("多轮对话", async () => {
    // 第一轮
    const result1 = await app.invoke({
      messages: [{ role: "user", content: "你好" }],
      currentStep: "start",
      finalResponse: "",
    });

    // 第二轮
    const result2 = await app.invoke({
      messages: [
        ...result1.messages,
        { role: "user", content: "天气怎么样?" },
      ],
      currentStep: result1.currentStep,
      finalResponse: "",
    });

    expect(result2.messages.length).toBeGreaterThan(result1.messages.length);
  });
});
```

## 测试工具

### 测试辅助函数

```typescript
import { StateGraph, END } from "@langchain/langgraph";

// 测试工具类
class TestHelper {
  // 创建测试状态
  static createTestState<T>(overrides: Partial<T> = {}): T {
    return {
      ...overrides,
    } as T;
  }

  // 创建测试工作流
  static createTestWorkflow<T>(nodes: Record<string, Function>) {
    const workflow = new StateGraph<T>({
      channels: Object.keys(nodes).reduce(
        (acc, key) => ({
          ...acc,
          [key]: {
            value: (prev: any, next: any) => next,
            default: () => null,
          },
        }),
        {}
      ),
    });

    Object.entries(nodes).forEach(([name, handler]) => {
      workflow.addNode(name, handler);
    });

    return workflow;
  }

  // 断言工作流输出
  static async assertWorkflowOutput<T>(
    app: any,
    input: T,
    expected: Partial<T>
  ) {
    const result = await app.invoke(input);
    Object.entries(expected).forEach(([key, value]) => {
      expect(result[key]).toEqual(value);
    });
  }
}

// 使用测试工具
describe("使用测试工具", () => {
  test("应该正确使用测试辅助函数", async () => {
    interface State {
      input: string;
      output: string;
    }

    const workflow = TestHelper.createTestWorkflow<State>({
      process: async (state: State) => ({
        output: `处理: ${state.input}`,
      }),
    });

    workflow.addEdge("process", END);
    workflow.setEntryPoint("process");

    const app = workflow.compile();

    await TestHelper.assertWorkflowOutput(
      app,
      { input: "测试", output: "" },
      { output: "处理: 测试" }
    );
  });
});
```

## 测试最佳实践

### 测试结构

```
tests/
├── unit/              # 单元测试
│   ├── nodes/         # 节点测试
│   └── utils/         # 工具函数测试
├── integration/       # 集成测试
│   └── workflows/     # 工作流测试
├── e2e/               # 端到端测试
└── fixtures/          # 测试夹具
```

### 测试最佳实践总结

| 实践 | 说明 |
|------|------|
| 测试独立性 | 每个测试应该独立运行，不依赖其他测试 |
| 测试覆盖 | 覆盖正常流程和边界情况 |
| Mock 外部依赖 | 使用 Mock 隔离外部依赖 |
| 测试命名 | 使用描述性的测试名称 |
| 测试数据 | 使用有意义的测试数据 |

> **提示**：使用 `jest --watch` 可以在开发过程中自动运行受影响的测试。

> **建议**：保持测试覆盖率在 80% 以上，但不要为了覆盖率而写无意义的测试。

> **注意**：端到端测试可能较慢，可以在 CI/CD 中单独运行。
