# 第十章 中断与人在回路

> 来源: [LangGraph 官方文档](https://langchain-ai.github.io/langgraph/)

---

## 中断的概念

中断（Interrupt）是 LangGraph 实现人在回路（Human-in-the-Loop）的核心机制。它允许图在执行过程中暂停，等待人工干预后再继续执行。

### 人在回路的应用场景

| 场景 | 说明 |
|------|------|
| **审批流程** | 在执行关键操作前需要人工审批 |
| **数据验证** | 人工验证 AI 生成的内容是否正确 |
| **决策点** | 在多个选项中由人工做出选择 |
| **错误修正** | 人工修正 AI 的错误输出 |
| **质量控制** | 在输出前进行人工质量检查 |

## interrupt() 函数

### 基本用法

```typescript
import { StateGraph, Annotation, interrupt, MemorySaver } from "@langchain/langgraph";

const State = Annotation.Root({
  input: Annotation<string>,
  approved: Annotation<boolean>({
    default: () => false,
    reducer: (_, curr) => curr,
  }),
  result: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    // 处理输入
    return { result: `处理结果: ${state.input}` };
  })
  .addNode("humanApproval", async (state) => {
    // 中断执行，等待人工审批
    const humanFeedback = interrupt({
      question: "请审批以下内容：",
      content: state.result,
      options: ["批准", "拒绝", "修改"],
    });

    // 根据人工反馈更新状态
    return {
      approved: humanFeedback === "批准",
      result: humanFeedback === "修改"
        ? `已修改: ${state.result}`
        : state.result,
    };
  })
  .addNode("finalize", async (state) => {
    if (state.approved) {
      return { result: `已批准: ${state.result}` };
    }
    return { result: `已拒绝: ${state.result}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "humanApproval")
  .addEdge("humanApproval", "finalize")
  .addEdge("finalize", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 第一次执行，会在 interrupt 处暂停
const config = { configurable: { thread_id: "approval-thread" } };
let result = await graph.invoke({ input: "重要数据" }, config);

// 检查是否被中断
if (result.__interrupt__) {
  console.log("图已暂停，等待人工审批");
  console.log("中断信息:", result.__interrupt__);
}
```

### 恢复执行

```typescript
import { StateGraph, Annotation, interrupt, MemorySaver } from "@langchain/langgraph";
import { Command } from "@langchain/langgraph";

const State = Annotation.Root({
  input: Annotation<string>,
  feedback: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
  result: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("askFeedback", async (state) => {
    const feedback = interrupt({
      question: "请输入您的反馈：",
      context: state.input,
    });
    return { feedback };
  })
  .addNode("process", async (state) => {
    return { result: `基于反馈 "${state.feedback}" 处理: ${state.input}` };
  })
  .addEdge("__start__", "askFeedback")
  .addEdge("askFeedback", "process")
  .addEdge("process", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "feedback-thread" } };

// 第一次执行，触发中断
await graph.invoke({ input: "用户请求" }, config);

// 恢复执行，提供人工输入
const resumeResult = await graph.invoke(
  new Command({ resume: "很好，继续" }),
  config
);

console.log("恢复执行结果:", resumeResult);
```

## 审批流程

### 完整审批工作流

```typescript
import {
  StateGraph,
  Annotation,
  interrupt,
  Command,
  MemorySaver,
} from "@langchain/langgraph";

// 审批状态定义
const ApprovalState = Annotation.Root({
  request: Annotation<{
    type: string;
    amount: number;
    description: string;
  }>,
  approvalLevel: Annotation<string>({
    default: () => "pending",
    reducer: (_, curr) => curr,
  }),
  approver: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
  comments: Annotation<string[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  finalDecision: Annotation<string>({
    default: () => "pending",
    reducer: (_, curr) => curr,
  }),
});

// 创建审批图
const approvalGraph = new StateGraph(ApprovalState)
  // 自动检查节点
  .addNode("autoCheck", async (state) => {
    const { amount } = state.request;

    // 小额自动批准
    if (amount < 1000) {
      return {
        approvalLevel: "auto-approved",
        finalDecision: "approved",
      };
    }

    // 大额需要人工审批
    return {
      approvalLevel: amount > 10000 ? "senior-approval" : "manager-approval",
    };
  })

  // 人工审批节点
  .addNode("humanApproval", async (state) => {
    // 如果已自动批准，跳过人工审批
    if (state.approvalLevel === "auto-approved") {
      return {};
    }

    // 中断等待人工审批
    const decision = interrupt({
      type: "approval_request",
      level: state.approvalLevel,
      request: state.request,
      question: `请审批以下 ${state.approvalLevel} 级别的请求`,
    });

    return {
      approver: decision.approver || "未知",
      comments: decision.comments ? [decision.comments] : [],
      finalDecision: decision.approved ? "approved" : "rejected",
    };
  })

  // 通知节点
  .addNode("notify", async (state) => {
    const message = state.finalDecision === "approved"
      ? `请求已批准 - 审批人: ${state.approver}`
      : `请求已拒绝 - 原因: ${state.comments.join(", ")}`;

    return { comments: [message] };
  })

  // 连接边
  .addEdge("__start__", "autoCheck")
  .addConditionalEdge("autoCheck", async (state) => {
    if (state.finalDecision === "approved") {
      return "notify";
    }
    return "humanApproval";
  })
  .addEdge("humanApproval", "notify")
  .addEdge("notify", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 使用审批图
const config = { configurable: { thread_id: "approval-1" } };

// 触发审批流程
await approvalGraph.invoke(
  {
    request: {
      type: "采购",
      amount: 5000,
      description: "购买办公设备",
    },
  },
  config
);

// 模拟人工审批
await approvalGraph.invoke(
  new Command({
    resume: {
      approver: "张经理",
      approved: true,
      comments: "同意采购",
    },
  }),
  config
);
```

## 人工编辑状态

### 允许人工修改状态

```typescript
import {
  StateGraph,
  Annotation,
  interrupt,
  Command,
  MemorySaver,
} from "@langchain/langgraph";

const State = Annotation.Root({
  draft: Annotation<string>,
  edited: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
  version: Annotation<number>({
    default: () => 1,
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("generate", async (state) => {
    // AI 生成草稿
    return {
      draft: "这是一份 AI 生成的文档草稿。包含了一些需要人工审核的内容。",
      version: state.version,
    };
  })
  .addNode("review", async (state) => {
    // 中断，允许人工编辑
    const editedContent = interrupt({
      type: "edit_request",
      content: state.draft,
      instructions: "请审核并编辑以下内容，完成后提交",
    });

    return {
      edited: editedContent,
      version: state.version + 1,
    };
  })
  .addNode("finalize", async (state) => {
    return {
      draft: state.edited || state.draft,
    };
  })
  .addEdge("__start__", "generate")
  .addEdge("generate", "review")
  .addEdge("review", "finalize")
  .addEdge("finalize", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "edit-thread" } };

// 生成草稿
await graph.invoke({ draft: "" }, config);

// 人工编辑并提交
const result = await graph.invoke(
  new Command({
    resume: "这是人工编辑后的内容。已修正了 AI 的错误。",
  }),
  config
);

console.log("最终版本:", result.draft);
console.log("版本号:", result.version);
```

## 条件中断

### 基于条件的中断

```typescript
import {
  StateGraph,
  Annotation,
  interrupt,
  Command,
  MemorySaver,
} from "@langchain/langgraph";

const State = Annotation.Root({
  data: Annotation<string>,
  confidence: Annotation<number>({
    default: () => 1.0,
    reducer: (_, curr) => curr,
  }),
  needsReview: Annotation<boolean>({
    default: () => false,
    reducer: (_, curr) => curr,
  }),
  result: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("analyze", async (state) => {
    // 模拟 AI 分析，产生不同的置信度
    const confidence = Math.random();
    return {
      confidence,
      needsReview: confidence < 0.8,  // 置信度低于 0.8 需要人工审核
    };
  })
  .addNode("humanReview", async (state) => {
    // 只有需要审核时才中断
    if (state.needsReview) {
      const feedback = interrupt({
        type: "low_confidence",
        confidence: state.confidence,
        data: state.data,
        question: "AI 置信度较低，请确认结果",
      });
      return { result: `人工确认: ${feedback}` };
    }
    return { result: `AI 高置信度结果: ${state.data}` };
  })
  .addEdge("__start__", "analyze")
  .addConditionalEdge("analyze", async (state) => {
    if (state.needsReview) {
      return "humanReview";
    }
    return "__end__";
  })
  .addEdge("humanReview", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 使用条件中断
const config = { configurable: { thread_id: "conditional-thread" } };
await graph.invoke({ data: "需要分析的数据" }, config);

// 如果被中断，恢复执行
const result = await graph.invoke(
  new Command({ resume: "确认无误" }),
  config
);

console.log("结果:", result);
```

### 多级审批中断

```typescript
import {
  StateGraph,
  Annotation,
  interrupt,
  Command,
  MemorySaver,
} from "@langchain/langgraph";

const State = Annotation.Root({
  request: Annotation<string>,
  level: Annotation<number>({
    default: () => 1,
    reducer: (_, curr) => curr,
  }),
  approvals: Annotation<string[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  status: Annotation<string>({
    default: () => "pending",
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("level1Approval", async (state) => {
    const decision = interrupt({
      level: 1,
      approver: "一级审批人",
      request: state.request,
    });

    return {
      approvals: [`一级: ${decision}`],
      level: 2,
      status: decision === "approved" ? "level1-passed" : "rejected",
    };
  })
  .addNode("level2Approval", async (state) => {
    if (state.status === "rejected") {
      return {};
    }

    const decision = interrupt({
      level: 2,
      approver: "二级审批人",
      request: state.request,
      previousApprovals: state.approvals,
    });

    return {
      approvals: [`二级: ${decision}`],
      level: 3,
      status: decision === "approved" ? "level2-passed" : "rejected",
    };
  })
  .addNode("level3Approval", async (state) => {
    if (state.status === "rejected") {
      return {};
    }

    const decision = interrupt({
      level: 3,
      approver: "三级审批人",
      request: state.request,
      previousApprovals: state.approvals,
    });

    return {
      approvals: [`三级: ${decision}`],
      status: decision === "approved" ? "fully-approved" : "rejected",
    };
  })
  .addEdge("__start__", "level1Approval")
  .addConditionalEdge("level1Approval", async (state) => {
    if (state.status === "rejected") return "__end__";
    return "level2Approval";
  })
  .addConditionalEdge("level2Approval", async (state) => {
    if (state.status === "rejected") return "__end__";
    return "level3Approval";
  })
  .addEdge("level3Approval", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 多级审批流程
const config = { configurable: { thread_id: "multi-approval" } };

// 第一级审批
await graph.invoke({ request: "重要申请" }, config);

// 模拟第一级批准
await graph.invoke(new Command({ resume: "approved" }), config);

// 模拟第二级批准
await graph.invoke(new Command({ resume: "approved" }), config);

// 模拟第三级批准
const finalResult = await graph.invoke(new Command({ resume: "approved" }), config);

console.log("最终状态:", finalResult.status);
console.log("审批记录:", finalResult.approvals);
```

> **注意**：`interrupt()` 函数只能在节点内部调用，不能在边或条件边中使用。

> **建议**：对于关键操作，建议实现多级审批机制，确保操作的安全性和合规性。

> **提示**：使用 `Command` 对象恢复执行时，`resume` 的值会作为 `interrupt()` 函数的返回值。
