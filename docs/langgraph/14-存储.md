## BaseStore 接口

LangGraph 提供了 `BaseStore` 接口用于管理长期记忆和共享数据。与检查点（Checkpointer）不同，存储（Store）用于跨线程共享数据。

### 存储 vs 检查点

| 特性 | Store | Checkpointer |
|------|-------|--------------|
| **用途** | 长期记忆、共享数据 | 状态持久化、时间旅行 |
| **作用域** | 跨线程共享 | 单线程隔离 |
| **生命周期** | 持久化 | 随线程 |
| **访问模式** | 键值对、语义搜索 | 状态快照 |

### BaseStore 接口定义

```typescript
import { BaseStore, Item } from "@langchain/langgraph";

// BaseStore 接口
interface BaseStoreInterface {
  // 获取单个项
  get(namespace: string[], key: string): Promise<Item | undefined>;

  // 批量获取
  mget(keys: [string[], string][]): Promise<(Item | undefined)[]>;

  // 设置项
  put(namespace: string[], key: string, value: any): Promise<void>;

  // 批量设置
  mput(items: [string[], string, any][]): Promise<void>;

  // 删除项
  delete(namespace: string[], key: string): Promise<void>;

  // 搜索
  search(
    namespace: string[],
    options?: {
      query?: string;
      filter?: Record<string, any>;
      limit?: number;
      offset?: number;
    }
  ): Promise<Item[]>;

  // 列出命名空间
  listNamespaces(options?: {
    prefix?: string[];
    suffix?: string[];
    maxDepth?: number;
    limit?: number;
    offset?: number;
  }): Promise<string[][]>;
}

// Item 类型
interface Item {
  namespace: string[];
  key: string;
  value: any;
  createdAt?: string;
  updatedAt?: string;
}
```

## InMemoryStore

### 基本用法

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";

// 创建内存存储
const store = new InMemoryStore();

const State = Annotation.Root({
  userId: Annotation<string>,
  preferences: Annotation<Record<string, any>>({
    reducer: (prev, curr) => ({ ...prev, ...curr }),
    default: () => ({}),
  }),
});

const graph = new StateGraph(State)
  .addNode("loadPreferences", async (state) => {
    // 从存储中加载用户偏好
    const item = await store.get(["users", state.userId], "preferences");
    return {
      preferences: item?.value || {},
    };
  })
  .addNode("updatePreferences", async (state) => {
    // 保存用户偏好到存储
    await store.put(
      ["users", state.userId],
      "preferences",
      state.preferences
    );
    return {};
  })
  .addEdge("__start__", "loadPreferences")
  .addEdge("loadPreferences", "updatePreferences")
  .addEdge("updatePreferences", "__end__")
  .compile();

// 使用图
const config = { configurable: { thread_id: "user-session" } };
await graph.invoke(
  {
    userId: "user-123",
    preferences: { theme: "dark", language: "zh-CN" },
  },
  config
);
```

### 批量操作

```typescript
import { InMemoryStore } from "@langchain/langgraph";

const store = new InMemoryStore();

// 批量设置
await store.mput([
  [["users", "user-1"], "name", "张三"],
  [["users", "user-1"], "email", "zhangsan@example.com"],
  [["users", "user-2"], "name", "李四"],
  [["users", "user-2"], "email", "lisi@example.com"],

// 批量获取
const items = await store.mget([
  [["users", "user-1"], "name"],
  [["users", "user-1"], "email"],
  [["users", "user-2"], "name"],
]);

console.log("批量获取结果:", items);
// [
//   { namespace: ["users", "user-1"], key: "name", value: "张三" },
//   { namespace: ["users", "user-1"], key: "email", value: "zhangsan@example.com" },

// ]
```

### 命名空间操作

```typescript
import { InMemoryStore } from "@langchain/langgraph";

const store = new InMemoryStore();

// 设置不同命名空间的数据
await store.put(["app", "config"], "version", "1.0.0");
await store.put(["app", "config"], "maxRetries", 3);
await store.put(["app", "users", "user-1"], "name", "张三");
await store.put(["app", "users", "user-2"], "name", "李四");

// 列出所有命名空间
const namespaces = await store.listNamespaces({
  prefix: ["app"],
  maxDepth: 3,
});

console.log("命名空间列表:", namespaces);
// [["app", "config"], ["app", "users", "user-1"], ["app", "users", "user-2"]]

// 按前缀搜索
const configs = await store.search(["app", "config"]);
console.log("配置项:", configs);
// [
//   { namespace: ["app", "config"], key: "version", value: "1.0.0" },
//   { namespace: ["app", "config"], key: "maxRetries", value: 3 }
// ]
```

## 持久化存储

### SQLite 存储

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";
import { SqliteStore } from "@langchain/langgraph-checkpoint-sqlite";

// 创建 SQLite 存储
const store = new SqliteStore({
  db: "./store.db",
});

const State = Annotation.Root({
  key: Annotation<string>,
  value: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("save", async (state) => {
    await store.put(["data"], state.key, state.value);
    return {};
  })
  .addNode("load", async (state) => {
    const item = await store.get(["data"], state.key);
    return { value: item?.value || "未找到" };
  })
  .addEdge("__start__", "save")
  .addEdge("save", "load")
  .addEdge("load", "__end__")
  .compile();

// 使用图
await graph.invoke({ key: "test", value: "hello" });
```

### Redis 存储

```typescript
import { BaseStore, Item } from "@langchain/langgraph";
import { createClient, RedisClientType } from "redis";

class RedisStore extends BaseStore {
  private client: RedisClientType;
  private prefix: string;

  constructor(options: { url?: string; prefix?: string }) {
    super();
    this.client = createClient({ url: options.url || "redis://localhost:6379" });
    this.prefix = options.prefix || "langgraph:store:";
  }

  async connect(): Promise<void> {
    await this.client.connect();
  }

  async disconnect(): Promise<void> {
    await this.client.disconnect();
  }

  async get(namespace: string[], key: string): Promise<Item | undefined> {
    const redisKey = this.getKey(namespace, key);
    const data = await this.client.get(redisKey);
    if (!data) return undefined;
    return JSON.parse(data);
  }

  async mget(keys: [string[], string][]): Promise<(Item | undefined)[]> {
    const redisKeys = keys.map(([ns, key]) => this.getKey(ns, key));
    const results = await this.client.mGet(redisKeys);
    return results.map((data) => (data ? JSON.parse(data) : undefined));
  }

  async put(namespace: string[], key: string, value: any): Promise<void> {
    const redisKey = this.getKey(namespace, key);
    await this.client.set(redisKey, JSON.stringify({ namespace, key, value }));
  }

  async mput(items: [string[], string, any][]): Promise<void> {
    const pipeline = this.client.multi();
    for (const [namespace, key, value] of items) {
      const redisKey = this.getKey(namespace, key);
      pipeline.set(redisKey, JSON.stringify({ namespace, key, value }));
    }
    await pipeline.exec();
  }

  async delete(namespace: string[], key: string): Promise<void> {
    const redisKey = this.getKey(namespace, key);
    await this.client.del(redisKey);
  }

  async search(
    namespace: string[],
    options?: {
      query?: string;
      filter?: Record<string, any>;
      limit?: number;
      offset?: number;
    }
  ): Promise<Item[]> {
    const pattern = this.getKey(namespace, "*");
    const keys = await this.client.keys(pattern);
    const limit = options?.limit || 100;
    const offset = options?.offset || 0;

    const slicedKeys = keys.slice(offset, offset + limit);
    const results: Item[] = [];

    for (const key of slicedKeys) {
      const data = await this.client.get(key);
      if (data) {
        const item = JSON.parse(data);
        if (!options?.query || JSON.stringify(item.value).includes(options.query)) {
          results.push(item);
        }
      }
    }

    return results;
  }

  async listNamespaces(options?: {
    prefix?: string[];
    suffix?: string[];
    maxDepth?: number;
    limit?: number;
    offset?: number;
  }): Promise<string[][]> {
    const pattern = this.prefix + "*";
    const keys = await this.client.keys(pattern);
    const namespaces = new Set<string>();

    for (const key of keys) {
      const parts = key.replace(this.prefix, "").split(":");
      namespaces.add(parts.slice(0, -1).join(":"));
    }

    return Array.from(namespaces).map((ns) => ns.split(":"));
  }

  private getKey(namespace: string[], key: string): string {
    return `${this.prefix}${namespace.join(":")}:${key}`;
  }
}

// 使用 Redis 存储
async function main() {
  const store = new RedisStore({
    url: "redis://localhost:6379",
    prefix: "myapp:store:",
  });

  await store.connect();

  // 使用存储
  await store.put(["users", "user-1"], "profile", {
    name: "张三",
    email: "zhangsan@example.com",
  });

  const profile = await store.get(["users", "user-1"], "profile");
  console.log("用户资料:", profile);

  await store.disconnect();
}

main().catch(console.error);
```

## 语义搜索

### 向量存储集成

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";

// 创建带语义搜索的存储
class SemanticStore extends InMemoryStore {
  private vectorStore: MemoryVectorStore;
  private embeddings: OpenAIEmbeddings;

  constructor() {
    super();
    this.embeddings = new OpenAIEmbeddings();
    this.vectorStore = new MemoryVectorStore(this.embeddings);
  }

  async putWithEmbedding(
    namespace: string[],
    key: string,
    value: any,
    text: string
  ): Promise<void> {
    // 保存到普通存储
    await this.put(namespace, key, value);

    // 保存到向量存储
    await this.vectorStore.addDocuments([
      {
        pageContent: text,
        metadata: { namespace, key },
      },
    ]);
  }

  async semanticSearch(
    query: string,
    options?: {
      namespace?: string[];
      limit?: number;
      filter?: Record<string, any>;
    }
  ): Promise<any[]> {
    const results = await this.vectorStore.similaritySearchWithScore(
      query,
      options?.limit || 5
    );

    return results.map(([doc, score]) => ({
      namespace: doc.metadata.namespace,
      key: doc.metadata.key,
      score,
      text: doc.pageContent,
    }));
  }
}

// 使用语义搜索存储
const State = Annotation.Root({
  query: Annotation<string>,
  results: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
});

const store = new SemanticStore();

// 预先填充数据
await store.putWithEmbedding(
  ["knowledge"],
  "doc-1",
  { title: "LangGraph 简介" },
  "LangGraph 是一个用于构建有状态、多参与者应用程序的框架"
);

await store.putWithEmbedding(
  ["knowledge"],
  "doc-2",
  { title: "状态管理" },
  "LangGraph 使用检查点机制来持久化图的状态"
);

const graph = new StateGraph(State)
  .addNode("search", async (state) => {
    const results = await store.semanticSearch(state.query, {
      limit: 3,
    });
    return { results };
  })
  .addEdge("__start__", "search")
  .addEdge("search", "__end__")
  .compile();

// 语义搜索
const result = await graph.invoke({
  query: "如何持久化状态？",
});

console.log("搜索结果:", result.results);
// [
//   {
//     namespace: ["knowledge"],
//     key: "doc-2",
//     score: 0.85,
//     text: "LangGraph 使用检查点机制来持久化图的状态"
//   },
//   ...
// ]
```

### 结合 RAG 使用

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";
import { ChatOpenAI, OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";

interface Document {
  pageContent: string;
  metadata: Record<string, any>;
}

const State = Annotation.Root({
  question: Annotation<string>,
  context: Annotation<string[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  answer: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

// 初始化向量存储
const embeddings = new OpenAIEmbeddings();
const vectorStore = new MemoryVectorStore(embeddings);

// 预填充知识库
const documents: Document[] = [
  {
    pageContent: "LangGraph 是 LangChain 的扩展，用于构建复杂的 Agent 应用",
    metadata: { source: "docs/intro.md" },
  },
  {
    pageContent: "检查点机制允许暂停和恢复图的执行",
    metadata: { source: "docs/checkpoint.md" },
  },
  {
    pageContent: "中断功能实现了人在回路的工作流",
    metadata: { source: "docs/interrupt.md" },
  },
];

await vectorStore.addDocuments(documents);

const model = new ChatOpenAI({ modelName: "gpt-4" });

const graph = new StateGraph(State)
  .addNode("retrieve", async (state) => {
    const results = await vectorStore.similaritySearch(state.question, 3);
    return {
      context: results.map((doc) => doc.pageContent),
    };
  })
  .addNode("generate", async (state) => {
    const prompt = `基于以下上下文回答问题：

上下文：
${state.context.join("\n")}

问题：${state.question}

回答：`;

    const response = await model.invoke(prompt);
    return { answer: response.content as string };
  })
  .addEdge("__start__", "retrieve")
  .addEdge("retrieve", "generate")
  .addEdge("generate", "__end__")
  .compile();

// 使用 RAG
const result = await graph.invoke({
  question: "LangGraph 如何处理状态持久化？",
});

console.log("回答:", result.answer);
```

### 存储工厂模式

```typescript
import { BaseStore, InMemoryStore } from "@langchain/langgraph";

// 存储工厂
class StoreFactory {
  static create(type: "memory" | "sqlite" | "postgres", options?: any): BaseStore {
    switch (type) {
      case "memory":
        return new InMemoryStore();

      case "sqlite":
        // 需要导入 SqliteStore
        // return new SqliteStore(options);
        throw new Error("需要安装 @langchain/langgraph-checkpoint-sqlite");

      case "postgres":
        // 需要导入 PostgresStore
        // return new PostgresStore(options);
        throw new Error("需要安装 @langchain/langgraph-checkpoint-postgres");

      default:
        throw new Error(`不支持的存储类型: ${type}`);
    }
  }
}

// 使用工厂创建存储
const store = StoreFactory.create("memory");

// 配置化创建
interface StoreConfig {
  type: "memory" | "sqlite" | "postgres";
  options?: Record<string, any>;
}

function createStoreFromConfig(config: StoreConfig): BaseStore {
  return StoreFactory.create(config.type, config.options);
}

// 示例配置
const config: StoreConfig = {
  type: "memory",
};

const store2 = createStoreFromConfig(config);
```

> **注意**：`InMemoryStore` 适用于开发和测试环境，生产环境建议使用持久化存储。

> **建议**：对于需要语义搜索的场景，可以结合向量存储（如 Pinecone、Weaviate）使用。

> **提示**：使用命名空间组织数据，可以更好地管理和查询存储中的数据。
