## 短期记忆

短期记忆（Short-term Memory）是指在单次对话或会话中使用的记忆，通常存储在图的状态中。

### 基于状态的短期记忆

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

// 定义带有对话历史的状态
const ConversationState = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  context: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

// 创建带记忆的图
const graph = new StateGraph(ConversationState)
  .addNode("chat", async (state) => {
    // 使用对话历史作为上下文
    const history = state.messages
      .map((msg) => `${msg.role}: ${msg.content}`)
      .join("\n");

    // 这里可以调用 LLM
    const response = `基于历史回复: ${history.slice(-100)}`;

    return {
      messages: [{ role: "assistant", content: response }],
      context: history,
    };
  })
  .addEdge("__start__", "chat")
  .addEdge("chat", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 使用会话记忆
const config = { configurable: { thread_id: "conversation-1" } };

// 第一轮对话
await graph.invoke(
  { messages: [{ role: "user", content: "你好！" }] },
  config
);

// 第二轮对话（自动携带历史）
const result = await graph.invoke(
  { messages: [{ role: "user", content: "我叫张三" }] },
  config
);

console.log("对话历史:", result.messages);
```

### 滑动窗口记忆

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

// 带滑动窗口的状态
const WindowedState = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => {
      const combined = [...prev, ...curr];
      // 只保留最近的 10 条消息
      const maxMessages = 10;
      return combined.slice(-maxMessages);
    },
    default: () => [],
  }),
});

const graph = new StateGraph(WindowedState)
  .addNode("chat", async (state) => {
    return {
      messages: [{
        role: "assistant",
        content: `收到 ${state.messages.length} 条消息`,
      }],
    };
  })
  .addEdge("__start__", "chat")
  .addEdge("chat", "__end__")
  .compile({ checkpointer: new MemorySaver() });

const config = { configurable: { thread_id: "windowed-chat" } };

// 模拟多轮对话
for (let i = 0; i < 15; i++) {
  await graph.invoke(
    { messages: [{ role: "user", content: `消息 ${i + 1}` }] },
    config
  );
}

// 获取最终状态
const finalState = await graph.getState(config);
console.log("消息数量:", finalState.values.messages.length);  // 最多 10 条
```

### Token 限制记忆

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";

// 估算 token 数量（简化版）
function estimateTokens(text: string): number {
  // 粗略估算：1 个中文字符约 2 个 token
  return Math.ceil(text.length * 1.5);
}

// 带 Token 限制的状态
const TokenLimitedState = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => {
      const combined = [...prev, ...curr];

      // 计算总 token 数
      let totalTokens = 0;
      const result: any[] = [];

      // 从最新的消息开始，保留尽可能多的消息
      for (let i = combined.length - 1; i >= 0; i--) {
        const msg = combined[i];
        const tokens = estimateTokens(JSON.stringify(msg));

        if (totalTokens + tokens > 4000) {
          break;  // 超过 token 限制
        }

        totalTokens += tokens;
        result.unshift(msg);
      }

      return result;
    },
    default: () => [],
  }),
});

const graph = new StateGraph(TokenLimitedState)
  .addNode("chat", async (state) => {
    return {
      messages: [{
        role: "assistant",
        content: "已处理",
      }],
    };
  })
  .addEdge("__start__", "chat")
  .addEdge("chat", "__end__")
  .compile({ checkpointer: new MemorySaver() });
```

## 长期记忆

长期记忆（Long-term Memory）是指跨会话持久化的记忆，通常使用存储（Store）来实现。

### 使用 Store 实现长期记忆

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";

// 创建存储
const store = new InMemoryStore();

const State = Annotation.Root({
  userId: Annotation<string>,
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  userPreferences: Annotation<Record<string, any>>({
    reducer: (prev, curr) => ({ ...prev, ...curr }),
    default: () => ({}),
  }),
});

const graph = new StateGraph(State)
  // 加载用户记忆
  .addNode("loadMemory", async (state) => {
    const prefs = await store.get(
      ["users", state.userId],
      "preferences"
    );
    const history = await store.get(
      ["users", state.userId],
      "history"
    );

    return {
      userPreferences: prefs?.value || {},
      messages: history?.value || [],
    };
  })
  // 处理对话
  .addNode("process", async (state) => {
    // 使用用户偏好定制回复
    const language = state.userPreferences.language || "zh-CN";
    const style = state.userPreferences.style || "formal";

    return {
      messages: [{
        role: "assistant",
        content: `使用 ${language}，${style} 风格回复`,
      }],
    };
  })
  // 保存记忆
  .addNode("saveMemory", async (state) => {
    // 保存对话历史
    await store.put(
      ["users", state.userId],
      "history",
      state.messages.slice(-50)  // 只保留最近 50 条
    );
    return {};
  })
  .addEdge("__start__", "loadMemory")
  .addEdge("loadMemory", "process")
  .addEdge("process", "saveMemory")
  .addEdge("saveMemory", "__end__")
  .compile();

// 使用长期记忆
const config = { configurable: { thread_id: "session-1" } };

// 第一次对话
await graph.invoke(
  {
    userId: "user-123",
    messages: [{ role: "user", content: "我喜欢简洁的回复" }],
    userPreferences: { style: "concise" },
  },
  config
);

// 第二次对话（记忆会自动加载）
const result = await graph.invoke(
  {
    userId: "user-123",
    messages: [{ role: "user", content: "今天天气怎么样？" }],
  },
  config
);

console.log("用户偏好:", result.userPreferences);
console.log("对话历史:", result.messages);
```

### 实体记忆

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";

interface EntityMemory {
  name: string;
  type: string;
  attributes: Record<string, any>;
  lastMentioned: number;
}

// 实体记忆管理器
class EntityMemoryManager {
  private store: InMemoryStore;

  constructor(store: InMemoryStore) {
    this.store = store;
  }

  async addEntity(
    userId: string,
    entity: EntityMemory
  ): Promise<void> {
    await this.store.put(
      ["entities", userId],
      entity.name,
      {
        ...entity,
        lastMentioned: Date.now(),
      }
    );
  }

  async getEntity(
    userId: string,
    entityName: string
  ): Promise<EntityMemory | undefined> {
    const item = await this.store.get(
      ["entities", userId],
      entityName
    );
    return item?.value;
  }

  async getAllEntities(userId: string): Promise<EntityMemory[]> {
    const items = await this.store.search(["entities", userId]);
    return items.map((item) => item.value);
  }

  async updateEntity(
    userId: string,
    entityName: string,
    updates: Partial<EntityMemory>
  ): Promise<void> {
    const existing = await this.getEntity(userId, entityName);
    if (existing) {
      await this.addEntity(userId, { ...existing, ...updates });
    }
  }
}

// 使用实体记忆
const store = new InMemoryStore();
const entityMemory = new EntityMemoryManager(store);

const State = Annotation.Root({
  userId: Annotation<string>,
  input: Annotation<string>,
  entities: Annotation<EntityMemory[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  response: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("extractEntities", async (state) => {
    // 简化的实体提取（实际应使用 NER 模型）
    const entities: EntityMemory[] = [];

    if (state.input.includes("张三")) {
      entities.push({
        name: "张三",
        type: "person",
        attributes: { role: "朋友" },
        lastMentioned: Date.now(),
      });
    }

    if (state.input.includes("公司")) {
      entities.push({
        name: "公司",
        type: "organization",
        attributes: { type: "工作单位" },
        lastMentioned: Date.now(),
      });
    }

    return { entities };
  })
  .addNode("loadEntityMemory", async (state) => {
    const loadedEntities: EntityMemory[] = [];

    for (const entity of state.entities) {
      const existing = await entityMemory.getEntity(
        state.userId,
        entity.name
      );
      if (existing) {
        loadedEntities.push(existing);
      } else {
        await entityMemory.addEntity(state.userId, entity);
        loadedEntities.push(entity);
      }
    }

    return { entities: loadedEntities };
  })
  .addNode("generateResponse", async (state) => {
    const entityInfo = state.entities
      .map((e) => `${e.name}(${e.type})`)
      .join("、");

    return {
      response: `识别到实体: ${entityInfo}`,
    };
  })
  .addEdge("__start__", "extractEntities")
  .addEdge("extractEntities", "loadEntityMemory")
  .addEdge("loadEntityMemory", "generateResponse")
  .addEdge("generateResponse", "__end__")
  .compile();

// 使用实体记忆
const config = { configurable: { thread_id: "entity-session" } };

await graph.invoke({
  userId: "user-1",
  input: "我和张三去了公司",
});
```

## 对话历史管理

### 压缩对话历史

```typescript
import { StateGraph, Annotation, MemorySaver } from "@langchain/langgraph";
import { ChatOpenAI } from "@langchain/openai";

const model = new ChatOpenAI({ modelName: "gpt-4" });

const State = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  summary: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

// 压缩对话历史
async function summarizeConversation(messages: any[]): Promise<string> {
  const conversationText = messages
    .map((msg) => `${msg.role}: ${msg.content}`)
    .join("\n");

  const response = await model.invoke(
    `请用一两句话总结以下对话的要点：\n${conversationText}`
  );

  return response.content as string;
}

const graph = new StateGraph(State)
  .addNode("checkLength", async (state) => {
    // 如果消息超过 20 条，触发压缩
    if (state.messages.length > 20) {
      const summary = await summarizeConversation(
        state.messages.slice(0, -5)  // 保留最近 5 条
      );
      return {
        summary,
        messages: state.messages.slice(-5),  // 只保留最近 5 条
      };
    }
    return {};
  })
  .addNode("chat", async (state) => {
    const context = state.summary
      ? `对话摘要: ${state.summary}\n\n`
      : "";

    const history = state.messages
      .map((msg) => `${msg.role}: ${msg.content}`)
      .join("\n");

    const response = await model.invoke(
      `${context}对话历史:\n${history}\n\n请回复。`
    );

    return {
      messages: [{ role: "assistant", content: response.content }],
    };
  })
  .addEdge("__start__", "checkLength")
  .addEdge("checkLength", "chat")
  .addEdge("chat", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 使用压缩记忆
const config = { configurable: { thread_id: "compressed-chat" } };

// 模拟长对话
for (let i = 0; i < 25; i++) {
  await graph.invoke(
    { messages: [{ role: "user", content: `消息 ${i + 1}` }] },
    config
  );
}

const finalState = await graph.getState(config);
console.log("摘要:", finalState.values.summary);
console.log("消息数量:", finalState.values.messages.length);
```

### 分层记忆管理

```typescript
import { StateGraph, Annotation, InMemoryStore, MemorySaver } from "@langchain/langgraph";

// 分层记忆结构
interface MemoryLayers {
  working: any[];      // 工作记忆（当前对话）
  episodic: any[];     // 情节记忆（重要事件）
  semantic: any;       // 语义记忆（知识和偏好）
}

const store = new InMemoryStore();

const State = Annotation.Root({
  userId: Annotation<string>,
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  importantEvents: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  userKnowledge: Annotation<Record<string, any>>({
    reducer: (prev, curr) => ({ ...prev, ...curr }),
    default: () => ({}),
  }),
});

const graph = new StateGraph(State)
  // 加载所有记忆层
  .addNode("loadMemories", async (state) => {
    // 加载语义记忆
    const knowledge = await store.get(
      ["users", state.userId],
      "knowledge"
    );

    // 加载情节记忆
    const episodic = await store.get(
      ["users", state.userId],
      "episodic"
    );

    return {
      userKnowledge: knowledge?.value || {},
      importantEvents: episodic?.value || [],
    };
  })
  // 处理对话
  .addNode("process", async (state) => {
    // 检测重要事件
    const isImportant = state.messages.some(
      (msg) =>
        msg.content.includes("重要") ||
        msg.content.includes("记住")
    );

    if (isImportant) {
      return {
        importantEvents: [
          ...state.importantEvents,
          {
            content: state.messages[state.messages.length - 1].content,
            timestamp: Date.now(),
          },
        ],
      };
    }

    return {};
  })
  // 保存记忆
  .addNode("saveMemories", async (state) => {
    // 保存语义记忆
    await store.put(
      ["users", state.userId],
      "knowledge",
      state.userKnowledge
    );

    // 保存情节记忆
    await store.put(
      ["users", state.userId],
      "episodic",
      state.importantEvents.slice(-100)
    );

    return {};
  })
  .addEdge("__start__", "loadMemories")
  .addEdge("loadMemories", "process")
  .addEdge("process", "saveMemories")
  .addEdge("saveMemories", "__end__")
  .compile({ checkpointer: new MemorySaver() });

// 使用分层记忆
const config = { configurable: { thread_id: "layered-memory" } };

await graph.invoke({
  userId: "user-1",
  messages: [{ role: "user", content: "请记住我喜欢蓝色" }],
});
```

## 记忆检索

### 语义检索记忆

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";

// 记忆检索器
class MemoryRetriever {
  private vectorStore: MemoryVectorStore;
  private store: InMemoryStore;

  constructor(store: InMemoryStore) {
    this.store = store;
    this.vectorStore = new MemoryVectorStore(new OpenAIEmbeddings());
  }

  async addMemory(
    userId: string,
    memoryId: string,
    content: string,
    metadata: Record<string, any> = {}
  ): Promise<void> {
    // 保存到存储
    await this.store.put(
      ["memories", userId],
      memoryId,
      { content, metadata, timestamp: Date.now() }
    );

    // 添加到向量索引
    await this.vectorStore.addDocuments([
      {
        pageContent: content,
        metadata: { userId, memoryId, ...metadata },
      },
    ]);
  }

  async searchMemories(
    userId: string,
    query: string,
    limit: number = 5
  ): Promise<any[]> {
    const results = await this.vectorStore.similaritySearchWithScore(
      query,
      limit * 2  // 获取更多结果以便过滤
    );

    return results
      .filter(([doc]) => doc.metadata.userId === userId)
      .slice(0, limit)
      .map(([doc, score]) => ({
        content: doc.pageContent,
        score,
        metadata: doc.metadata,
      }));
  }
}

// 使用记忆检索
const store = new InMemoryStore();
const memoryRetriever = new MemoryRetriever(store);

const State = Annotation.Root({
  userId: Annotation<string>,
  query: Annotation<string>,
  relevantMemories: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  response: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("retrieveMemories", async (state) => {
    const memories = await memoryRetriever.searchMemories(
      state.userId,
      state.query,
      3
    );
    return { relevantMemories: memories };
  })
  .addNode("generate", async (state) => {
    const memoryContext = state.relevantMemories
      .map((m) => `- ${m.content}`)
      .join("\n");

    return {
      response: `基于记忆回复。相关记忆:\n${memoryContext}`,
    };
  })
  .addEdge("__start__", "retrieveMemories")
  .addEdge("retrieveMemories", "generate")
  .addEdge("generate", "__end__")
  .compile();

// 预填充记忆
await memoryRetriever.addMemory(
  "user-1",
  "preference-1",
  "用户喜欢简洁的回复风格"
);

await memoryRetriever.addMemory(
  "user-1",
  "preference-2",
  "用户是程序员，熟悉 TypeScript"
);

// 使用记忆检索
const result = await graph.invoke({
  userId: "user-1",
  query: "我的偏好是什么？",
});

console.log("回复:", result.response);
console.log("相关记忆:", result.relevantMemories);
```

## 记忆更新

### 自动记忆更新

```typescript
import { StateGraph, Annotation, InMemoryStore } from "@langchain/langgraph";
import { ChatOpenAI } from "@langchain/openai";

const model = new ChatOpenAI({ modelName: "gpt-4" });
const store = new InMemoryStore();

const State = Annotation.Root({
  userId: Annotation<string>,
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
  extractedInfo: Annotation<Record<string, any>>({
    reducer: (prev, curr) => ({ ...prev, ...curr }),
    default: () => ({}),
  }),
});

const graph = new StateGraph(State)
  // 提取用户信息
  .addNode("extractInfo", async (state) => {
    const lastMessage = state.messages[state.messages.length - 1];

    const response = await model.invoke(
      `从以下消息中提取用户信息（如姓名、偏好、工作等），以 JSON 格式返回：
      ${lastMessage.content}
      
      如果没有新信息，返回空对象 {}`
    );

    try {
      const info = JSON.parse(response.content as string);
      return { extractedInfo: info };
    } catch {
      return { extractedInfo: {} };
    }
  })
  // 更新记忆
  .addNode("updateMemory", async (state) => {
    if (Object.keys(state.extractedInfo).length > 0) {
      // 加载现有记忆
      const existing = await store.get(
        ["users", state.userId],
        "profile"
      );

      const updatedProfile = {
        ...(existing?.value || {}),
        ...state.extractedInfo,
        lastUpdated: Date.now(),
      };

      // 保存更新后的记忆
      await store.put(
        ["users", state.userId],
        "profile",
        updatedProfile
      );
    }

    return {};
  })
  .addEdge("__start__", "extractInfo")
  .addEdge("extractInfo", "updateMemory")
  .addEdge("updateMemory", "__end__")
  .compile();

// 使用自动记忆更新
const config = { configurable: { thread_id: "auto-memory" } };

await graph.invoke({
  userId: "user-1",
  messages: [{ role: "user", content: "我叫张三，是一名前端工程师" }],
});

// 检查更新的记忆
const profile = await store.get(["users", "user-1"], "profile");
console.log("用户资料:", profile?.value);
// { name: "张三", job: "前端工程师", lastUpdated: ... }
```

### 记忆合并策略

```typescript
import { InMemoryStore } from "@langchain/langgraph";

interface MemoryMerger {
  merge(existing: any, incoming: any): any;
}

// 不同的合并策略
const mergeStrategies: Record<string, MemoryMerger> = {
  // 覆盖策略
  overwrite: {
    merge: (_, incoming) => incoming,
  },

  // 追加策略（数组）
  append: {
    merge: (existing, incoming) => {
      if (Array.isArray(existing)) {
        return [...existing, ...incoming];
      }
      return incoming;
    },
  },

  // 深度合并策略（对象）
  deepMerge: {
    merge: (existing, incoming) => {
      if (typeof existing === "object" && typeof incoming === "object") {
        return { ...existing, ...incoming };
      }
      return incoming;
    },
  },

  // 保留最新策略
  keepLatest: {
    merge: (existing, incoming) => {
      if (existing.timestamp > incoming.timestamp) {
        return existing;
      }
      return incoming;
    },
  },
};

// 使用合并策略的记忆管理器
class MemoryManager {
  private store: InMemoryStore;
  private strategy: MemoryMerger;

  constructor(store: InMemoryStore, strategy: MemoryMerger) {
    this.store = store;
    this.strategy = strategy;
  }

  async updateMemory(
    namespace: string[],
    key: string,
    value: any
  ): Promise<void> {
    const existing = await this.store.get(namespace, key);
    const merged = existing
      ? this.strategy.merge(existing.value, value)
      : value;

    await this.store.put(namespace, key, merged);
  }
}

// 使用示例
const store = new InMemoryStore();
const manager = new MemoryManager(store, mergeStrategies.deepMerge);

// 更新记忆
await manager.updateMemory(["users", "user-1"], "preferences", {
  theme: "dark",
});

await manager.updateMemory(["users", "user-1"], "preferences", {
  language: "zh-CN",
});

// 最终记忆
const final = await store.get(["users", "user-1"], "preferences");
console.log("最终偏好:", final?.value);
// { theme: "dark", language: "zh-CN" }
```

> **注意**：短期记忆存储在图的状态中，会随检查点持久化；长期记忆需要使用 Store 单独管理。

> **建议**：对于聊天应用，建议结合短期记忆（对话历史）和长期记忆（用户偏好）使用。

> **提示**：定期压缩对话历史可以避免 token 超限，同时保留重要信息。
