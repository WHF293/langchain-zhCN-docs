## 项目结构

良好的项目结构是构建可维护 LangGraph 应用的基础。

### 推荐的项目结构

```
my-langgraph-app/
├── src/
│   ├── graphs/              # 图定义
│   │   ├── index.ts         # 主图
│   │   ├── workflow.ts      # 工作流图
│   │   └── subgraphs/       # 子图
│   │       ├── processing.ts
│   │       └── validation.ts
│   │
│   ├── nodes/               # 节点实现
│   │   ├── index.ts         # 节点导出
│   │   ├── process.ts       # 处理节点
│   │   ├── validate.ts      # 验证节点
│   │   └── transform.ts     # 转换节点
│   │
│   ├── state/               # 状态定义
│   │   ├── index.ts         # 状态导出
│   │   └── types.ts         # 类型定义
│   │
│   ├── tools/               # 工具实现
│   │   ├── index.ts         # 工具导出
│   │   ├── search.ts        # 搜索工具
│   │   └── database.ts      # 数据库工具
│   │
│   ├── utils/               # 工具函数
│   │   ├── logger.ts        # 日志工具
│   │   ├── config.ts        # 配置管理
│   │   └── helpers.ts       # 辅助函数
│   │
│   └── index.ts             # 应用入口
│
├── tests/                   # 测试文件
│   ├── unit/                # 单元测试
│   │   ├── nodes/
```

## 模块划分

### 节点模块

```typescript
// src/nodes/process.ts
import { Annotation } from "@langchain/langgraph";

// 定义节点输入输出类型
export const ProcessNodeInput = Annotation.Root({
  input: Annotation<string>,
  options: Annotation<Record<string, any>>,
});

export const ProcessNodeOutput = Annotation.Root({
  result: Annotation<any>,
  metadata: Annotation<Record<string, any>>,
});

// 节点实现
export async function processNode(
  state: typeof ProcessNodeInput.State
): Promise<typeof ProcessNodeOutput.State> {
  const { input, options } = state;

  // 处理逻辑
  const result = await process(input, options);

  return {
    result,
    metadata: {
      processedAt: new Date().toISOString(),
      node: "processNode",
    },
  };
}
```

### 状态模块

```typescript
// src/state/types.ts
import { Annotation } from "@langchain/langgraph";

// 主状态定义
export const MainStateAnnotation = Annotation.Root({
  // 输入数据
  input: Annotation<string>,
  userId: Annotation<string>,

  // 处理状态
  status: Annotation<"pending" | "processing" | "completed" | "failed">,
  currentStep: Annotation<string>,

  // 中间结果
  intermediateResults: Annotation<any[]>({
    reducer: (prev, next) => [...prev, ...next],
    default: () => [],
  }),

  // 输出
  output: Annotation<string>,

  // 错误信息
  errors: Annotation<string[]>({
    reducer: (prev, next) => [...prev, ...next],
    default: () => [],
  }),

  // 元数据
  metadata: Annotation<Record<string, any>>({
    reducer: (prev, next) => ({ ...prev, ...next }),
    default: () => ({}),
  }),
});

// 状态类型导出
export type MainState = typeof MainStateAnnotation.State;
```

### 工具模块

```typescript
// src/tools/search.ts
import { tool } from "@langchain/core/tools";
import { z } from "zod";

// 搜索工具定义
export const searchTool = tool(
  async ({ query, limit }: { query: string; limit?: number }) => {
    // 实现搜索逻辑
    const results = await search(query, limit || 10);
    return JSON.stringify(results);
  },
  {
    name: "search",
    description: "搜索相关信息",
    schema: z.object({
      query: z.string().describe("搜索查询"),
      limit: z.number().optional().describe("结果数量限制"),
    }),
  }
);

// 工具集合导出
export const tools = [searchTool];
```

## 配置管理

### 配置文件结构

```typescript
// src/utils/config.ts
import { z } from "zod";

// 配置 schema
const ConfigSchema = z.object({
  // 应用配置
  app: z.object({
    name: z.string(),
    version: z.string(),
    environment: z.enum(["development", "staging", "production"]),
  }),

  // LLM 配置
  llm: z.object({
    provider: z.enum(["openai", "anthropic", "google"]),
    model: z.string(),
    temperature: z.number(),
    maxTokens: z.number(),
  }),

  // 数据库配置
  database: z.object({
    host: z.string(),
    port: z.number(),
    username: z.string(),
    password: z.string(),
    database: z.string(),
  }),

  // 缓存配置
  cache: z.object({
    enabled: z.boolean(),
    ttl: z.number(),
    maxSize: z.number(),
  }),

  // 日志配置
  logging: z.object({
    level: z.enum(["error", "warn", "info", "debug"]),
    file: z.string().optional(),
  }),
});

// 配置类型
export type Config = z.infer<typeof ConfigSchema>;

// 加载配置
export function loadConfig(): Config {
  const config = {
    app: {
      name: process.env.APP_NAME || "my-langgraph-app",
      version: process.env.APP_VERSION || "1.0.0",
      environment: process.env.NODE_ENV || "development",
    },
    llm: {
      provider: process.env.LLM_PROVIDER || "openai",
      model: process.env.LLM_MODEL || "gpt-4",
      temperature: parseFloat(process.env.LLM_TEMPERATURE || "0.7"),
      maxTokens: parseInt(process.env.LLM_MAX_TOKENS || "4096"),
    },
    database: {
      host: process.env.DB_HOST || "localhost",
      port: parseInt(process.env.DB_PORT || "5432"),
      username: process.env.DB_USERNAME || "user",
      password: process.env.DB_PASSWORD || "password",
      database: process.env.DB_NAME || "mydb",
    },
    cache: {
      enabled: process.env.CACHE_ENABLED === "true",
      ttl: parseInt(process.env.CACHE_TTL || "3600"),
      maxSize: parseInt(process.env.CACHE_MAX_SIZE || "1000"),
    },
    logging: {
      level: process.env.LOG_LEVEL || "info",
      file: process.env.LOG_FILE,
    },
  };

  return ConfigSchema.parse(config);
}

// 配置单例
let configInstance: Config | null = null;

export function getConfig(): Config {
  if (!configInstance) {
    configInstance = loadConfig();
  }
  return configInstance;
}
```

## 测试策略

### 单元测试

```typescript
// tests/unit/nodes/process.test.ts
import { processNode } from "../../../src/nodes/process";

describe("processNode", () => {
  it("应该正确处理输入", async () => {
    const state = {
      input: "test input",
      options: {},
    };

    const result = await processNode(state);

    expect(result.result).toBeDefined();
    expect(result.metadata.processedAt).toBeDefined();
  });

  it("应该处理错误情况", async () => {
    const state = {
      input: "",
      options: {},
    };

    await expect(processNode(state)).rejects.toThrow();
  });
});
```

### 集成测试

```typescript
// tests/integration/graph.test.ts
import { StateGraph, Annotation } from "@langchain/langgraph";
import { processNode, validateNode, transformNode } from "../../src/nodes";

describe("图集成测试", () => {
  let app: any;

  beforeAll(() => {
    const StateAnnotation = Annotation.Root({
      input: Annotation<string>,
      output: Annotation<string>,
    });

    const workflow = new StateGraph(StateAnnotation)
      .addNode("validate", validateNode)
      .addNode("process", processNode)
      .addNode("transform", transformNode)
      .addEdge("__start__", "validate")
      .addEdge("validate", "process")
      .addEdge("process", "transform")
      .addEdge("transform", "__end__");

    app = workflow.compile();
  });

  it("应该完成完整的处理流程", async () => {
    const result = await app.invoke({
      input: "test input",
    });

    expect(result.output).toBeDefined();
    expect(result.output).not.toBe("");
  });

  it("应该正确处理验证失败", async () => {
    const result = await app.invoke({
      input: "", // 无效输入
    });

    expect(result.errors).toContain("验证失败");
  });
});
```

### 端到端测试

```typescript
// tests/e2e/app.test.ts
import { createApp } from "../../src/index";

describe("应用端到端测试", () => {
  let app: any;

  beforeAll(async () => {
    app = await createApp();
  });

  it("应该处理完整的用户请求", async () => {
    const request = {
      userId: "user-123",
      input: "用户问题",
    };

    const response = await app.invoke(request);

    expect(response.success).toBe(true);
    expect(response.data).toBeDefined();
  });

  it("应该处理并发请求", async () => {
    const requests = Array.from({ length: 10 }, (_, i) => ({
      userId: `user-${i}`,
      input: `问题 ${i}`,
    }));

    const responses = await Promise.all(
      requests.map((req) => app.invoke(req))
    );

    responses.forEach((response) => {
      expect(response.success).toBe(true);
    });
  });
});
```

### 测试最佳实践

| 实践 | 说明 |
|------|------|
| **隔离测试** | 每个测试独立，不依赖外部状态 |
| **模拟依赖** | 使用 mock 隔离外部依赖 |
| **覆盖边界** | 测试正常、异常、边界情况 |
| **测试命名** | 清晰描述测试意图 |
| **持续集成** | 自动运行测试 |

> **注意**：在项目初期就建立良好的测试习惯，可以大大减少后期的维护成本。

> **提示**：使用 TypeScript 的类型系统可以在编译时捕获很多错误，结合运行时测试可以提高代码质量。

> **建议**：为关键业务逻辑编写详细的测试用例，确保核心功能的正确性。
