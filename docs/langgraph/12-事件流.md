## 事件类型

LangGraph 提供了丰富的事件类型，用于监控和调试图的执行过程。

### 核心事件类型

| 事件类型 | 说明 | 触发时机 |
|---------|------|---------|
| `on_chain_start` | 链开始执行 | 节点开始时 |
| `on_chain_end` | 链执行完成 | 节点结束时 |
| `on_chain_stream` | 链流式输出 | 流式数据时 |
| `on_tool_start` | 工具开始执行 | 工具调用时 |
| `on_tool_end` | 工具执行完成 | 工具返回时 |
| `on_chat_model_start` | 模型开始生成 | LLM 调用时 |
| `on_chat_model_stream` | 模型流式输出 | Token 生成时 |
| `on_chat_model_end` | 模型生成完成 | LLM 返回时 |
| `on_llm_start` | LLM 开始 | 旧版兼容 |
| `on_llm_end` | LLM 结束 | 旧版兼容 |

### 事件数据结构

```typescript
// 事件对象的基本结构
interface LangGraphEvent {
  event: string;           // 事件类型
  name: string;            // 事件名称（通常是节点名）
  run_id: string;          // 运行 ID
  tags: string[];          // 标签
  metadata: Record<string, any>;  // 元数据
  data: {
    input?: any;           // 输入数据
    output?: any;          // 输出数据
    chunk?: any;           // 流式数据块
  };
  timestamp: string;       // 时间戳
}
```

## 事件订阅

### 基本事件订阅

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

const State = Annotation.Root({
  input: Annotation<string>,
  output: Annotation<string>({
    default: () => "",
    reducer: (_, curr) => curr,
  }),
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    await new Promise((resolve) => setTimeout(resolve, 100));
    return { output: `处理: ${state.input}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();

// 订阅所有事件
const eventStream = graph.streamEvents(
  { input: "测试" },
  { version: "v2" }
);

for await (const event of eventStream) {
  console.log(`事件: ${event.event}`);
  console.log(`名称: ${event.name}`);
  console.log(`数据:`, event.data);
  console.log("---");
}
```

### 按节点订阅事件

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

const State = Annotation.Root({
  data: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("nodeA", async (state) => {
    return { data: `A: ${state.data}` };
  })
  .addNode("nodeB", async (state) => {
    return { data: `B: ${state.data}` };
  })
  .addNode("nodeC", async (state) => {
    return { data: `C: ${state.data}` };
  })
  .addEdge("__start__", "nodeA")
  .addEdge("nodeA", "nodeB")
  .addEdge("nodeB", "nodeC")
  .addEdge("nodeC", "__end__")
  .compile();

// 只订阅特定节点的事件
const eventStream = graph.streamEvents(
  { data: "输入" },
  {
    version: "v2",
    // 只关注 nodeB 的事件
    callbacks: [
      {
        handleChainStart: async (output, runId, parentRunId, tags, metadata, name) => {
          if (name === "nodeB") {
            console.log("nodeB 开始执行");
          }
        },
        handleChainEnd: async (output, runId, parentRunId, tags, name) => {
          if (name === "nodeB") {
            console.log("nodeB 执行完成:", output);
          }
        },
      },
    ],
  }
);

for await (const event of eventStream) {
  // 处理其他事件
}
```

## 事件过滤

### 按事件类型过滤

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

const State = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
});

const graph = new StateGraph(State)
  .addNode("chat", async (state) => {
    return {
      messages: [{ role: "assistant", content: "你好！" }],
    };
  })
  .addEdge("__start__", "chat")
  .addEdge("chat", "__end__")
  .compile();

const eventStream = graph.streamEvents(
  { messages: [{ role: "user", content: "你好" }] },
  { version: "v2" }
);

// 只处理特定类型的事件
for await (const event of eventStream) {
  switch (event.event) {
    case "on_chain_start":
      console.log("[开始]", event.name);
      break;
    case "on_chain_end":
      console.log("[结束]", event.name);
      break;
    // 忽略其他事件
  }
}
```

### 按条件过滤事件

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

interface EventFilter {
  eventTypes?: string[];
  nodeNames?: string[];
  includeMetadata?: boolean;
  minDuration?: number;
}

class EventFilterStream {
  private stream: AsyncIterable<any>;
  private filter: EventFilter;

  constructor(stream: AsyncIterable<any>, filter: EventFilter) {
    this.stream = stream;
    this.filter = filter;
  }

  async *[Symbol.asyncIterator]() {
    for await (const event of this.stream) {
      // 按事件类型过滤
      if (this.filter.eventTypes && !this.filter.eventTypes.includes(event.event)) {
        continue;
      }

      // 按节点名称过滤
      if (this.filter.nodeNames && !this.filter.nodeNames.includes(event.name)) {
        continue;
      }

      yield event;
    }
  }
}

// 使用示例
const State = Annotation.Root({
  data: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("fastNode", async (state) => {
    return { data: `快速: ${state.data}` };
  })
  .addNode("slowNode", async (state) => {
    await new Promise((resolve) => setTimeout(resolve, 500));
    return { data: `慢速: ${state.data}` };
  })
  .addEdge("__start__", "fastNode")
  .addEdge("fastNode", "slowNode")
  .addEdge("slowNode", "__end__")
  .compile();

const rawStream = graph.streamEvents(
  { data: "测试" },
  { version: "v2" }
);

// 创建过滤流
const filteredStream = new EventFilterStream(rawStream, {
  eventTypes: ["on_chain_start", "on_chain_end"],
  nodeNames: ["slowNode"],
});

for await (const event of filteredStream) {
  console.log("过滤后的事件:", event.event, event.name);
}
```

## 事件处理

### 事件收集器

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

interface CollectedEvents {
  starts: any[];
  ends: any[];
  streams: any[];
  errors: any[];
  metadata: {
    totalDuration: number;
    nodeTimings: Record<string, number>;
  };
}

class EventCollector {
  private events: CollectedEvents = {
    starts: [],
    ends: [],
    streams: [],
    errors: [],
    metadata: {
      totalDuration: 0,
      nodeTimings: {},
    },
  };

  private startTime: number = 0;

  async collect(stream: AsyncIterable<any>): Promise<CollectedEvents> {
    this.startTime = Date.now();

    for await (const event of stream) {
      switch (event.event) {
        case "on_chain_start":
          this.events.starts.push({
            name: event.name,
            timestamp: Date.now(),
            input: event.data?.input,
          });
          break;

        case "on_chain_end":
          const endTime = Date.now();
          const startEvent = this.events.starts.find(
            (e) => e.name === event.name
          );
          const duration = startEvent ? endTime - startEvent.timestamp : 0;

          this.events.ends.push({
            name: event.name,
            timestamp: endTime,
            output: event.data?.output,
            duration,
          });

          this.events.metadata.nodeTimings[event.name] = duration;
          break;

        case "on_chain_stream":
          this.events.streams.push({
            name: event.name,
            timestamp: Date.now(),
            chunk: event.data?.chunk,
          });
          break;

        default:
          if (event.event.includes("error")) {
            this.events.errors.push({
              event: event.event,
              name: event.name,
              timestamp: Date.now(),
              error: event.data,
            });
          }
      }
    }

    this.events.metadata.totalDuration = Date.now() - this.startTime;
    return this.events;
  }
}

// 使用事件收集器
const State = Annotation.Root({
  data: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    await new Promise((resolve) => setTimeout(resolve, 100));
    return { data: `处理: ${state.data}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();

const stream = graph.streamEvents(
  { data: "测试" },
  { version: "v2" }
);

const collector = new EventCollector();
const events = await collector.collect(stream);

console.log("收集的事件:", {
  开始事件数: events.starts.length,
  结束事件数: events.ends.length,
  流式事件数: events.streams.length,
  错误事件数: events.errors.length,
  总耗时: events.metadata.totalDuration,
  节点耗时: events.metadata.nodeTimings,
});
```

### 实时事件处理器

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";

interface EventHandlerConfig {
  onStart?: (name: string, input: any) => void;
  onEnd?: (name: string, output: any, duration: number) => void;
  onStream?: (name: string, chunk: any) => void;
  onError?: (name: string, error: any) => void;
  onToken?: (token: string) => void;
}

class RealtimeEventHandler {
  private config: EventHandlerConfig;
  private nodeStartTimes: Map<string, number> = new Map();

  constructor(config: EventHandlerConfig) {
    this.config = config;
  }

  async process(stream: AsyncIterable<any>): Promise<void> {
    for await (const event of stream) {
      switch (event.event) {
        case "on_chain_start":
          this.nodeStartTimes.set(event.name, Date.now());
          this.config.onStart?.(event.name, event.data?.input);
          break;

        case "on_chain_end":
          const startTime = this.nodeStartTimes.get(event.name) || Date.now();
          const duration = Date.now() - startTime;
          this.config.onEnd?.(event.name, event.data?.output, duration);
          this.nodeStartTimes.delete(event.name);
          break;

        case "on_chain_stream":
          this.config.onStream?.(event.name, event.data?.chunk);
          break;

        case "on_chat_model_stream":
          if (event.data?.chunk?.content) {
            this.config.onToken?.(event.data.chunk.content);
          }
          break;

        default:
          if (event.event.includes("error")) {
            this.config.onError?.(event.name, event.data);
          }
      }
    }
  }
}

// 使用实时事件处理器
const State = Annotation.Root({
  messages: Annotation<any[]>({
    reducer: (prev, curr) => [...prev, ...curr],
    default: () => [],
  }),
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    return {
      messages: [{ role: "assistant", content: "处理完成" }],
    };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();

const stream = graph.streamEvents(
  { messages: [{ role: "user", content: "你好" }] },
  { version: "v2" }
);

const handler = new RealtimeEventHandler({
  onStart: (name, input) => {
    console.log(`[${name}] 开始处理`, input);
  },
  onEnd: (name, output, duration) => {
    console.log(`[${name}] 处理完成 (${duration}ms)`, output);
  },
  onStream: (name, chunk) => {
    console.log(`[${name}] 流式数据:`, chunk);
  },
  onToken: (token) => {
    process.stdout.write(token);
  },
  onError: (name, error) => {
    console.error(`[${name}] 错误:`, error);
  },
});

await handler.process(stream);
```

### 事件持久化

```typescript
import { StateGraph, Annotation } from "@langchain/langgraph";
import * as fs from "fs/promises";

interface PersistedEvent {
  timestamp: number;
  event: string;
  name: string;
  data: any;
}

class EventPersister {
  private events: PersistedEvent[] = [];
  private filePath: string;

  constructor(filePath: string) {
    this.filePath = filePath;
  }

  async collectAndPersist(stream: AsyncIterable<any>): Promise<void> {
    for await (const event of stream) {
      const persistedEvent: PersistedEvent = {
        timestamp: Date.now(),
        event: event.event,
        name: event.name,
        data: event.data,
      };

      this.events.push(persistedEvent);

      // 实时写入文件
      await this.saveToFile();
    }
  }

  private async saveToFile(): Promise<void> {
    await fs.writeFile(
      this.filePath,
      JSON.stringify(this.events, null, 2),
      "utf-8"
    );
  }

  getEvents(): PersistedEvent[] {
    return [...this.events];
  }

  getEventsByType(eventType: string): PersistedEvent[] {
    return this.events.filter((e) => e.event === eventType);
  }

  getEventsByNode(nodeName: string): PersistedEvent[] {
    return this.events.filter((e) => e.name === nodeName);
  }

  getSummary(): Record<string, number> {
    const summary: Record<string, number> = {};
    for (const event of this.events) {
      summary[event.event] = (summary[event.event] || 0) + 1;
    }
    return summary;
  }
}

// 使用事件持久化器
const State = Annotation.Root({
  data: Annotation<string>,
});

const graph = new StateGraph(State)
  .addNode("process", async (state) => {
    return { data: `处理: ${state.data}` };
  })
  .addEdge("__start__", "process")
  .addEdge("process", "__end__")
  .compile();

const stream = graph.streamEvents(
  { data: "测试" },
  { version: "v2" }
);

const persister = new EventPersister("./events.json");
await persister.collectAndPersist(stream);

console.log("事件摘要:", persister.getSummary());
```

> **注意**：事件流是异步的，需要使用 `for await...of` 语法遍历。

> **建议**：在生产环境中，建议使用事件收集器来记录执行日志，便于问题排查和性能分析。

> **提示**：使用 `streamEvents` 时，可以通过 `version` 参数选择事件格式版本，推荐使用 `"v2"`。
