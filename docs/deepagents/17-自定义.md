# 第十七章 自定义（Customization）

> 来源: [Deep Agents 官方文档](https://docs.langchain.com/deep-agents)

---

Deep Agents 提供全面的自定义能力，涵盖模型、工具、系统提示、中间件、子代理、后端、沙箱、人在回路、技能、记忆、配置文件和结构化输出等各个方面。本章提供完整的自定义指南。

## 模型自定义

### 使用自定义模型

```typescript
import { Agent, CustomModel } from "deepagents";

// 创建自定义模型
const customModel = new CustomModel({
  // 模型名称
  name: "my-custom-model",
  // 调用函数
  invoke: async (messages, options) => {
    // 调用您的模型 API
    const response = await callMyModelAPI({
      messages,
      temperature: options.temperature,
      maxTokens: options.maxTokens,
    });
    
    return {
      content: response.text,
      usage: response.usage,
      toolCalls: response.toolCalls,
    };
  },
  // 流式调用函数
  stream: async function* (messages, options) {
    const stream = await streamMyModelAPI({
      messages,
      temperature: options.temperature,
    });
    
    for await (const chunk of stream) {
      yield {
        content: chunk.text,
        done: chunk.done,
      };
    }
  },
});

// 使用自定义模型
const agent = new Agent({
  model: customModel,
});
```

## 工具自定义

### 定义自定义工具

```typescript
import { Agent, Tool } from "deepagents";

// 方式一：使用 Tool.define()
const weatherTool = Tool.define({
  name: "get_weather",
  description: "获取指定城市的天气信息",
  parameters: {
    type: "object",
    properties: {
      city: {
        type: "string",
        description: "城市名称",
      },
      unit: {
        type: "string",
        enum: ["celsius", "fahrenheit"],
        description: "温度单位",
        default: "celsius",
      },
    },
    required: ["city"],
  },
  // 工具执行函数
  execute: async (params, context) => {
    const { city, unit = "celsius" } = params;
    
    // 调用天气 API
    const weather = await fetchWeatherAPI(city);
    
    return {
      city,
      temperature: unit === "celsius" 
        ? weather.temp_c 
        : weather.temp_f,
      unit,
      description: weather.condition.text,
    };
  },
});

// 方式二：使用函数定义
const calculatorTool = {
  name: "calculator",
  description: "执行数学计算",
  parameters: {
    type: "object",
    properties: {
      expression: {
        type: "string",
        description: "数学表达式",
      },
    },
    required: ["expression"],
  },
  execute: async (params) => {
    try {
      const result = eval(params.expression);
      return { result };
    } catch (error) {
      return { error: "计算失败" };
    }
  },
};

// 使用自定义工具
const agent = new Agent({
  model: "gpt-4",
  tools: [weatherTool, calculatorTool],
});
```

### 工具中间件

```typescript
import { Agent, ToolMiddleware } from "deepagents";

// 创建日志中间件
const loggingMiddleware: ToolMiddleware = {
  name: "logging",
  beforeExecute: async (tool, params, context) => {
    console.log(`调用工具: ${tool.name}`, params);
    return params;
  },
  afterExecute: async (tool, params, result, context) => {
    console.log(`工具结果: ${tool.name}`, result);
    return result;
  },
  onError: async (tool, params, error, context) => {
    console.error(`工具错误: ${tool.name}`, error);
    throw error;
  },
};

// 创建权限检查中间件
const permissionMiddleware: ToolMiddleware = {
  name: "permission",
  beforeExecute: async (tool, params, context) => {
    if (!context.permissions.canExecute(tool.name)) {
      throw new Error(`没有权限执行工具: ${tool.name}`);
    }
    return params;
  },
};

// 应用中间件
const agent = new Agent({
  model: "gpt-4",
  tools: [weatherTool, calculatorTool],
  toolMiddleware: [loggingMiddleware, permissionMiddleware],
});
```

```typescript
const promptBuilder = new SystemPromptBuilder({

      },
    },
  ],

  systemPrompt: promptBuilder,
```

## 中间件（默认栈）

Deep Agents 使用中间件栈处理请求和响应：

```typescript
import { Agent, Middleware, DefaultMiddlewareStack } from "deepagents";

// 默认中间件栈
const defaultStack = new DefaultMiddlewareStack([
  // 1. 请求验证
  Middleware.validate(),
  // 2. 权限检查
  Middleware.permissions(),
  // 3. 速率限制
  Middleware.rateLimit({ maxRequests: 100, window: 60 }),
  // 4. 日志记录
  Middleware.logging({ level: "info" }),
  // 5. 错误处理
  Middleware.errorHandler(),
  // 6. 重试逻辑
  Middleware.retry({ maxRetries: 3 }),
]);

// 使用默认中间件栈
const agent = new Agent({
  model: "gpt-4",
  middleware: defaultStack,
});
```

### 自定义中间件

```typescript
import { Agent, Middleware } from "deepagents";

// 创建自定义中间件
const customMiddleware: Middleware = {
  name: "custom-middleware",
  
  // 处理请求
  async processRequest(request, next) {
    // 预处理
    console.log("请求预处理:", request);
    
    // 调用下一个中间件
    const response = await next(request);
    
    // 后处理
    console.log("响应后处理:", response);
    
    return response;
  },
  
  // 处理错误
  async processError(error, request, next) {
    console.error("中间件错误:", error);
    
    // 尝试恢复
    if (error.code === "TIMEOUT") {
      return next(request); // 重试
    }
    
    throw error;
  },
};

// 注册中间件
const agent = new Agent({
  model: "gpt-4",
  middleware: [customMiddleware],
});
```

### 子代理通信

```typescript
// 子代理之间可以共享上下文
const context = {
  sharedData: {},
  
  // 设置共享数据
  set(key, value) {
    this.sharedData[key] = value;
  },
  
  // 获取共享数据

  },

    return analysis;
  },
});
```

## 后端自定义

### 自定义文件系统后端

```typescript
import { FilesystemBackend } from "deepagents";

// 创建自定义文件系统后端
class S3FilesystemBackend extends FilesystemBackend {
  constructor(config) {
    super(config);
    this.s3Client = new S3Client(config);
  }
  
  async read(path) {
    const params = {
      Bucket: this.config.bucket,
      Key: path,
    };
    
    const data = await this.s3Client.getObject(params);
    return data.Body.toString();
  }
  
  async write(path, content) {
    const params = {
      Bucket: this.config.bucket,
      Key: path,
      Body: content,
    };
    
    await this.s3Client.putObject(params);
  }
  
  async list(prefix) {
    const params = {
      Bucket: this.config.bucket,
      Prefix: prefix,
    };
    
    const data = await this.s3Client.listObjectsV2(params);
    return data.Contents.map(item => item.Key);
  }
}

// 使用自定义后端
const backend = new CompositeBackend({
  backends: {
    fs: new S3FilesystemBackend({
      bucket: "my-bucket",
      region: "us-east-1",
    }),
  },
});
```

## 沙箱自定义

### 自定义沙箱提供者

```typescript
import { Sandbox, SandboxProvider } from "deepagents";

// 创建自定义沙箱提供者
class CustomSandboxProvider implements SandboxProvider {
  name = "custom";
  
  async create(config) {
    // 创建容器
    const container = await this.createContainer({
      image: config.image,
      resources: config.resources,
    });
    
    return new CustomSandbox(container);
  }
  
  async destroy(sandbox) {
    await sandbox.container.remove();
  }
}

// 自定义沙箱实现
class CustomSandbox implements Sandbox {
  constructor(container) {
    this.container = container;
  }
  
  async execute(code) {
    // 在容器中执行代码
    const result = await this.container.exec(code);
    return result;
  }
  
  async uploadFile(path, content) {
    await this.container.writeFile(path, content);
  }
  
  async downloadFile(path) {
    return await this.container.readFile(path);
  }
}

// 使用自定义沙箱
const agent = new Agent({
  model: "gpt-4",
  sandbox: new CustomSandboxProvider(),
});
```

## 技能自定义

### 定义技能

```typescript
import { Agent, Skill } from "deepagents";

// 定义数据分析技能
const dataAnalysisSkill: Skill = {
  name: "data_analysis",
  description: "数据分析技能，支持数据清洗、统计分析和可视化",
  
  // 技能提示模板
  promptTemplate: `
你正在执行数据分析任务。

## 数据概览
{{dataOverview}}

## 分析目标
{{analysisGoal}}

## 可用工具
- read_data: 读取数据文件
- clean_data: 清洗数据
- analyze: 统计分析
- visualize: 生成可视化图表

请按照以下步骤进行：
1. 读取并理解数据
2. 清洗和预处理
3. 执行统计分析
4. 生成可视化图表
5. 撰写分析报告
`,
  
  // 技能工具
  tools: [
    "read_data",
    "clean_data",
    "analyze",
    "visualize",
  ],
  
  // 技能执行
  execute: async (input, context) => {
    const { dataPath, goal } = input;
    
    // 填充提示模板
    const prompt = this.promptTemplate
      .replace("{{dataOverview}}", await getDataOverview(dataPath))
      .replace("{{analysisGoal}}", goal);
    
    return { prompt, tools: this.tools };
  },
};

// 使用技能
const agent = new Agent({
  model: "gpt-4",
  skills: [dataAnalysisSkill],
});

// 调用技能

  dataPath: "/data/sales.csv",
```

## 结构化输出

### 定义输出格式

```typescript
import { Agent, StructuredOutput } from "deepagents";

// 定义结构化输出格式
const analysisOutput = StructuredOutput.define({
  name: "analysis_result",
  description: "数据分析结果",
  schema: {
    type: "object",
    properties: {
      summary: {
        type: "string",
        description: "分析摘要",
      },
      statistics: {
        type: "object",
        properties: {
          mean: { type: "number" },
          median: { type: "number" },
          stdDev: { type: "number" },
        },
      },
      insights: {
        type: "array",
        items: {
          type: "object",
          properties: {
            finding: { type: "string" },
            confidence: { type: "number" },
            impact: { type: "string", enum: ["high", "medium", "low"] },
          },
        },
      },
      recommendations: {
        type: "array",
        items: { type: "string" },
      },
    },
    required: ["summary", "statistics", "insights"],
  },
});

// 使用结构化输出
const agent = new Agent({
  model: "gpt-4",
  structuredOutput: analysisOutput,
});

const result = await agent.run("分析这份销售数据");
// result 自动解析为 schema 定义的格式
console.log(result.summary);
console.log(result.statistics.mean);
console.log(result.insights[0].finding);
```

```typescript
  model: "gpt-4",
  outputValidator: validator,
});
```

## 最佳实践

1. **模块化设计**：将自定义组件拆分为独立模块，便于维护和复用
2. **类型安全**：使用 TypeScript 定义接口和类型，确保类型安全
3. **配置驱动**：尽量使用配置文件而非硬编码，提高灵活性
4. **渐进式增强**：先实现基本功能，再逐步添加高级特性
5. **测试覆盖**：为自定义组件编写单元测试和集成测试
6. **文档完善**：为自定义组件提供清晰的文档和示例
7. **性能优化**：注意自定义组件的性能影响，避免过度设计
