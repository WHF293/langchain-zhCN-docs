# 第十一章 人在回路（Human-in-the-Loop）

> 来源: [Deep Agents 官方文档](https://docs.langchain.com/deep-agents)

---

人在回路（Human-in-the-Loop, HITL）是 Deep Agents 的重要安全机制，允许人类在关键决策点介入代理的执行过程。本章详细介绍 HITL 的配置和使用方法。

## 配置工具调用审批（Tool Call Approval）

为敏感工具配置审批流程：

```typescript
// 工具调用审批配置
import { HumanInTheLoop } from "@langchain/deep-agents";

const hitlConfig = new HumanInTheLoop({
  // 审批策略
  approval: {
    // 需要审批的工具
    tools: {
      // 数据库操作
      "database-write": {
        // 审批级别
        level: "required",  // 'required' | 'optional' | 'auto'

        // 审批者
        approvers: ["admin", "dba"],

        // 审批超时（毫秒）
        timeout: 300000,  // 5分钟

        // 超时处理
        onTimeout: "reject"  // 'reject' | 'approve' | 'escalate'
      },

      // 文件删除
      "file-delete": {
        level: "required",
        approvers: ["admin"],
        // 需要二次确认
        requireConfirmation: true,
        // 确认消息
        confirmationMessage: "确定要删除此文件吗？此操作不可撤销。"
      },

      // 外部 API 调用
      "external-api": {
        level: "optional",
        // 自动批准条件
        autoApproveConditions: [
          {
            // 低风险操作自动批准
            field: "riskLevel",
            operator: "equals",
            value: "low"
          },
          {
            // 熟悉的域名自动批准
            field: "domain",
            operator: "in",
            value: ["api.trusted.com", "data.internal.com"]
          }
        ]
      }
    },

    // 默认策略
    defaultPolicy: {
      level: "optional",
      timeout: 60000
    }
  }
});

// 应用审批配置
const agent = deepAgents.create({
  humanInTheLoop: hitlConfig
});

// 审批请求处理器
hitlConfig.onApprovalRequest(async (request) => {
  const { toolName, args, context } = request;

  // 发送审批通知
  await sendNotification({
    to: request.approvers,
    message: `工具调用审批请求: ${toolName}`,
    details: {
      tool: toolName,
      arguments: args,
      requester: context.userId,
      timestamp: new Date()
    },
    // 审批链接
    approvalUrl: `https://admin.example.com/approvals/${request.id}`
  });

  // 返回审批结果
  return {
    approved: await waitForApproval(request.id),
    reason: "人工审批"
  };
});
```

## 决策类型（Decision Types）

HITL 支持多种决策类型：

```typescript
// 决策类型定义
const decisionTypes = {

        enabled: true,

          {
            field: "riskLevel",

      }

        ]

      clarification: {

        fields: [
          {
            name: "clarifiedIntent",
            type: "string",
            description: "澄清后的意图"
          },
          {
            name: "additionalContext",
            type: "string",
            description: "额外上下文"
          }
        ]
      },

      // 选择选项
      selection: {
        description: "从多个选项中选择",
        fields: [
          {
            name: "selectedOption",
            type: "string",
            enum: ["option-a", "option-b", "option-c"],
            description: "选择的选项"
          }
        ]
      }
    }
  }
};

// 使用决策类型
agent.on("human-decision", async (event) => {
  const { decisionType, context } = event;

  switch (decisionType) {
    case "approve":
      const approved = await requestApproval(context);
      return { type: "approve", approved };

    case "edit":
      const edited = await requestEdit(context);
      return { type: "edit", editedValue: edited };

    case "reject":
      const reason = await requestRejectionReason(context);
      return { type: "reject", reason };

    case "respond":
      const response = await requestResponse(context);
      return { type: "respond", response };
  }
});
```

## 条件中断（Conditional Interrupts）

根据条件自动中断代理执行：

```typescript
// 条件中断配置
const conditionalInterrupts = {
  // 中断条件
  conditions: [
    // 风险阈值
    {
      name: "high-risk-operation",
      description: "高风险操作中断",

      // 触发条件
      condition: (context) => {
        return context.riskScore > 0.8;
      },

      // 中断行为
      interrupt: {
        // 中断消息
        message: "检测到高风险操作，需要人工确认",
        // 中断级别
        level: "critical",
        // 是否阻塞
        blocking: true
      }
    },

    // 敏感数据访问
    {
      name: "sensitive-data-access",
      description: "敏感数据访问中断",

      condition: (context) => {
        const sensitivePatterns = [
          /password/i,
          /credit.?card/i,
          /ssn/i,
          /social.?security/i
        ];

        return sensitivePatterns.some(
          pattern => pattern.test(context.data)
        );
      },

      interrupt: {
        message: "检测到敏感数据访问，请确认",
        level: "warning",
        blocking: true
      }
    },

    // 资源限制
    {
      name: "resource-limit",
      description: "资源使用超限中断",

      condition: (context) => {
        return context.tokenUsage > 10000 ||
               context.executionTime > 300000;
      },

      interrupt: {
        message: "资源使用即将超限",
        level: "info",
        blocking: false
      }
    },

    // 业务规则
    {
      name: "business-rule",
      description: "业务规则违反中断",

      condition: (context) => {
        // 检查业务规则
        const violations = checkBusinessRules(context);
        return violations.length > 0;
      },

      interrupt: {
        message: "业务规则检查失败",
        level: "error",
        blocking: true
      }
    }
  ],

  // 中断处理器
  handler: async (interrupt) => {
    const { name, level, context, message } = interrupt;

    // 记录中断
    await logInterrupt({
      name,
      level,
      context,
      timestamp: Date.now()
    });

    // 根据级别处理
    switch (level) {
      case "critical":
        // 立即通知管理员
        await notifyAdmin({
          subject: `紧急中断: ${name}`,
          body: message,
          context
        });
        // 等待管理员响应
        return await waitForAdminResponse();

      case "warning":
        // 发送警告通知
        await sendWarning(message);
        // 请求用户确认
        return await requestUserConfirmation(message);

      case "info":
        // 仅记录日志
        console.info(`中断: ${name} - ${message}`);
        return { continue: true };

      case "error":
        // 发送错误通知并停止
        await sendErrorNotification(message);
        return { continue: false, error: message };
    }
  }
};

// 应用条件中断
const agent = deepAgents.create({
  interrupts: conditionalInterrupts
});

// 运行时检查中断
agent.on("before-action", async (action, context) => {
  // 检查所有中断条件
  for (const condition of conditionalInterrupts.conditions) {
    if (condition.condition(context)) {
      // 触发中断
      const result = await conditionalInterrupts.handler({
        name: condition.name,
        level: condition.interrupt.level,
        message: condition.interrupt.message,
        context
      });

      if (!result.continue) {
        throw new InterruptedError(condition.name, result.error);
      }
    }
  }
});
```

## 子代理中断（Sub-Agent Interrupts）

在子代理执行过程中实现中断：

```typescript
// 子代理中断配置
const subAgentInterrupts = {
  // 中断点定义
  interruptPoints: {
    // 在特定节点前中断
    beforeNode: {
      "critical-operation": {
        message: "即将执行关键操作",
        requireApproval: true
      }
    },

    // 在特定节点后中断
    afterNode: {
      "data-collection": {
        message: "数据收集完成，请确认",
        requireConfirmation: true
      }
    },

    // 条件中断
    conditional: [
      {
        // 在任何错误后中断
        afterError: true,
        message: "发生错误，需要人工介入",
        level: "error"
      },
      {
        // 在特定状态时中断
        whenState: (state) => state.needsReview === true,
        message: "需要审查当前状态",
        level: "warning"
      }
    ]
  },

  // 中断恢复策略
  recovery: {
    // 从中断点恢复
    resumeFromInterrupt: true,

    // 回滚到上一个检查点
    rollbackToCheckpoint: true,

    // 检查点间隔
    checkpointInterval: 5  // 每5个节点创建检查点
  }
};

// 创建带中断的子代理
const subAgent = {
  name: "sensitive-processor",

  // 应用中断配置
  interrupts: subAgentInterrupts,

  // 执行处理器
  handler: async (input, context) => {
    // 创建检查点管理器
    const checkpointManager = context.createCheckpointManager();

    // 执行步骤
    await context.executeStep("prepare", async () => {
      // 准备数据
      return await prepareData(input);
    });

    // 关键操作 - 会触发中断
    await context.executeStep("critical-operation", async () => {
      // 这里会触发 beforeNode 中断
      return await performCriticalOperation(input);
    });

    // 数据收集 - 完成后会触发中断
    const data = await context.executeStep("data-collection", async () => {
      return await collectData(input);
    });

    // 条件中断 - 如果需要审查
    if (data.needsReview) {
      await context.interrupt({
        message: "数据需要审查",
        level: "warning",
        data: data
      });
    }

    return data;
  }
};

// 处理子代理中断
agent.on("subagent-interrupt", async (event) => {
  const { subAgent, interrupt, checkpoint } = event;

  console.log(`子代理 ${subAgent.name} 在 ${interrupt.point} 中断`);
  console.log(`原因: ${interrupt.message}`);

  // 请求人工决策
  const decision = await requestHumanDecision({
    message: interrupt.message,
    options: ["继续", "回滚", "终止"],
    context: interrupt.context
  });

  switch (decision.action) {
    case "继续":
      // 从中断点恢复
      await event.resume();
      break;

    case "回滚":
      // 回滚到检查点
      await event.rollback(checkpoint);
      break;

    case "终止":
      // 终止子代理
      await event.terminate();
      break;
  }
});
```

## 文件系统权限中断（Filesystem Permission Interrupts）

在文件操作时进行权限检查和中断：

```typescript
// 文件系统权限中断配置
const filesystemInterrupts = {
  // 权限规则
  rules: [
    // 读取权限
    {
      operation: "read",
      paths: ["/data/**", "/shared/**"],
      // 自动批准
      autoApprove: true
    },

    // 写入权限
    {
      operation: "write",
      paths: ["/data/output/**"],
      // 需要审批
      requireApproval: true,
      approvers: ["admin"]
    },

    // 删除权限
    {
      operation: "delete",
      paths: ["/data/**"],
      // 严格控制
      requireApproval: true,
      approvers: ["admin"],
      // 需要二次确认
      requireConfirmation: true,
      confirmationMessage: "确定要删除此文件吗？此操作不可撤销。"
    },

    // 敏感目录
    {
      operation: "any",
      paths: ["/etc/**", "/root/**", "/system/**"],
      // 完全禁止
      deny: true,
      message: "不允许访问系统目录"
    }
  ],

  // 临时权限
  temporaryPermissions: {
    // 允许临时授权
    enabled: true,

    // 临时权限有效期
    duration: 3600000,  // 1小时

    // 临时权限范围
    scope: "session"  // 'session' | 'global'
  },

  // 审计日志
  audit: {
    enabled: true,

    // 记录所有操作
    logAllOperations: true,

    // 日志存储
    storage: {
      type: "file",
      path: "./audit.log"
    }
  }
};

// 创建文件系统权限检查器
const fsPermissionChecker = {
  // 检查权限
  check: async (operation, path, context) => {
    // 查找匹配的规则
    const rule = filesystemInterrupts.rules.find(r =>
      r.operation === operation || r.operation === "any"
    );

    if (!rule) {
      return { allowed: false, reason: "没有匹配的权限规则" };
    }

    // 检查是否拒绝
    if (rule.deny) {
      return {
        allowed: false,
        reason: rule.message || "操作被拒绝"
      };
    }

    // 检查是否需要审批
    if (rule.requireApproval) {
      // 请求审批
      const approved = await requestFilesystemApproval({
        operation,
        path,
        approvers: rule.approvers,
        context
      });

      if (!approved) {
        return { allowed: false, reason: "审批被拒绝" };
      }
    }

    // 检查是否需要确认
    if (rule.requireConfirmation) {
      const confirmed = await requestConfirmation({
        message: rule.confirmationMessage,
        details: { operation, path }
      });

      if (!confirmed) {
        return { allowed: false, reason: "用户取消" };
      }
    }

    return { allowed: true };
  },

  // 授予临时权限
  grantTemporary: async (operation, path, duration) => {
    const permission = {
      operation,
      path,
      grantedAt: Date.now(),
      expiresAt: Date.now() + duration,
      scope: "session"
    };

    await saveTemporaryPermission(permission);
    return permission;
  }
};

// 应用文件系统中断
agent.use("filesystem", {
  // 权限检查
  beforeOperation: async (operation, path, context) => {
    const result = await fsPermissionChecker.check(
      operation,
      path,
      context
    );

    if (!result.allowed) {
      throw new PermissionDeniedError(result.reason);
    }

    // 记录审计日志
    await auditLog({
      operation,
      path,
      user: context.userId,
      allowed: true,
      timestamp: Date.now()
    });
  }
});

// 文件操作中断处理
agent.on("filesystem-interrupt", async (event) => {
  const { operation, path, reason } = event;

  console.log(`文件系统中断: ${operation} ${path}`);
  console.log(`原因: ${reason}`);

  // 提供选项
  const choice = await presentOptions({
    message: `需要权限才能${operation}文件: ${path}`,
    options: [
      { label: "批准", value: "approve" },
      { label: "授予临时权限", value: "grant-temporary" },
      { label: "拒绝", value: "reject" }
    ]
  });

  switch (choice.value) {
    case "approve":
      await event.approve();
      break;

    case "grant-temporary":
      await fsPermissionChecker.grantTemporary(
        operation,
        path,
        3600000
      );
      await event.approve();
      break;

    case "reject":
      await event.reject();
      break;
  }
});
```
