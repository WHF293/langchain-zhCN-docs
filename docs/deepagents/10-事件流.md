# 第十章 事件流（Event Stream）

> 来源: [Deep Agents 官方文档](https://docs.langchain.com/deep-agents)

---

事件流是 Deep Agents 中处理异步事件的核心机制，提供了统一的事件处理接口。本章详细介绍事件流的架构和使用方法。

## 流式子代理（Streaming Sub-Agents）

子代理可以通过事件流进行流式通信：

```typescript
// 流式子代理配置
import { EventStream, SubAgentStream } from "@langchain/deep-agents";

// 创建事件流
const eventStream = new EventStream({
  // 缓冲区大小
  bufferSize: 1000,

  // 背压处理
  backpressure: {
    strategy: "drop-oldest",  // 'drop-oldest' | 'drop-newest' | 'block'
    highWaterMark: 800,
    lowWaterMark: 200
  }
});

// 流式子代理
const streamingSubAgent = {
  name: "stream-processor",

  // 使用事件流
  stream: eventStream,

  handler: async function* (input, context) {
    // 发送开始事件
    yield {
      type: "start",
      data: { timestamp: Date.now() }
    };

    // 处理数据流
    for (const chunk of input.chunks) {
      const result = await processChunk(chunk);

      // 发送进度事件
      yield {
        type: "progress",
        data: {
          processed: result.count,
          total: input.chunks.length
        }
      };

      // 发送数据事件
      yield {
        type: "data",
        data: result
      };
    }

    // 发送完成事件
    yield {
      type: "complete",
      data: { timestamp: Date.now() }
    };
  }
};

// 消费子代理事件流
const agent = deepAgents.create({
  subAgents: { "streamer": streamingSubAgent }
});

const stream = agent.invokeStream({
  input: { chunks: largeDataset },
  subAgent: "streamer"
});

for await (const event of stream) {
  switch (event.type) {
    case "start":
      console.log("处理开始");
      break;
    case "progress":
      console.log(`进度: ${event.data.processed}/${event.data.total}`);
      break;
    case "data":
      await saveResult(event.data);
      break;
    case "complete":
      console.log("处理完成");
      break;
  }
}
```

## 消息流（Message Stream）

处理代理间的消息传递：

```typescript
// 消息流配置
const messageStreamConfig = {
  // 消息类型
  messageTypes: {
    // 系统消息
    system: {
      priority: "high",
      persistent: true
    },

    // 业务消息
    business: {
      priority: "normal",
      persistent: true
    },

    // 事件消息
    event: {
      priority: "low",
      persistent: false
    }
  },

  // 消息路由
  routing: {
    // 直接路由
    direct: {
      enabled: true,
      // 点对点通信
      peerToPeer: true
    },

    // 广播路由
    broadcast: {
      enabled: true,
      // 广播范围
      scope: "group"  // 'all' | 'group' | 'team'
    },

    // 主题路由
    topic: {
      enabled: true,
      // 订阅模式
      subscription: "pub-sub"
    }
  },

  // 消息持久化
  persistence: {
    enabled: true,
    // 存储后端
    storage: "redis",
    // 过期时间
    ttl: 86400000  // 24小时
  }
};

// 创建消息流
const messageStream = new MessageStream(messageStreamConfig);

// 发送消息
await messageStream.send({
  type: "business",
  from: "agent-a",
  to: "agent-b",
  topic: "task-assignment",
  payload: {
    taskId: "task-123",
    description: "处理用户请求"
  }
});

// 订阅消息
messageStream.subscribe({
  // 订阅者
  subscriber: "agent-b",

  // 过滤条件
  filter: {
    types: ["business", "system"],
    topics: ["task-assignment", "system-alert"]
  },

  // 消息处理器
  handler: async (message) => {
    console.log(`收到消息: ${message.type}`);
    console.log(`来自: ${message.from}`);
    console.log(`内容:`, message.payload);

    // 处理消息
    await handleMessage(message);

    // 确认消息
    await message.ack();
  }
});

// 广播消息
await messageStream.broadcast({
  type: "system",
  topic: "system-alert",
  payload: {
    level: "warning",
    message: "系统即将维护"
  }
});
```

## 工具调用事件流（Tool Call Event Stream）

跟踪工具调用的完整生命周期：

```typescript
// 工具调用事件流
const toolCallStream = {
  // 事件类型
  events: {
    // 调用前
    beforeCall: {
      // 参数验证
      validation: true,
      // 权限检查
      permissionCheck: true
    },

    // 调用中
    duringCall: {
      // 进度更新
      progress: true,
      // 日志记录
      logging: true
    },

    // 调用后
    afterCall: {
      // 结果验证
      resultValidation: true,
      // 副作用记录
      sideEffects: true
    },

    // 错误处理
    onError: {
      // 重试策略
      retry: true,
      // 降级处理
      fallback: true
    }
  }
};

// 监听工具调用事件
agent.on("tool-call", async (event) => {
  const { phase, toolName, args, result, error } = event;

  switch (phase) {
    case "before":
      console.log(`准备调用工具: ${toolName}`);
      console.log(`参数:`, args);

      // 权限检查
      const hasPermission = await checkPermission(toolName, args);
      if (!hasPermission) {
        throw new PermissionDeniedError(`无权调用工具: ${toolName}`);
      }
      break;

    case "during":
      console.log(`正在执行: ${toolName}`);
      break;

    case "after":
      console.log(`工具完成: ${toolName}`);
      console.log(`结果:`, result);

      // 记录审计日志
      await auditLog({
        tool: toolName,
        args: args,
        result: result,
        timestamp: Date.now()
      });
      break;

    case "error":
      console.error(`工具错误: ${toolName}`, error);
      break;
  }
});

// 使用工具调用流
const result = await agent.invoke({
  input: "查询数据库",
  tools: ["database-query"]
});
```

## 嵌套工作流（Nested Workflows）

支持在事件流中嵌套复杂的工作流：

```typescript
// 嵌套工作流定义
const nestedWorkflow = {
  // 主工作流
  main: {
    name: "data-processing-pipeline",

    // 步骤定义
    steps: [
      {
        id: "extract",
        name: "数据提取",
        // 嵌套子工作流
        subworkflow: {
          name: "extraction-workflow",
          steps: [
            { id: "connect", action: "连接数据源" },
            { id: "query", action: "执行查询" },
            { id: "validate", action: "验证数据" }
          ]
        }
      },
      {
        id: "transform",
        name: "数据转换",
        subworkflow: {
          name: "transformation-workflow",
          steps: [
            { id: "clean", action: "数据清洗" },
            { id: "normalize", action: "数据标准化" },
            { id: "aggregate", action: "数据聚合" }
          ]
        }
      },
      {
        id: "load",
        name: "数据加载",
        subworkflow: {
          name: "loading-workflow",
          steps: [
            { id: "prepare", action: "准备目标表" },
            { id: "insert", action: "插入数据" },
            { id: "verify", action: "验证加载" }
          ]
        }
      }
    ]
  }
};

// 执行嵌套工作流
const workflowStream = agent.executeWorkflow({
  workflow: nestedWorkflow.main,
  input: { source: "database", target: "warehouse" }
});

// 处理嵌套事件
for await (const event of workflowStream) {
  // 事件包含层级信息
  const { level, workflow, step, action, status } = event;

  // 格式化输出
  const indent = "  ".repeat(level);
  console.log(`${indent}[${workflow}] ${step}: ${action} - ${status}`);

  // 处理子工作流事件
  if (event.subworkflow) {
    console.log(`${indent}  进入子工作流: ${event.subworkflow.name}`);
  }
}

// 获取工作流状态
const status = await agent.getWorkflowStatus();
console.log("工作流状态:", {
  main: status.main.status,
  progress: status.main.progress,
  subworkflows: status.subworkflows
});
```

## 并发消费（Concurrent Consumption）

支持多个消费者并发处理事件流：

```typescript
// 并发消费配置
const concurrentConfig = {
  // 消费者数量
  consumers: 5,

  // 消费策略
  strategy: "competing",  // 'competing' | 'partitioned' | 'broadcast'

  // 负载均衡
  loadBalancing: {
    enabled: true,
    algorithm: "round-robin"  // 'round-robin' | 'least-loaded' | 'random'
  },

  // 消息确认
  acknowledgment: {
    mode: "manual",  // 'auto' | 'manual'
    timeout: 30000,  // 确认超时
    // 重试策略
    retry: {
      maxAttempts: 3,
      backoff: "exponential"
    }
  },

  // 并发控制
  concurrency: {
    // 每个消费者的并发处理数
    perConsumer: 10,
    // 全局并发限制
    global: 50
  }
};

// 创建并发消费者
const consumerGroup = new ConsumerGroup({
  groupId: "processing-group",
  config: concurrentConfig
});

// 添加消费者
for (let i = 0; i < concurrentConfig.consumers; i++) {
  consumerGroup.addConsumer({
    id: `consumer-${i}`,
    handler: async (event) => {
      console.log(`Consumer ${i} 处理事件:`, event.id);

      // 模拟处理
      await processEvent(event);

      // 手动确认
      await event.ack();
    }
  });
}

// 启动消费
await consumerGroup.start({
  stream: eventStream,
  // 从哪里开始消费
  startFrom: "latest",  // 'earliest' | 'latest' | 'specific'
  // 具体位置（当 startFrom 为 'specific' 时）
  startPosition: "offset-12345"
});

// 监控消费状态
consumerGroup.on("metrics", (metrics) => {
  console.log("消费指标:", {
    throughput: metrics.throughput,
    latency: metrics.latency,
    lag: metrics.lag,
    errors: metrics.errors
  });
});

// 优雅关闭
process.on("SIGTERM", async () => {
  await consumerGroup.shutdown({
    // 等待当前消息处理完成
    waitForCompletion: true,
    // 超时时间
    timeout: 30000
  });
});
```

## v3 流式 API

Deep Agents v3 提供了全新的流式 API：

```typescript
// v3 流式 API
import { Stream } from "@langchain/deep-agents/v3";

// 创建 v3 流
const v3Stream = new Stream({
  // 流配置
  config: {
    // 高性能模式
    highPerformance: true,

    // 零拷贝
    zeroCopy: true,

    // 批处理
    batching: {
      enabled: true,
      maxBatchSize: 100,
      maxWaitTime: 50
    }
  },

  // 转换管道
  pipeline: [
    // 过滤器
    {
      type: "filter",
      predicate: (event) => event.priority !== "low"
    },

    // 转换器
    {
      type: "transform",
      transformer: (event) => ({
        ...event,
        processedAt: Date.now()
      })
    },

    // 聚合器
    {
      type: "aggregate",
      window: 1000,  // 1秒窗口
      aggregator: (events) => ({
        count: events.length,
        types: [...new Set(events.map(e => e.type))]
      })
    }
  ]
});

// 使用 v3 流式 API
const result = await agent.invoke({
  input: input,
  stream: {
    // 使用 v3 流
    version: "v3",

    // 响应式流
    reactive: true,

    // 背压处理
    backpressure: {
      strategy: "buffer",
      bufferSize: 10000
    }
  }
});

// v3 流操作符
v3Stream
  // 过滤
  .filter(event => event.type === "data")
  // 转换
  .map(event => event.payload)
  // 分组
  .groupBy(event => event.category)
  // 窗口
  .window(1000)
  // 聚合
  .reduce((acc, events) => [...acc, ...events], [])
  // 输出
  .subscribe({
    next: (data) => console.log("数据:", data),
    error: (err) => console.error("错误:", err),
    complete: () => console.log("完成")
  });

// v3 流组合
const combinedStream = Stream.merge([
  stream1,
  stream2,
  stream3
]);

// v3 流错误处理
const safeStream = v3Stream
  .catchError(async (error, event) => {
    // 记录错误
    await logError(error, event);

    // 返回恢复事件
    return {
      type: "recovered",
      originalEvent: event,
      error: error.message
    };
  });

// v3 流重试
const retryStream = v3Stream
  .retry({
    maxAttempts: 3,
    delay: 1000,
    backoff: "exponential",
    // 只在特定错误时重试
    retryIf: (error) => error.code === "TEMPORARY_ERROR"
  });

// v3 流超时
const timeoutStream = v3Stream
  .timeout(30000)
  .onTimeout(() => {
    console.log("流超时");
    return { type: "timeout" };
  });
```

> **注意**：v3 流式 API 是最新版本，提供了更好的性能和更丰富的功能。建议新项目使用 v3 API。

> **提示**：在处理大量事件时，使用批处理和窗口操作可以显著提高性能。

> **建议**：合理配置背压策略，避免消费者处理不过来导致内存溢出。可以根据实际场景选择 `drop-oldest`、`drop-newest` 或 `block` 策略。
