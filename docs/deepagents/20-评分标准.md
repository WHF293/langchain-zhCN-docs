# 第二十章 评分标准

> 来源: [Deep Agents 官方文档](https://docs.langchain.com/deep-agents)

---

评分标准（Rubric）是评估代理输出质量的核心机制。通过定义明确的评分维度和标准，我们可以系统性地衡量代理的表现，并驱动持续改进。本章介绍评分标准的定义、LLM-as-a-Judge 评分方法以及自评估循环的实现。

## 评分标准概述

评分标准（Rubric）是一组结构化的评估维度和质量等级，用于衡量代理输出在不同方面的表现。

### 评分维度


## LLM-as-a-Judge 评分

LLM-as-a-Judge 是使用另一个 LLM 来评估代理输出质量的方法。这种方法可以自动化评分过程，实现大规模的质量评估。

```typescript
// LLM-as-a-Judge 评分器
class LLMJudge {
  private llm: BaseLanguageModel;
  private rubric: Rubric;

  constructor(llm: BaseLanguageModel, rubric: Rubric) {
    this.llm = llm;
    this.rubric = rubric;
  }

  // 评估代理输出
  async evaluate(input: string, output: string): Promise<EvaluationResult> {
    const dimensionScores: DimensionScore[] = [];

    // 对每个维度进行评分
    for (const dimension of this.rubric.dimensions) {
      const score = await this.evaluateDimension(
        dimension,
        input,
        output
      );
      dimensionScores.push({
        dimension: dimension.name,
        score: score.score,
        reasoning: score.reasoning,
        evidence: score.evidence,
      });
    }

    // 计算加权总分
    const totalScore = this.calculateWeightedScore(dimensionScores);

    // 生成总体评价
    const overallAssessment = await this.generateAssessment(
      input,
      output,
      dimensionScores
    );

    return {
      rubricName: this.rubric.name,
      rubricVersion: this.rubric.version,
      totalScore,
      passing: totalScore >= this.rubric.passingScore,
      dimensionScores,
      overallAssessment,
      evaluatedAt: new Date(),
    };
  }

  // 评估单个维度
  private async evaluateDimension(
    dimension: RubricDimension,
    input: string,
    output: string
  ): Promise<DimensionEvaluation> {
    // 构建评分提示
    const prompt = this.buildEvaluationPrompt(dimension, input, output);

    // 调用 LLM 进行评分
    const response = await this.llm.invoke(prompt);

    // 解析评分结果
    const parsed = this.parseEvaluationResponse(response);

    // 验证评分是否在有效范围内
    this.validateScore(parsed.score, dimension.levels);

    return {
      score: parsed.score,
      reasoning: parsed.reasoning,
      evidence: parsed.evidence,
      level: this.findLevel(parsed.score, dimension.levels),
    };
  }

  // 构建评分提示
  private buildEvaluationPrompt(
    dimension: RubricDimension,
    input: string,
    output: string
  ): string {
    const levelsDescription = dimension.levels
      .map(
        (level) =>
          `${level.score}分（${level.label}）：${level.description}
    标准：${level.criteria.join("；")}`
      )
      .join("\n\n");

    return `你是一个专业的评估专家。请根据以下评分标准，对给定的输出进行评分。

## 评分维度
名称：${dimension.name}
描述：${dimension.description}

## 评分等级
${levelsDescription}

## 待评估内容

### 输入（用户请求）
${input}

### 输出（代理响应）
${output}

## 评估要求
1. 仔细分析输出内容
2. 根据上述评分标准进行评分
3. 提供具体的证据支持你的评分
4. 给出评分理由

## 输出格式（JSON）
{
  "score": <1-5的整数>,
  "reasoning": "<详细的评分理由>",
  "evidence": ["<证据1>", "<证据2>", ...]
}

请严格按照上述 JSON 格式输出，不要包含其他内容。`;
  }

  // 解析评分响应
  private parseEvaluationResponse(response: string): ParsedEvaluation {
    try {
      // 提取 JSON 部分
      const jsonMatch = response.match(/\{[\s\S]*\}/);
      if (!jsonMatch) {
        throw new Error("无法从响应中提取 JSON");
      }

      const parsed = JSON.parse(jsonMatch[0]);

      // 验证必需字段
      if (typeof parsed.score !== "number") {
        throw new Error("评分必须是数字");
      }
      if (typeof parsed.reasoning !== "string") {
        throw new Error("评分理由必须是字符串");
      }
      if (!Array.isArray(parsed.evidence)) {
        throw new Error("证据必须是数组");
      }

      return parsed;
    } catch (error) {
      throw new Error(`解析评分响应失败: ${error.message}`);
    }
  }

  // 计算加权总分
  private calculateWeightedScore(dimensionScores: DimensionScore[]): number {
    let totalWeight = 0;
    let weightedSum = 0;

    for (const ds of dimensionScores) {
      const dimension = this.rubric.dimensions.find(
        (d) => d.name === ds.dimension
      );
      if (dimension) {
        weightedSum += ds.score * dimension.weight;
        totalWeight += dimension.weight;
      }
    }

    return totalWeight > 0 ? weightedSum / totalWeight : 0;
  }

  // 生成总体评价
  private async generateAssessment(
    input: string,
    output: string,
    dimensionScores: DimensionScore[]
  ): Promise<string> {
    const scoresSummary = dimensionScores
      .map((ds) => `${ds.dimension}: ${ds.score}/5 - ${ds.reasoning}`)
      .join("\n");

    const prompt = `基于以下各维度的评分结果，生成一段简洁的总体评价：

${scoresSummary}

请生成 2-3 句话的总体评价，指出主要优点和需要改进的地方。`;

    return this.llm.invoke(prompt);
  }
}

// 评分结果接口
interface EvaluationResult {
  rubricName: string;
  rubricVersion: string;
  totalScore: number;
  passing: boolean;
  dimensionScores: DimensionScore[];
  overallAssessment: string;
  evaluatedAt: Date;
}

interface DimensionScore {
  dimension: string;
  score: number;
  reasoning: string;
  evidence: string[];
}

interface DimensionEvaluation {
  score: number;
  reasoning: string;
  evidence: string[];
  level: RubricLevel;
}
```

> **注意**：LLM-as-a-Judge 方法可能存在偏差。建议使用多个评估模型进行交叉验证，并定期与人工评估结果进行对比校准。

## RubricMiddleware

RubricMiddleware 是一个中间件组件，可以在代理运行时实时评估输出质量。

```typescript
// RubricMiddleware 实现
class RubricMiddleware {
  private judge: LLMJudge;
  private config: RubricMiddlewareConfig;

  constructor(judge: LLMJudge, config: RubricMiddlewareConfig) {
    this.judge = judge;
    this.config = config;
  }

  // 中间件处理函数
  async process(
    context: MiddlewareContext,
    next: () => Promise<AgentResponse>
  ): Promise<AgentResponse> {
    // 执行代理
    const response = await next();

    // 根据配置决定是否评估
    if (this.shouldEvaluate(context)) {
      // 异步评估（不阻塞响应）
      if (this.config.asyncEvaluation) {
        this.evaluateAsync(context, response);
        return response;
      }

      // 同步评估
      const evaluation = await this.evaluate(context, response);

      // 将评估结果附加到响应
      response.metadata = {
        ...response.metadata,
        evaluation,
      };

      // 如果评估不通过，触发相应处理
      if (!evaluation.passing) {
        return this.handleFailingEvaluation(context, response, evaluation);
      }
    }

    return response;
  }

  // 判断是否需要评估
  private shouldEvaluate(context: MiddlewareContext): boolean {
    // 按采样率评估
    if (Math.random() > this.config.samplingRate) {
      return false;
    }

    // 强制评估的情况
    if (context.forceEvaluation) {
      return true;
    }

    // 特定任务类型强制评估
    if (this.config.alwaysEvaluateTasks?.includes(context.taskType)) {
      return true;
    }

    return true;
  }

  // 执行评估
  private async evaluate(
    context: MiddlewareContext,
    response: AgentResponse
  ): Promise<EvaluationResult> {
    const startTime = Date.now();

    try {
      const evaluation = await this.judge.evaluate(
        context.input,
        response.output
      );

      // 记录评估指标
      this.recordMetrics({
        evaluationTime: Date.now() - startTime,
        totalScore: evaluation.totalScore,
        passing: evaluation.passing,
        dimensionScores: evaluation.dimensionScores,
      });

      return evaluation;
    } catch (error) {
      // 评估失败不应影响正常响应
      console.error("评估失败:", error);
      return {
        rubricName: this.judge.rubric.name,
        rubricVersion: this.judge.rubric.version,
        totalScore: 0,
        passing: true, // 评估失败时默认通过
        dimensionScores: [],
        overallAssessment: "评估失败",
        evaluatedAt: new Date(),
      };
    }
  }

  // 异步评估
  private async evaluateAsync(
    context: MiddlewareContext,
    response: AgentResponse
  ): Promise<void> {
    // 使用队列进行异步评估
    await this.evaluationQueue.add({
      context,
      response,
      timestamp: new Date(),
    });
  }

  // 处理评估不通过的情况
  private async handleFailingEvaluation(
    context: MiddlewareContext,
    response: AgentResponse,
    evaluation: EvaluationResult
  ): Promise<AgentResponse> {
    switch (this.config.onFailure) {
      case "retry":
        // 重试代理调用
        return this.retryAgent(context);

      case "augment":
        // 在响应中添加质量警告
        return {
          ...response,
          output: this.augmentWithWarning(response.output, evaluation),
          metadata: {
            ...response.metadata,
            qualityWarning: true,
            evaluation,
          },
        };

      case "fallback":
        // 使用降级响应
        return this.getFallbackResponse(context, evaluation);

      case "log":
        // 仅记录，不影响响应
        await this.logFailingEvaluation(context, evaluation);
        return response;

      default:
        return response;
    }
  }

  // 重试代理调用
  private async retryAgent(
    context: MiddlewareContext
  ): Promise<AgentResponse> {
    const maxRetries = this.config.maxRetries || 2;

    for (let i = 0; i < maxRetries; i++) {
      const response = await context.executeAgent();

      const evaluation = await this.evaluate(context, response);

      if (evaluation.passing) {
        return response;
      }
    }

    // 重试后仍不通过，返回最后一次响应
    return context.executeAgent();
  }

  // 添加质量警告
  private augmentWithWarning(
    output: string,
    evaluation: EvaluationResult
  ): string {
    const failingDimensions = evaluation.dimensionScores
      .filter((ds) => ds.score < 3)
      .map((ds) => ds.dimension);

    const warning = `\n\n---\n⚠️ 质量提示：此响应在以下维度评分较低：${failingDimensions.join("、")}。请谨慎参考。`;

    return output + warning;
  }
}

// 配置接口
interface RubricMiddlewareConfig {
  // 采样率（0-1）
  samplingRate: number;
  // 是否异步评估
  asyncEvaluation: boolean;
  // 评估失败时的处理方式
  onFailure: "retry" | "augment" | "fallback" | "log";
  // 最大重试次数
  maxRetries?: number;
  // 强制评估的任务类型
  alwaysEvaluateTasks?: string[];
}
```

> **提示**：在生产环境中，建议将采样率设置为较低值（如 0.1），以平衡评估成本和监控覆盖度。

## 自评估循环

自评估循环（Self-Evaluation Loop）允许代理在生成输出后自行评估并改进。

```typescript
// 自评估循环实现
class SelfEvaluationLoop {
  private agent: Agent;
  private judge: LLMJudge;
  private config: SelfEvalConfig;

  constructor(agent: Agent, judge: LLMJudge, config: SelfEvalConfig) {
    this.agent = agent;
    this.judge = judge;
    this.config = config;
  }

  // 执行自评估循环
  async execute(input: string): Promise<SelfEvalResult> {
    let currentOutput = "";
    let iteration = 0;
    const history: IterationHistory[] = [];

    // 初始生成
    currentOutput = await this.agent.execute(input);

    while (iteration < this.config.maxIterations) {
      iteration++;

      // 评估当前输出
      const evaluation = await this.judge.evaluate(input, currentOutput);

      // 记录历史
      history.push({
        iteration,
        output: currentOutput,
        evaluation,
        timestamp: new Date(),
      });

      // 检查是否满足退出条件
      if (this.shouldExit(evaluation, iteration)) {
        break;
      }

      // 根据评估结果改进输出
      currentOutput = await this.improveOutput(
        input,
        currentOutput,
        evaluation
      );
    }

    // 选择最佳输出
    const bestIteration = this.selectBestIteration(history);

    return {
      finalOutput: bestIteration.output,
      finalEvaluation: bestIteration.evaluation,
      totalIterations: iteration,
      history,
      improvement: this.calculateImprovement(history),
    };
  }

  // 判断是否应该退出循环
  private shouldExit(
    evaluation: EvaluationResult,
    iteration: number
  ): boolean {
    // 达到满意分数
    if (evaluation.totalScore >= this.config.targetScore) {
      return true;
    }

    // 达到最大迭代次数
    if (iteration >= this.config.maxIterations) {
      return true;
    }

    // 分数不再提升（早停）
    if (this.config.earlyStopping && this.isPlateauReached()) {
      return true;
    }

    return false;
  }

  // 改进输出
  private async improveOutput(
    input: string,
    currentOutput: string,
    evaluation: EvaluationResult
  ): Promise<string> {
    // 构建改进提示
    const improvementPrompt = this.buildImprovementPrompt(
      input,
      currentOutput,
      evaluation
    );

    // 调用代理进行改进
    return this.agent.execute(improvementPrompt);
  }

  // 构建改进提示
  private buildImprovementPrompt(
    input: string,
    currentOutput: string,
    evaluation: EvaluationResult
  ): string {
    // 提取需要改进的维度
    const weakDimensions = evaluation.dimensionScores
      .filter((ds) => ds.score < this.config.targetScore)
      .map(
        (ds) =>
          `维度：${ds.dimension}
当前评分：${ds.score}/5
问题：${ds.reasoning}
改进建议：${ds.evidence.join("；")}`
      )
      .join("\n\n");

    return `请改进以下输出，重点解决评估中指出的问题。

## 原始输入
${input}

## 当前输出
${currentOutput}

## 需要改进的方面
${weakDimensions}

## 改进要求
1. 保持当前输出的优点
2. 针对性地解决上述问题
3. 确保改进后的输出完整、准确、可操作
4. 不要引入新的问题

请直接输出改进后的完整内容，不要包含解释。`;
  }

  // 选择最佳迭代
  private selectBestIteration(
    history: IterationHistory[]
  ): IterationHistory {
    return history.reduce((best, current) =>
      current.evaluation.totalScore > best.evaluation.totalScore
        ? current
        : best
    );
  }

  // 计算改进幅度
  private calculateImprovement(history: IterationHistory[]): ImprovementInfo {
    if (history.length < 2) {
      return { absolute: 0, percentage: 0 };
    }

    const firstScore = history[0].evaluation.totalScore;
    const lastScore = history[history.length - 1].evaluation.totalScore;

    return {
      absolute: lastScore - firstScore,
      percentage: ((lastScore - firstScore) / firstScore) * 100,
      dimensionImprovements: this.calculateDimensionImprovements(history),
    };
  }
```

## 裁决

裁决（Adjudication）是在多个评估结果不一致时，通过仲裁机制得出最终结论的过程。

```typescript
// 裁决系统
class AdjudicationSystem {
  private judges: LLMJudge[];
  private humanReviewer?: HumanReviewer;
  private config: AdjudicationConfig;

  constructor(config: AdjudicationConfig) {
    this.judges = config.judges;
    this.humanReviewer = config.humanReviewer;
    this.config = config;
  }

  // 执行裁决
  async adjudicate(
    input: string,
    output: string
  ): Promise<AdjudicationResult> {
    // 收集所有评委的评分
    const evaluations = await Promise.all(
      this.judges.map((judge) => judge.evaluate(input, output))
    );

    // 检查一致性
    const consistency = this.checkConsistency(evaluations);

    // 如果高度一致，直接返回共识结果
    if (consistency.isHighlyConsistent) {
      return {
        method: "consensus",
        finalScore: consistency.averageScore,
        evaluations,
        confidence: consistency.confidence,
        needsHumanReview: false,
      };
    }

    // 如果不一致，进行仲裁
    const adjudication = await this.resolveDisagreement(
      input,
      output,
      evaluations
    );

    // 判断是否需要人工审核
    const needsHumanReview = this.needsHumanReview(adjudication);

    if (needsHumanReview && this.humanReviewer) {
      const humanResult = await this.humanReviewer.review(
        input,
        output,
        evaluations
      );

      return {
        method: "human_adjudication",
        finalScore: humanResult.score,
        evaluations,
        humanReview: humanResult,
        confidence: 1.0,
        needsHumanReview: false,
      };
    }

    return adjudication;
  }

  // 检查评分一致性
  private checkConsistency(evaluations: EvaluationResult[]): ConsistencyCheck {
    const scores = evaluations.map((e) => e.totalScore);
    const mean = scores.reduce((a, b) => a + b, 0) / scores.length;
    const variance =
      scores.reduce((sum, score) => sum + Math.pow(score - mean, 2), 0) /
      scores.length;
    const standardDeviation = Math.sqrt(variance);

    // 变异系数（CV）
    const coefficientOfVariation = standardDeviation / mean;

    return {
      averageScore: mean,
      standardDeviation,
      coefficientOfVariation,
      isHighlyConsistent: coefficientOfVariation < 0.1, // CV < 10% 视为高度一致
      confidence: 1 - coefficientOfVariation,
    };
  }

  // 解决分歧
  private async resolveDisagreement(
    input: string,
    output: string,
    evaluations: EvaluationResult[]
  ): Promise<AdjudicationResult> {
    switch (this.config.resolutionStrategy) {
      case "majority":
        return this.majorityVote(evaluations);

      case "weighted":
        return this.weightedAverage(evaluations);

      case "meta_judge":
        return this.metaJudgeAdjudication(input, output, evaluations);

      case "highest_confidence":
        return this.highestConfidence(evaluations);

      default:
        return this.weightedAverage(evaluations);
    }
  }

  // 多数投票
  private majorityVote(evaluations: EvaluationResult[]): AdjudicationResult {
    // 将分数分桶
    const buckets = new Map<number, EvaluationResult[]>();
    for (const eval_ of evaluations) {
      const roundedScore = Math.round(eval_.totalScore);
      const bucket = buckets.get(roundedScore) || [];
      bucket.push(eval_);
      buckets.set(roundedScore, bucket);
    }

    // 找到多数票
    let majorityScore = 0;
    let majorityCount = 0;
    for (const [score, bucket] of buckets) {
      if (bucket.length > majorityCount) {
        majorityScore = score;
        majorityCount = bucket.length;
      }
    }

    return {
      method: "majority_vote",
      finalScore: majorityScore,
      evaluations,
      confidence: majorityCount / evaluations.length,
      needsHumanReview: majorityCount / evaluations.length < 0.5,
    };
  }

  // 加权平均
  private weightedAverage(evaluations: EvaluationResult[]): AdjudicationResult {
    const weights = this.config.judgeWeights || evaluations.map(() => 1);
    const totalWeight = weights.reduce((a, b) => a + b, 0);

    const weightedSum = evaluations.reduce(
      (sum, eval_, index) => sum + eval_.totalScore * weights[index],
      0
    );

    return {
      method: "weighted_average",
      finalScore: weightedSum / totalWeight,
      evaluations,
      confidence: this.calculateConfidence(evaluations),
      needsHumanReview: false,
    };
  }

  // 元评委仲裁
  private async metaJudgeAdjudication(
    input: string,
    output: string,
    evaluations: EvaluationResult[]
  ): Promise<AdjudicationResult> {
    // 使用另一个 LLM 作为元评委
    const metaJudgePrompt = this.buildMetaJudgePrompt(
      input,
      output,
      evaluations
    );

    const metaJudgeLLM = this.config.metaJudgeLLM;
    const metaResult = await metaJudgeLLM.invoke(metaJudgePrompt);

    const parsed = JSON.parse(metaResult);

    return {
      method: "meta_judge",
      finalScore: parsed.score,
      evaluations,
      metaJudgment: parsed.reasoning,
      confidence: parsed.confidence,
      needsHumanReview: parsed.confidence < 0.7,
    };
  }

  // 判断是否需要人工审核
  private needsHumanReview(result: AdjudicationResult): boolean {
    // 低置信度需要人工审核
    if (result.confidence < this.config.humanReviewThreshold) {
      return true;
    }

    // 临界分数需要人工审核
    const passingScore = this.config.passingScore;
    const margin = this.config.criticalMargin;
    if (
      Math.abs(result.finalScore - passingScore) < margin
    ) {
      return true;
    }

    return false;
  }
}

// 裁决配置
interface AdjudicationConfig {
  judges: LLMJudge[];
  humanReviewer?: HumanReviewer;
  resolutionStrategy: "majority" | "weighted" | "meta_judge" | "highest_confidence";
  judgeWeights?: number[];
  metaJudgeLLM?: BaseLanguageModel;
  humanReviewThreshold: number;
  passingScore: number;
  criticalMargin: number;
}
```

## 流式事件

在评分过程中，流式事件（Streaming Events）可以实时报告评估进度和结果。

```typescript
// 评分流式事件类型
interface RubricStreamEvent {
  type: RubricEventType;
  timestamp: Date;
  data: any;
}

type RubricEventType =
  | "evaluation_started"
  | "dimension_evaluating"
  | "dimension_completed"
  | "evaluation_completed"
  | "adjudication_started"
  | "adjudication_completed"
  | "improvement_started"
  | "improvement_completed"
  | "error";

// 流式评分器
class StreamingRubricEvaluator {
  private judge: LLMJudge;
  private eventEmitter: EventEmitter;

  constructor(judge: LLMJudge) {
    this.judge = judge;
    this.eventEmitter = new EventEmitter();
  }

  // 流式评估
  async *evaluateStream(
    input: string,
    output: string
  ): AsyncGenerator<RubricStreamEvent> {
    // 发出开始事件
    yield {
      type: "evaluation_started",
      timestamp: new Date(),
      data: { input, output },
    };

    const dimensionScores: DimensionScore[] = [];

    // 逐维度评估
    for (const dimension of this.judge.rubric.dimensions) {
      // 发出维度开始事件
      yield {
        type: "dimension_evaluating",
        timestamp: new Date(),
        data: {
          dimension: dimension.name,
          weight: dimension.weight,
        },
      };

      try {
        // 评估该维度
        const score = await this.judge.evaluateDimension(
          dimension,
          input,
          output
        );

        dimensionScores.push({
          dimension: dimension.name,
          score: score.score,
          reasoning: score.reasoning,
          evidence: score.evidence,
        });

        // 发出维度完成事件
        yield {
          type: "dimension_completed",
          timestamp: new Date(),
          data: {
            dimension: dimension.name,
            score: score.score,
            reasoning: score.reasoning,
          },
        };
      } catch (error) {
        // 发出错误事件
        yield {
          type: "error",
          timestamp: new Date(),
          data: {
            dimension: dimension.name,
            error: error.message,
          },
        };
      }
    }

    // 计算总分
    const totalScore = this.calculateWeightedScore(dimensionScores);

    // 发出评估完成事件
    yield {
      type: "evaluation_completed",
      timestamp: new Date(),
      data: {
        totalScore,
        dimensionScores,
        passing: totalScore >= this.judge.rubric.passingScore,
      },
    };
  }

  // 订阅事件
  on(event: RubricEventType, handler: (event: RubricStreamEvent) => void) {
    this.eventEmitter.on(event, handler);
  }
}

// 前端使用示例
async function displayEvaluationProgress(
  evaluator: StreamingRubricEvaluator,
  input: string,
  output: string
) {
  const progressBar = document.getElementById("evaluation-progress");
  const dimensionList = document.getElementById("dimension-scores");

  for await (const event of evaluator.evaluateStream(input, output)) {
    switch (event.type) {
      case "evaluation_started":
        progressBar.textContent = "开始评估...";
        break;

      case "dimension_evaluating":
        const dimItem = document.createElement("li");
        dimItem.id = `dim-${event.data.dimension}`;
        dimItem.textContent = `${event.data.dimension}: 评估中...`;
        dimensionList.appendChild(dimItem);
        break;

      case "dimension_completed":
        const item = document.getElementById(`dim-${event.data.dimension}`);
        item.textContent = `${event.data.dimension}: ${event.data.score}/5`;
        item.className = event.data.score >= 3 ? "pass" : "fail";
        break;

      case "evaluation_completed":
        progressBar.textContent = `评估完成！总分: ${event.data.totalScore.toFixed(2)}/5`;
        progressBar.className = event.data.passing ? "success" : "failure";
        break;

      case "error":
        const errorItem = document.getElementById(`dim-${event.data.dimension}`);
        errorItem.textContent = `${event.data.dimension}: 评估失败`;
        errorItem.className = "error";
        break;
    }
  }
}
```

```typescript
class CodeGenerationEvaluator {
  private judge: LLMJudge;
  private selfEvalLoop: SelfEvaluationLoop;

  constructor(llm: BaseLanguageModel) {
    this.judge = new LLMJudge(llm, codeGenerationRubric);
    this.selfEvalLoop = new SelfEvaluationLoop(
      codeGenerationAgent,
      this.judge,
      {
        targetScore: 4.0,
        maxIterations: 3,
        earlyStopping: true,
        patience: 2,
      }
    );
  }

  // 评估代码生成结果
  async evaluateCodeGeneration(
    requirements: string,
    generatedCode: string
  ): Promise<CodeGenerationEvaluation> {
    // 基础评估
    const evaluation = await this.judge.evaluate(
      requirements,
      generatedCode
    );

    // 代码特定的额外检查
    const additionalChecks = await this.performAdditionalChecks(generatedCode);

    // 综合评估结果
    return {
      ...evaluation,
      additionalChecks,
      recommendations: this.generateRecommendations(evaluation, additionalChecks),
    };
  }

  // 执行自评估改进
  async improveWithSelfEvaluation(
    requirements: string
  ): Promise<SelfEvalResult> {
    return this.selfEvalLoop.execute(requirements);
  }

  // 额外的代码检查
  private async performAdditionalChecks(
    code: string
  ): Promise<AdditionalChecks> {
    return {
      // 语法检查
      syntaxValid: await this.checkSyntax(code),
      // 依赖检查
      dependencies: await this.checkDependencies(code),
      // 安全检查
      securityIssues: await this.checkSecurity(code),
      // 性能检查
      performanceIssues: await this.checkPerformance(code),
    };
  }

  // 生成改进建议
  private generateRecommendations(
    evaluation: EvaluationResult,
    checks: AdditionalChecks
  ): string[] {
    const recommendations: string[] = [];

    // 基于评分维度的建议
    for (const ds of evaluation.dimensionScores) {
      if (ds.score < 3) {
        recommendations.push(
          `改进 ${ds.dimension}: ${ds.reasoning}`
        );
      }
    }

    // 基于额外检查的建议
    if (!checks.syntaxValid) {
      recommendations.push("修复语法错误");
    }
    if (checks.securityIssues.length > 0) {
      recommendations.push(`解决安全问题: ${checks.securityIssues.join(", ")}`);
    }
    if (checks.performanceIssues.length > 0) {
      recommendations.push(`优化性能问题: ${checks.performanceIssues.join(", ")}`);
    }

    return recommendations;
  }
}
```
