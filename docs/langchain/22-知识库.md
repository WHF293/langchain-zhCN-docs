## 概述

知识库（Knowledge Base）是存储和管理结构化知识的系统，为 RAG 提供数据支持。本章将介绍知识库的创建、更新、检索和管理。

### 混合知识库

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { BM25Retriever } from "langchain/retrievers/bm25";
import { EnsembleRetriever } from "langchain/retrievers/ensemble";
import { Document } from "@langchain/core/documents";

// 混合知识库 - 结合向量检索和关键词检索
class HybridKnowledgeBase {
  private vectorStore: MemoryVectorStore;
  private bm25Retriever: BM25Retriever;
  private ensembleRetriever: EnsembleRetriever;

  constructor(
    private embeddings: OpenAIEmbeddings,
    private documents: Document[]
  ) {}

  async initialize(): Promise<void> {
    // 初始化向量存储
    this.vectorStore = await MemoryVectorStore.fromDocuments(
      this.documents,
      this.embeddings
    );

    // 初始化 BM25 检索器
    this.bm25Retriever = BM25Retriever.fromDocuments(this.documents, {
      k: 10
    });

    // 创建集成检索器
    const vectorRetriever = this.vectorStore.asRetriever({ k: 10 });

    this.ensembleRetriever = new EnsembleRetriever({
      retrievers: [vectorRetriever, this.bm25Retriever],
      weights: [0.6, 0.4]  // 向量检索权重 60%，BM25 权重 40%
    });
  }

  async search(query: string, k: number = 5): Promise<Document[]> {
    if (!this.ensembleRetriever) {
      await this.initialize();
    }

    return this.ensembleRetriever.invoke(query);
  }
}

// 使用混合知识库
const hybridKB = new HybridKnowledgeBase(
  new OpenAIEmbeddings(),
  documents
);

const results = await hybridKB.search("机器学习教程");
```

### 分层知识库

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

// 分层知识库 - 按主题或类别组织
class HierarchicalKnowledgeBase {
  // 主题 -> 向量存储的映射
  private topicStores: Map<string, MemoryVectorStore> = new Map();

  // 主题元数据
  private topicMetadata: Map<string, {
    description: string;
    documentCount: number;
    lastUpdated: Date;
  }> = new Map();

  constructor(private embeddings: OpenAIEmbeddings) {}

  // 添加主题
  async addTopic(
    topicName: string,
    description: string,
    documents: Document[]
  ): Promise<void> {
    // 为该主题创建向量存储
    const vectorStore = await MemoryVectorStore.fromDocuments(
      documents,
      this.embeddings
    );

    this.topicStores.set(topicName, vectorStore);
    this.topicMetadata.set(topicName, {
      description,
      documentCount: documents.length,
      lastUpdated: new Date()
    });
  }

  // 在特定主题中搜索
  async searchInTopic(
    topicName: string,
    query: string,
    k: number = 5
  ): Promise<Document[]> {
    const store = this.topicStores.get(topicName);

    if (!store) {
      throw new Error(`主题 "${topicName}" 不存在`);
    }

    return store.similaritySearch(query, k);
  }

  // 跨主题搜索
  async searchAcrossTopics(
    query: string,
    k: number = 5
  ): Promise<Array<{ topic: string; documents: Document[] }>> {
    const results: Array<{ topic: string; documents: Document[] }> = [];

    for (const [topicName, store] of this.topicStores) {
      const docs = await store.similaritySearch(query, Math.ceil(k / this.topicStores.size));

      if (docs.length > 0) {
        results.push({ topic: topicName, documents: docs });
      }
    }

    return results;
  }

  // 获取所有主题
  getTopics(): Array<{ name: string; description: string; documentCount: number }> {
    const topics: Array<{ name: string; description: string; documentCount: number }> = [];

    for (const [name, metadata] of this.topicMetadata) {
      topics.push({
        name,
        description: metadata.description,
        documentCount: metadata.documentCount
      });
    }

    return topics;
  }

  // 自动分类主题
  async classifyTopic(query: string): Promise<string> {
    // 简单实现：找到与查询最相关的主题
    let bestTopic = "";
    let bestScore = -Infinity;

    for (const [topicName, store] of this.topicStores) {
      const results = await store.similaritySearchWithScore(query, 1);

      if (results.length > 0 && results[0][1] > bestScore) {
        bestScore = results[0][1];
        bestTopic = topicName;
      }
    }

    return bestTopic;
  }
}

// 使用分层知识库
const hierarchicalKB = new HierarchicalKnowledgeBase(new OpenAIEmbeddings());

// 添加不同主题的知识
await hierarchicalKB.addTopic(
  "机器学习",
  "机器学习相关的文档和教程",
  mlDocuments
);

await hierarchicalKB.addTopic(
  "深度学习",
  "深度学习相关的文档和教程",
  dlDocuments
);

await hierarchicalKB.addTopic(
  "自然语言处理",
  "NLP 相关的文档和教程",
  nlpDocuments
);

// 搜索特定主题
const mlResults = await hierarchicalKB.searchInTopic("机器学习", "监督学习");

// 跨主题搜索
const crossResults = await hierarchicalKB.searchAcrossTopics("神经网络");
```

## 知识库更新

### 增量更新

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { Chroma } from "@langchain/community/vectorstores/chroma";
import { Document } from "@langchain/core/documents";

// 知识库更新管理器
class KnowledgeBaseUpdater {
  private vectorStore: Chroma;

  constructor(vectorStore: Chroma) {
    this.vectorStore = vectorStore;
  }

  // 添加新文档
  async addDocuments(documents: Document[]): Promise<void> {
    await this.vectorStore.addDocuments(documents);
    console.log(`成功添加 ${documents.length} 个文档`);
  }

  // 更新文档
  async updateDocument(
    documentId: string,
    newContent: string,
    newMetadata?: Record<string, any>
  ): Promise<void> {
    // 删除旧文档
    await this.deleteDocument(documentId);

    // 添加新文档
    const newDoc = new Document({
      pageContent: newContent,
      metadata: {
        ...newMetadata,
        id: documentId,
        updatedAt: new Date().toISOString()
      }
    });

    await this.vectorStore.addDocuments([newDoc]);
    console.log(`成功更新文档 ${documentId}`);
  }

  // 删除文档
  async deleteDocument(documentId: string): Promise<void> {
    // 根据元数据删除
    await this.vectorStore.delete({
      filter: { id: documentId }
    });
    console.log(`成功删除文档 ${documentId}`);
  }

  // 批量更新
  async batchUpdate(operations: Array<{
    type: "add" | "update" | "delete";
    documentId?: string;
    document?: Document;
    content?: string;
    metadata?: Record<string, any>;
  }>): Promise<void> {
    for (const op of operations) {
      switch (op.type) {
        case "add":
          if (op.document) {
            await this.addDocuments([op.document]);
          }
          break;
        case "update":
          if (op.documentId && op.content) {
            await this.updateDocument(op.documentId, op.content, op.metadata);
          }
          break;
        case "delete":
          if (op.documentId) {
            await this.deleteDocument(op.documentId);
          }
          break;
      }
    }

    console.log(`批量更新完成，共处理 ${operations.length} 个操作`);
  }
}
```

### 版本控制

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

// 带版本控制的知识库
class VersionedKnowledgeBase {
  private currentVersion: number = 0;
  private versions: Map<number, {
    timestamp: Date;
    documents: Document[];
    description: string;
  }> = new Map();

  private vectorStore: MemoryVectorStore | null = null;
  private embeddings: OpenAIEmbeddings;

  constructor(embeddings: OpenAIEmbeddings) {
    this.embeddings = embeddings;
  }

  // 创建新版本
  async createVersion(
    documents: Document[],
    description: string
  ): Promise<number> {
    this.currentVersion++;

    this.versions.set(this.currentVersion, {
      timestamp: new Date(),
      documents: [...documents],
      description
    });

    // 更新向量存储
    this.vectorStore = await MemoryVectorStore.fromDocuments(
      documents,
      this.embeddings
    );

    console.log(`创建版本 ${this.currentVersion}: ${description}`);
    return this.currentVersion;
  }

  // 获取当前版本
  getCurrentVersion(): number {
    return this.currentVersion;
  }

  // 获取版本信息
  getVersionInfo(version: number): {
    timestamp: Date;
    description: string;
    documentCount: number;
  } | null {
    const versionData = this.versions.get(version);

    if (!versionData) {
      return null;
    }

    return {
      timestamp: versionData.timestamp,
      description: versionData.description,
      documentCount: versionData.documents.length
    };
  }

  // 切换到指定版本
  async switchVersion(version: number): Promise<boolean> {
    const versionData = this.versions.get(version);

    if (!versionData) {
      console.error(`版本 ${version} 不存在`);
      return false;
    }

    this.currentVersion = version;
    this.vectorStore = await MemoryVectorStore.fromDocuments(
      versionData.documents,
      this.embeddings
    );

    console.log(`切换到版本 ${version}`);
    return true;
  }

  // 获取所有版本
  getVersions(): Array<{
    version: number;
    timestamp: Date;
    description: string;
    documentCount: number;
  }> {
    const versions: Array<{
      version: number;
      timestamp: Date;
      description: string;
      documentCount: number;
    }> = [];

    for (const [version, data] of this.versions) {
      versions.push({
        version,
        timestamp: data.timestamp,
        description: data.description,
        documentCount: data.documents.length
      });
    }

    return versions.sort((a, b) => b.version - a.version);
  }

  // 搜索当前版本
  async search(query: string, k: number = 5): Promise<Document[]> {
    if (!this.vectorStore) {
      throw new Error("知识库未初始化");
    }

    return this.vectorStore.similaritySearch(query, k);
  }
}

// 使用版本控制知识库
const versionedKB = new VersionedKnowledgeBase(new OpenAIEmbeddings());

// 创建初始版本
await versionedKB.createVersion(initialDocuments, "初始版本");

// 添加新文档后创建新版本
await versionedKB.createVersion(updatedDocuments, "添加了新文档");

// 查看版本历史
const versions = versionedKB.getVersions();
console.log("版本历史:", versions);

// 切换到旧版本
await versionedKB.switchVersion(1);
```

## 知识库检索

### 多知识库检索

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

// 多知识库检索器
class MultiKnowledgeBaseRetriever {
  private knowledgeBases: Map<string, MemoryVectorStore> = new Map();
  private embeddings: OpenAIEmbeddings;

  constructor(embeddings: OpenAIEmbeddings) {
    this.embeddings = embeddings;
  }

  // 添加知识库
  async addKnowledgeBase(
    name: string,
    documents: Document[]
  ): Promise<void> {
    const vectorStore = await MemoryVectorStore.fromDocuments(
      documents,
      this.embeddings
    );

    this.knowledgeBases.set(name, vectorStore);
    console.log(`添加知识库: ${name} (${documents.length} 个文档)`);
  }

  // 在指定知识库中搜索
  async searchIn(
    kbName: string,
    query: string,
    k: number = 5
  ): Promise<Document[]> {
    const kb = this.knowledgeBases.get(kbName);

    if (!kb) {
      throw new Error(`知识库 "${kbName}" 不存在`);
    }

    return kb.similaritySearch(query, k);
  }

  // 跨所有知识库搜索
  async searchAll(
    query: string,
    k: number = 5
  ): Promise<Array<{ kbName: string; documents: Document[] }>> {
    const results: Array<{ kbName: string; documents: Document[] }> = [];

    for (const [name, kb] of this.knowledgeBases) {
      const docs = await kb.similaritySearch(
        query,
        Math.ceil(k / this.knowledgeBases.size)
      );

      if (docs.length > 0) {
        results.push({ kbName: name, documents: docs });
      }
    }

    return results;
  }

  // 智能搜索 - 自动选择最相关的知识库
  async smartSearch(
    query: string,
    k: number = 5
  ): Promise<Document[]> {
    // 首先确定最相关的知识库
    let bestKB = "";
    let bestScore = -Infinity;

    for (const [name, kb] of this.knowledgeBases) {
      const results = await kb.similaritySearchWithScore(query, 1);

      if (results.length > 0 && results[0][1] > bestScore) {
        bestScore = results[0][1];
        bestKB = name;
      }
    }

    console.log(`智能搜索选择知识库: ${bestKB}`);

    // 在最相关的知识库中搜索
    return this.searchIn(bestKB, query, k);
  }

  // 获取知识库列表
  getKnowledgeBases(): Array<{ name: string; documentCount: number }> {
    const kbs: Array<{ name: string; documentCount: number }> = [];

    for (const [name, kb] of this.knowledgeBases) {
      kbs.push({
        name,
        documentCount: (kb as any).memoryVectors?.length || 0
      });
    }

    return kbs;
  }
}

// 使用多知识库检索
const multiKBRetriever = new MultiKnowledgeBaseRetriever(new OpenAIEmbeddings());

await multiKBRetriever.addKnowledgeBase("产品文档", productDocs);
await multiKBRetriever.addKnowledgeBase("技术文档", techDocs);
await multiKBRetriever.addKnowledgeBase("常见问题", faqDocs);

// 智能搜索
const results = await multiKBRetriever.smartSearch("如何配置数据库？");
```

### 知识库查询优化

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

// 查询优化器
class QueryOptimizer {
  private model: ChatOpenAI;

  constructor() {
    this.model = new ChatOpenAI({ modelName: "gpt-4" });
  }

  // 查询重写
  async rewriteQuery(originalQuery: string): Promise<string[]> {
    const response = await this.model.invoke(`
请将以下查询重写为3个不同的版本，以提高检索效果。
每个版本占一行，不要编号。

原始查询: ${originalQuery}
`);

    const rewrittenQueries = (response as string)
      .trim()
      .split('\n')
      .filter(q => q.trim().length > 0);

    return [originalQuery, ...rewrittenQueries];
  }

  // 查询分解
  async decomposeQuery(complexQuery: string): Promise<string[]> {
    const response = await this.model.invoke(`
请将以下复杂查询分解为多个简单的子查询。
每个子查询占一行，不要编号。

复杂查询: ${complexQuery}
`);

    return (response as string)
      .trim()
      .split('\n')
      .filter(q => q.trim().length > 0);
  }

  // 查询扩展
  async expandQuery(query: string): Promise<string[]> {
    const response = await this.model.invoke(`
请为以下查询生成相关的同义词和扩展词。
每个词占一行，不要编号。

查询: ${query}
`);

    return [query, ...(response as string)
      .trim()
      .split('\n')
      .filter(q => q.trim().length > 0)
    ];
  }
}

// 优化后的检索
class OptimizedRetriever {
  private vectorStore: MemoryVectorStore;
  private optimizer: QueryOptimizer;

  constructor(vectorStore: MemoryVectorStore) {
    this.vectorStore = vectorStore;
    this.optimizer = new QueryOptimizer();
  }

  async search(query: string, k: number = 5): Promise<Document[]> {
    // 1. 重写查询
    const queries = await this.optimizer.rewriteQuery(query);

    // 2. 对每个查询进行检索
    const allResults: Document[] = [];

    for (const q of queries) {
      const results = await this.vectorStore.similaritySearch(q, k);
      allResults.push(...results);
    }

    // 3. 去重
    const seen = new Set<string>();
    const uniqueResults = allResults.filter(doc => {
      const key = doc.pageContent;
      if (seen.has(key)) {
        return false;
      }
      seen.add(key);
      return true;
    });

    return uniqueResults.slice(0, k);
  }

  async searchWithDecomposition(
    complexQuery: string,
    k: number = 5
  ): Promise<Document[]> {
    // 1. 分解查询
    const subQueries = await this.optimizer.decomposeQuery(complexQuery);

    // 2. 对每个子查询进行检索
    const allResults: Document[] = [];

    for (const q of subQueries) {
      const results = await this.vectorStore.similaritySearch(q, Math.ceil(k / subQueries.length));
      allResults.push(...results);
    }

    // 3. 去重并返回
    const seen = new Set<string>();
    return allResults.filter(doc => {
      const key = doc.pageContent;
      if (seen.has(key)) {
        return false;
      }
      seen.add(key);
      return true;
    }).slice(0, k);
  }
}
```

### 知识库监控

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

// 知识库监控器
class KnowledgeBaseMonitor {
  private metrics: {
    totalDocuments: number;
    totalQueries: number;
    averageQueryTime: number;
    cacheHitRate: number;
    lastUpdated: Date;
  };

  private queryCache: Map<string, { result: Document[]; timestamp: Date }> = new Map();
  private queryTimes: number[] = [];

  constructor() {
    this.metrics = {
      totalDocuments: 0,
      totalQueries: 0,
      averageQueryTime: 0,
      cacheHitRate: 0,
      lastUpdated: new Date()
    };
  }

  // 记录查询
  recordQuery(duration: number, cacheHit: boolean): void {
    this.metrics.totalQueries++;
    this.queryTimes.push(duration);

    // 更新平均查询时间
    this.metrics.averageQueryTime =
      this.queryTimes.reduce((a, b) => a + b, 0) / this.queryTimes.length;

    // 更新缓存命中率
    if (cacheHit) {
      const totalHits = this.metrics.cacheHitRate * (this.metrics.totalQueries - 1) + 1;
      this.metrics.cacheHitRate = totalHits / this.metrics.totalQueries;
    } else {
      this.metrics.cacheHitRate =
        (this.metrics.cacheHitRate * (this.metrics.totalQueries - 1)) /
        this.metrics.totalQueries;
    }

    this.metrics.lastUpdated = new Date();
  }

  // 记录文档添加
  recordDocumentAdd(count: number): void {
    this.metrics.totalDocuments += count;
    this.metrics.lastUpdated = new Date();
  }

  // 记录文档删除
  recordDocumentDelete(count: number): void {
    this.metrics.totalDocuments -= count;
    this.metrics.lastUpdated = new Date();
  }

  // 获取指标
  getMetrics(): typeof this.metrics {
    return { ...this.metrics };
  }

  // 生成报告
  generateReport(): string {
    const metrics = this.getMetrics();

    return `
知识库监控报告
==============
总文档数: ${metrics.totalDocuments}
总查询数: ${metrics.totalQueries}
平均查询时间: ${metrics.averageQueryTime.toFixed(2)}ms
缓存命中率: ${(metrics.cacheHitRate * 100).toFixed(2)}%
最后更新: ${metrics.lastUpdated.toISOString()}
`;
  }
}

// 使用监控器
const monitor = new KnowledgeBaseMonitor();

// 记录文档添加
monitor.recordDocumentAdd(100);

// 记录查询
const startTime = Date.now();
const results = await vectorStore.similaritySearch(query, 5);

monitor.recordQuery(duration, false);

// 生成报告
```

### 知识库备份

```typescript
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";
import * as fs from "fs";

// 知识库备份管理器
class KnowledgeBaseBackup {
  private backupPath: string;

  constructor(backupPath: string) {
    this.backupPath = backupPath;
  }

  // 备份知识库
  async backup(
    name: string,
    documents: Document[]
  ): Promise<string> {
    const timestamp = new Date().toISOString().replace(/[:.]/g, "-");
    const filename = `${name}_${timestamp}.json`;
    const filepath = `${this.backupPath}/${filename}`;

    const backupData = {
      name,
      timestamp: new Date().toISOString(),
      documentCount: documents.length,
      documents: documents.map(doc => ({
        pageContent: doc.pageContent,
        metadata: doc.metadata
      }))
    };

    fs.writeFileSync(filepath, JSON.stringify(backupData, null, 2));

    console.log(`备份完成: ${filepath}`);
    console.log(`备份了 ${documents.length} 个文档`);

    return filepath;
  }

  // 恢复知识库
  async restore(filepath: string): Promise<Document[]> {
    const backupData = JSON.parse(fs.readFileSync(filepath, "utf-8"));

    const documents = backupData.documents.map((doc: any) => new Document({
      pageContent: doc.pageContent,
      metadata: doc.metadata
    }));

    console.log(`恢复完成: ${filepath}`);
    console.log(`恢复了 ${documents.length} 个文档`);

    return documents;
  }

  // 列出所有备份
  listBackups(): Array<{
    filename: string;
    name: string;
    timestamp: string;
    documentCount: number;
  }> {
    const files = fs.readdirSync(this.backupPath);

    return files
      .filter(file => file.endsWith(".json"))
      .map(file => {
        const data = JSON.parse(
          fs.readFileSync(`${this.backupPath}/${file}`, "utf-8")
        );

        return {
          filename: file,
          name: data.name,
          timestamp: data.timestamp,
          documentCount: data.documentCount
        };
      })
      .sort((a, b) =>
        new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime()
      );
  }

  // 删除旧备份
  cleanup(keepCount: number = 5): void {
    const backups = this.listBackups();

    if (backups.length <= keepCount) {
      return;
    }

    const toDelete = backups.slice(keepCount);

    for (const backup of toDelete) {
      fs.unlinkSync(`${this.backupPath}/${backup.filename}`);
      console.log(`删除旧备份: ${backup.filename}`);
    }
  }
}

const backupManager = new KnowledgeBaseBackup("./backups");
```
