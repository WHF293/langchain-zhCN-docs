## 概述

长期记忆 (Long-term Memory) 是指跨会话持久化存储和检索信息的能力。与短期记忆不同，长期记忆允许 AI 应用在多次对话之间记住用户偏好、历史事实和重要信息。这对于构建个性化和连贯的用户体验至关重要。

## 持久化记忆

### 基于文件的持久化

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { BaseMessage, HumanMessage, AIMessage } from "@langchain/core/messages";
import fs from "fs/promises";

interface MemoryEntry {
  id: string;
  content: string;
  timestamp: number;
  metadata: Record<string, any>;
  embedding?: number[];
}

class FileBasedMemory {
  private filePath: string;
  private memories: MemoryEntry[] = [];
  private model: ChatOpenAI;

  constructor(filePath: string) {
    this.filePath = filePath;
    this.model = new ChatOpenAI({ modelName: "gpt-4", temperature: 0 });
  }

  async initialize() {
    try {
      const data = await fs.readFile(this.filePath, "utf-8");
      this.memories = JSON.parse(data);
    } catch {
      this.memories = [];
    }
  }

  async saveMemory(content: string, metadata: Record<string, any> = {}) {
    const entry: MemoryEntry = {
      id: Date.now().toString(),
      content,
      timestamp: Date.now(),
      metadata,
    };

    this.memories.push(entry);
    await this.persist();

    return entry;
  }

  async getMemories(filter?: (memory: MemoryEntry) => boolean): Promise<MemoryEntry[]> {
    if (filter) {
      return this.memories.filter(filter);
    }
    return [...this.memories];
  }

  async deleteMemory(id: string) {
    this.memories = this.memories.filter((m) => m.id !== id);
    await this.persist();
  }

  async updateMemory(id: string, updates: Partial<MemoryEntry>) {
    const index = this.memories.findIndex((m) => m.id === id);
    if (index !== -1) {
      this.memories[index] = { ...this.memories[index], ...updates };
      await this.persist();
    }
  }

  private async persist() {
    await fs.writeFile(this.filePath, JSON.stringify(this.memories, null, 2));
  }

  async clear() {
    this.memories = [];
    await this.persist();
  }
}

// 使用示例
const memory = new FileBasedMemory("./memories.json");
await memory.initialize();

// 保存记忆
await memory.saveMemory("用户喜欢 TypeScript", { type: "preference" });
await memory.saveMemory("用户在北京工作", { type: "personal_info" });

// 检索记忆
const preferences = await memory.getMemories((m) => m.metadata.type === "preference");
console.log("用户偏好:", preferences);
```

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { Pool } from "pg"; // PostgreSQL

interface DatabaseMemoryConfig {
  host: string;
  port: number;
  database: string;
  user: string;
  password: string;
}

class DatabaseMemory {
  private pool: Pool;
  private tableName: string;

  constructor(config: DatabaseMemoryConfig, tableName: string = "memories") {
    this.pool = new Pool(config);
    this.tableName = tableName;
  }

  async initialize() {
    await this.pool.query(`
      CREATE TABLE IF NOT EXISTS ${this.tableName} (
        id SERIAL PRIMARY KEY,
        user_id VARCHAR(255) NOT NULL,
        content TEXT NOT NULL,
        metadata JSONB DEFAULT '{}',
        embedding vector(1536),
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
        updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
      )
    `);

    // 创建索引
    await this.pool.query(`
      CREATE INDEX IF NOT EXISTS idx_${this.tableName}_user_id
      ON ${this.tableName} (user_id)
    `);
  }

  async saveMemory(
    userId: string,
    content: string,
    metadata: Record<string, any> = {}
  ) {
    const result = await this.pool.query(
      `INSERT INTO ${this.tableName} (user_id, content, metadata)
       VALUES ($1, $2, $3)
       RETURNING *`,
      [userId, content, JSON.stringify(metadata)]
    );

    return result.rows[0];
  }

  async getMemories(
    userId: string,
    limit: number = 100,
    offset: number = 0
  ) {
    const result = await this.pool.query(
      `SELECT * FROM ${this.tableName}
       WHERE user_id = $1
       ORDER BY created_at DESC
       LIMIT $2 OFFSET $3`,
      [userId, limit, offset]
    );

    return result.rows;
  }

  async deleteMemory(id: string) {
    await this.pool.query(
      `DELETE FROM ${this.tableName} WHERE id = $1`,
      [id]
    );
  }

  async searchMemories(userId: string, query: string) {
    // 全文搜索
    const result = await this.pool.query(
      `SELECT * FROM ${this.tableName}
       WHERE user_id = $1 AND content ILIKE $2
       ORDER BY created_at DESC`,
      [userId, `%${query}%`]
    );

    return result.rows;
  }

  async close() {
    await this.pool.end();
  }
}
```

## 记忆检索

### 基于向量的语义检索

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";
import { Document } from "@langchain/core/documents";

class SemanticMemory {
  private vectorStore: MemoryVectorStore;
  private embeddings: OpenAIEmbeddings;
  private model: ChatOpenAI;

  constructor() {
    this.embeddings = new OpenAIEmbeddings();
    this.vectorStore = new MemoryVectorStore(this.embeddings);
    this.model = new ChatOpenAI({ modelName: "gpt-4", temperature: 0 });
  }

  async addMemory(content: string, metadata: Record<string, any> = {}) {
    const doc = new Document({
      pageContent: content,
      metadata: {
        ...metadata,
        timestamp: Date.now(),
      },
    });

    await this.vectorStore.addDocuments([doc]);
  }

  async searchMemories(query: string, k: number = 5) {
    const results = await this.vectorStore.similaritySearchWithScore(query, k);

    return results.map(([doc, score]) => ({
      content: doc.pageContent,
      metadata: doc.metadata,
      relevance: 1 - score, // 转换为相似度分数
    }));
  }

  async searchWithFilter(
    query: string,
    filter: (metadata: Record<string, any>) => boolean,
    k: number = 5
  ) {
    const results = await this.searchMemories(query, k * 2); // 获取更多结果用于过滤
    return results.filter((r) => filter(r.metadata)).slice(0, k);
  }

  async getContextForQuery(query: string, maxMemories: number = 3) {
    const memories = await this.searchMemories(query, maxMemories);

    if (memories.length === 0) {
      return "";
    }

    const memoryText = memories
      .map((m, i) => `${i + 1}. ${m.content}`)
      .join("\n");

    return `相关记忆:\n${memoryText}`;
  }

  async getMemoryCount() {
    // 获取记忆总数
    const results = await this.vectorStore.similaritySearch("", 1000);
    return results.length;
  }

  async clear() {
    this.vectorStore = new MemoryVectorStore(this.embeddings);
  }
}

// 使用示例
const semanticMemory = new SemanticMemory();

// 添加记忆
await semanticMemory.addMemory("用户是一名前端工程师", {
  type: "profession",
});
await semanticMemory.addMemory("用户喜欢使用 React 和 TypeScript", {
  type: "preference",
});
await semanticMemory.addMemory("用户住在北京", {
  type: "location",
});

// 语义检索
const results = await semanticMemory.searchMemories("用户的技术栈");
console.log("语义检索结果:", results);

// 获取上下文
const context = await semanticMemory.getContextForQuery("推荐一些前端框架");
console.log("上下文:", context);
```

### 混合检索策略

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { OpenAIEmbeddings } from "@langchain/openai";
import { MemoryVectorStore } from "langchain/vectorstores/memory";

interface HybridSearchResult {
  content: string;
  metadata: Record<string, any>;
  vectorScore: number;
  keywordScore: number;
  combinedScore: number;
}

class HybridMemory {
  private vectorStore: MemoryVectorStore;
  private embeddings: OpenAIEmbeddings;
  private keywordIndex: Map<string, string[]> = new Map();

  constructor() {
    this.embeddings = new OpenAIEmbeddings();
    this.vectorStore = new MemoryVectorStore(this.embeddings);
  }

  async addMemory(content: string, metadata: Record<string, any> = {}) {
    // 添加到向量存储
    const doc = new Document({ pageContent: content, metadata });
    await this.vectorStore.addDocuments([doc]);

    // 构建关键词索引
    const keywords = this.extractKeywords(content);
    for (const keyword of keywords) {
      if (!this.keywordIndex.has(keyword)) {
        this.keywordIndex.set(keyword, []);
      }
      this.keywordIndex.get(keyword)!.push(content);
    }
  }

  private extractKeywords(text: string): string[] {
    // 简单的关键词提取
    const stopWords = new Set(["的", "了", "是", "在", "和", "有", "我", "你", "他"]);
    return text
      .split(/[\s,，。！？、]+/)
      .filter((word) => word.length > 1 && !stopWords.has(word))
      .map((word) => word.toLowerCase());
  }

  private keywordSearch(query: string): Map<string, number> {
    const keywords = this.extractKeywords(query);
    const scores = new Map<string, number>();

    for (const keyword of keywords) {
      const matchingContents = this.keywordIndex.get(keyword) || [];
      for (const content of matchingContents) {
        scores.set(content, (scores.get(content) || 0) + 1);
      }
    }

    return scores;
  }

  async searchMemories(
    query: string,
    k: number = 5,
    vectorWeight: number = 0.7,
    keywordWeight: number = 0.3
  ): Promise<HybridSearchResult[]> {
    // 向量检索
    const vectorResults = await this.vectorStore.similaritySearchWithScore(query, k * 2);

    // 关键词检索
    const keywordScores = this.keywordSearch(query);

    // 合并结果
    const resultsMap = new Map<string, HybridSearchResult>();

    // 处理向量结果
    for (const [doc, score] of vectorResults) {
      const content = doc.pageContent;
      resultsMap.set(content, {
        content,
        metadata: doc.metadata,
        vectorScore: 1 - score,
        keywordScore: 0,
        combinedScore: 0,
      });
    }

    // 添加关键词分数
    for (const [content, keywordScore] of keywordScores) {
      if (resultsMap.has(content)) {
        resultsMap.get(content)!.keywordScore = keywordScore;
      }
    }

    // 计算综合分数
    const maxKeywordScore = Math.max(...Array.from(keywordScores.values()), 1);

    for (const result of resultsMap.values()) {
      result.keywordScore = result.keywordScore / maxKeywordScore;
      result.combinedScore =
        result.vectorScore * vectorWeight + result.keywordScore * keywordWeight;
    }

    // 排序并返回
    return Array.from(resultsMap.values())
      .sort((a, b) => b.combinedScore - a.combinedScore)
      .slice(0, k);
  }
}

// 使用示例
const hybridMemory = new HybridMemory();

await hybridMemory.addMemory("我喜欢吃火锅", { type: "preference" });
await hybridMemory.addMemory("火锅是中国传统美食", { type: "fact" });
await hybridMemory.addMemory("我经常去海底捞吃火锅", { type: "habit" });

const results = await hybridMemory.searchMemories("火锅");
console.log("混合检索结果:", results);
```

## 记忆更新

### 自动记忆提取

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { HumanMessage, AIMessage } from "@langchain/core/messages";

interface ExtractedMemory {
  content: string;
  type: "fact" | "preference" | "habit" | "relationship" | "goal";
  confidence: number;
  source: string;
}

class AutoMemoryExtractor {
  private model: ChatOpenAI;
  private memoryStore: SemanticMemory;

  constructor(memoryStore: SemanticMemory) {
    this.model = new ChatOpenAI({ modelName: "gpt-4", temperature: 0 });
    this.memoryStore = memoryStore;
  }

  async extractFromConversation(
    userMessage: string,
    aiResponse: string
  ): Promise<ExtractedMemory[]> {
    const prompt = `分析以下对话，提取关于用户的重要信息。

用户: ${userMessage}
助手: ${aiResponse}

请提取以下类型的信息（如果有的话）：
1. 事实 (fact) - 用户的客观信息，如职业、年龄、位置
2. 偏好 (preference) - 用户的喜好和厌恶
3. 习惯 (habit) - 用户的行为模式
4. 关系 (relationship) - 用户提到的其他人或组织
5. 目标 (goal) - 用户想要达成的目标

以 JSON 数组格式返回，每个元素包含:
- content: 信息内容
- type: 信息类型
- confidence: 置信度 (0-1)

如果没有值得记住的信息，返回空数组 []`;

    const response = await this.model.invoke(prompt);
    const content = response.content as string;

    try {
      // 提取 JSON 部分
      const jsonMatch = content.match(/\[[\s\S]*\]/);
      if (jsonMatch) {
        const memories: ExtractedMemory[] = JSON.parse(jsonMatch[0]);
        return memories
          .filter((m) => m.confidence > 0.7) // 只保留高置信度的记忆
          .map((m) => ({
            ...m,
            source: `对话: "${userMessage.substring(0, 50)}..."`,
          }));
      }
    } catch (e) {
      console.error("解析记忆失败:", e);
    }

    return [];
  }

  async processAndStore(
    userMessage: string,
    aiResponse: string
  ): Promise<ExtractedMemory[]> {
    const memories = await this.extractFromConversation(userMessage, aiResponse);

    for (const memory of memories) {
      await this.memoryStore.addMemory(memory.content, {
        type: memory.type,
        confidence: memory.confidence,
        source: memory.source,
        extractedAt: Date.now(),
      });
    }

    return memories;
  }
}

// 使用示例
const semanticMemory = new SemanticMemory();
const extractor = new AutoMemoryExtractor(semanticMemory);

// 模拟对话
const userMsg = "我是一名前端工程师，最近在学习 Rust";
const aiResponse = "很好！Rust 是一门很棒的语言，特别适合系统编程。";

const extracted = await extractor.processAndStore(userMsg, aiResponse);
console.log("提取的记忆:", extracted);

// 之后可以检索
const results = await semanticMemory.searchMemories("用户的职业");
console.log("检索结果:", results);
```

### 记忆合并与去重

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { OpenAIEmbeddings } from "@langchain/openai";

interface MemoryWithSimilarity {
  id: string;
  content: string;
  metadata: Record<string, any>;
  similarMemories: string[];
}

class MemoryManager {
  private model: ChatOpenAI;
  private embeddings: OpenAIEmbeddings;
  private memoryStore: SemanticMemory;

  constructor(memoryStore: SemanticMemory) {
    this.model = new ChatOpenAI({ modelName: "gpt-4", temperature: 0 });
    this.embeddings = new OpenAIEmbeddings();
    this.memoryStore = memoryStore;
  }

  async findSimilarMemories(
    content: string,
    threshold: number = 0.9
  ): Promise<string[]> {
    const results = await this.memoryStore.searchMemories(content, 10);
    return results
      .filter((r) => r.relevance > threshold)
      .map((r) => r.content);
  }

  async mergeMemories(memories: string[]): Promise<string> {
    const prompt = `请将以下相似的记忆合并为一条简洁、完整的记忆：

${memories.map((m, i) => `${i + 1}. ${m}`).join("\n")}

合并后的记忆:`;

    const response = await this.model.invoke(prompt);
    return response.content as string;
  }

  async addMemoryWithDeduplication(
    content: string,
    metadata: Record<string, any> = {}
  ) {
    // 查找相似记忆
    const similar = await this.findSimilarMemories(content);

    if (similar.length > 0) {
      // 合并记忆
      const allMemories = [content, ...similar];
      const merged = await this.mergeMemories(allMemories);

      // 删除旧记忆（简化处理，实际需要更复杂的逻辑）
      // 这里只是添加合并后的记忆
      await this.memoryStore.addMemory(merged, {
        ...metadata,
        merged: true,
        originalCount: allMemories.length,
      });

      return {
        action: "merged",
        mergedContent: merged,
        originalMemories: allMemories,
      };
    } else {
      // 没有相似记忆，直接添加
      await this.memoryStore.addMemory(content, metadata);
      return {
        action: "added",
        content,
      };
    }
  }

  async deduplicateAll() {
    // 获取所有记忆
    const allMemories = await this.memoryStore.searchMemories("", 1000);

    // 按相似度分组
    const groups: string[][] = [];
    const processed = new Set<string>();

    for (const memory of allMemories) {
      if (processed.has(memory.content)) continue;

      const similar = await this.findSimilarMemories(memory.content, 0.85);
      if (similar.length > 1) {
        groups.push(similar);
        similar.forEach((m) => processed.add(m));
      }
    }

    // 合并每组
    const results = [];
    for (const group of groups) {
      const merged = await this.mergeMemories(group);
      results.push({
        original: group,
        merged,
      });
    }

    return results;
  }
}

// 使用示例
const memoryManager = new MemoryManager(semanticMemory);

// 添加可能重复的记忆
await memoryManager.addMemoryWithDeduplication("用户是一名程序员");
await memoryManager.addMemoryWithDeduplication("用户的职业是软件工程师");
await memoryManager.addMemoryWithDeduplication("用户从事编程工作");

// 去重
const deduplicationResults = await memoryManager.deduplicateAll();
console.log("去重结果:", deduplicationResults);
```

## 记忆删除

### 基于时间的过期

```typescript
import { ChatOpenAI } from "@langchain/openai";

interface ExpiringMemory {
  content: string;
  metadata: Record<string, any>;
  createdAt: number;
  expiresAt: number | null; // null 表示永不过期
}

class ExpiringMemoryStore {
  private memories: ExpiringMemory[] = [];
  private defaultTTL: number; // 默认存活时间（毫秒）

  constructor(defaultTTL: number = 30 * 24 * 60 * 60 * 1000) {
    // 默认 30 天
    this.defaultTTL = defaultTTL;
  }

  addMemory(
    content: string,
    metadata: Record<string, any> = {},
    ttl?: number
  ) {
    const now = Date.now();
    this.memories.push({
      content,
      metadata,
      createdAt: now,
      expiresAt: ttl ? now + ttl : now + this.defaultTTL,
    });

    // 清理过期记忆
    this.cleanup();
  }

  addPermanentMemory(content: string, metadata: Record<string, any> = {}) {
    this.memories.push({
      content,
      metadata,
      createdAt: Date.now(),
      expiresAt: null, // 永不过期
    });
  }

  private cleanup() {
    const now = Date.now();
    this.memories = this.memories.filter(
      (m) => m.expiresAt === null || m.expiresAt > now
    );
  }

  getMemories(): ExpiringMemory[] {
    this.cleanup();
    return [...this.memories];
  }

  getExpiredMemories(): ExpiringMemory[] {
    const now = Date.now();
    return this.memories.filter(
      (m) => m.expiresAt !== null && m.expiresAt <= now
    );
  }

  deleteExpired() {
    const now = Date.now();
    this.memories = this.memories.filter(
      (m) => m.expiresAt === null || m.expiresAt > now
    );
  }

  getMemoryCount(): number {
    this.cleanup();
    return this.memories.length;
  }
}

// 使用示例
const expiringStore = new ExpiringMemoryStore();

// 添加 7 天过期的记忆
expiringStore.addMemory(
  "用户正在学习 Python",
  { type: "current_activity" },
  7 * 24 * 60 * 60 * 1000
);

  type: "personal_info",
});

// 添加默认过期时间的记忆
expiringStore.addMemory("用户今天很开心", { type: "mood" });

console.log("记忆数量:", expiringStore.getMemoryCount());
```

### 基于重要性的淘汰

```typescript
import { ChatOpenAI } from "@langchain/openai";

interface ScoredMemory {
  content: string;
  metadata: Record<string, any>;
  importance: number; // 0-10
  accessCount: number;
  lastAccessed: number;
  createdAt: number;
}

class ImportanceBasedMemoryStore {
  private memories: ScoredMemory[] = [];
  private maxMemories: number;
  private model: ChatOpenAI;

  constructor(maxMemories: number = 100) {
    this.maxMemories = maxMemories;
    this.model = new ChatOpenAI({ modelName: "gpt-4", temperature: 0 });
  }

  async addMemory(content: string, metadata: Record<string, any> = {}) {
    const importance = await this.calculateImportance(content);

    this.memories.push({
      content,
      metadata,
      importance,
      accessCount: 0,
      lastAccessed: Date.now(),
      createdAt: Date.now(),
    });

    // 如果超过限制，淘汰不重要的记忆
    if (this.memories.length > this.maxMemories) {
      await this.evictLeastImportant();
    }
  }

  private async calculateImportance(content: string): Promise<number> {
    const prompt = `评估以下信息的重要性（0-10 分）：
- 10 分：关键个人信息（生日、地址、职业）
- 7-9 分：重要偏好和习惯
- 4-6 分：一般信息
- 1-3 分：临时信息

信息: ${content}

只返回数字分数:`;

    const response = await this.model.invoke(prompt);
    const score = parseInt(response.content as string);
    return isNaN(score) ? 5 : Math.max(0, Math.min(10, score));
  }

  private async evictLeastImportant() {
    // 计算综合分数
    const now = Date.now();
    const scoredMemories = this.memories.map((m) => {
      const recencyScore = Math.exp(-((now - m.lastAccessed) / (7 * 24 * 60 * 60 * 1000)));
      const accessScore = Math.log(m.accessCount + 1) / 10;
      const totalScore = m.importance * 0.5 + recencyScore * 3 + accessScore * 2;

      return { ...m, totalScore };
    });

    // 按综合分数排序
    scoredMemories.sort((a, b) => b.totalScore - a.totalScore);

    // 保留前 maxMemories 个
    this.memories = scoredMemories.slice(0, this.maxMemories);
  }

  searchMemories(query: string): ScoredMemory[] {
    const queryLower = query.toLowerCase();
    return this.memories
      .filter((m) => m.content.toLowerCase().includes(queryLower))
      .map((m) => {
        m.accessCount++;
        m.lastAccessed = Date.now();
        return m;
      })
      .sort((a, b) => b.importance - a.importance);
  }

  getMemories(): ScoredMemory[] {
    return [...this.memories];
  }

  getMemoryStats() {
    return {
      total: this.memories.length,
      averageImportance:
        this.memories.reduce((sum, m) => sum + m.importance, 0) / this.memories.length,
      mostAccessed: this.memories.sort((a, b) => b.accessCount - a.accessCount)[0],
      newest: this.memories.sort((a, b) => b.createdAt - a.createdAt)[0],
    };
  }
}

// 使用示例
const importanceStore = new ImportanceBasedMemoryStore(50);

await importanceStore.addMemory("用户的生日是 5 月 15 日");
await importanceStore.addMemory("用户今天午饭吃了面条");
await importanceStore.addMemory("用户是一名数据科学家");

const stats = importanceStore.getMemoryStats();
console.log("记忆统计:", stats);
```

## 最佳实践

1. **选择合适的存储后端**：根据应用规模选择文件、数据库或向量数据库
2. **实现记忆去重**：避免存储重复信息
3. **设置过期策略**：清理不再需要的记忆
4. **保护隐私**：敏感信息需要加密存储
5. **优化检索**：使用向量检索提高语义搜索效果

> **提示**：对于生产环境，推荐使用专门的向量数据库（如 Pinecone、Weaviate、Milvus）来存储和检索记忆。

> **注意**：长期记忆可能包含敏感信息，需要考虑数据保护和隐私合规问题。
