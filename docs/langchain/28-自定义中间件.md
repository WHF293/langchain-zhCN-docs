## 自定义中间件创建

当内置中间件无法满足特定需求时，可以创建自定义中间件。自定义中间件允许开发者实现完全符合业务需求的逻辑。

### 中间件接口

```typescript
// 中间件基础接口
interface Middleware {
  // 中间件名称
  name: string;

  // 优先级（数字越小，优先级越高）
  priority: number;

  // 请求前处理
  onBeforeRequest?(request: Request): Promise<Request>;

  // 响应后处理
  onAfterResponse?(response: Response): Promise<Response>;

  // 错误处理
  onError?(error: Error, request?: Request): Promise<Response | void>;

  // 是否应该执行（条件判断）
  shouldExecute?(request: Request): boolean;

  // 初始化
  onInit?(): Promise<void>;

  // 清理
  onCleanup?(): Promise<void>;
}
```

### 基础中间件模板

```typescript
// 基础中间件模板
class BaseMiddleware implements Middleware {
  name: string;
  priority: number;
  protected options: Record<string, any>;

  constructor(
    name: string,
    priority: number,
    options: Record<string, any> = {}
  ) {
    this.name = name;
    this.priority = priority;
    this.options = options;
  }

  async onBeforeRequest(request: Request): Promise<Request> {
    // 默认实现：直接返回原请求
    return request;
  }

  async onAfterResponse(response: Response): Promise<Response> {
    // 默认实现：直接返回原响应
    return response;
  }

  async onError(error: Error, request?: Request): Promise<Response | void> {
    // 默认实现：抛出错误
    throw error;
  }

  shouldExecute(request: Request): boolean {
    // 默认实现：总是执行
    return true;
  }

  async onInit(): Promise<void> {
    // 默认实现：无操作
  }

  async onCleanup(): Promise<void> {
    // 默认实现：无操作
  }
}
```

### 创建自定义中间件

```typescript
// 示例：请求ID中间件
class RequestIdMiddleware extends BaseMiddleware {
  constructor() {
    super("request-id", 10);  // 高优先级
  }

  async onBeforeRequest(request: Request): Promise<Request> {
    // 生成或获取请求ID
    const requestId =
      request.headers.get("X-Request-ID") ||
      this.generateRequestId();

    // 添加到请求头
    request.headers.set("X-Request-ID", requestId);

    // 存储到请求对象中，供后续使用
    (request as any).requestId = requestId;

    return request;
  }

  async onAfterResponse(response: Response): Promise<Response> {
    // 将请求ID添加到响应头
    const requestId = (response as any).requestId;
    if (requestId) {
      response.headers.set("X-Request-ID", requestId);
    }
    return response;
  }

  private generateRequestId(): string {
    return `req_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

// 示例：请求计时中间件
class TimingMiddleware extends BaseMiddleware {
  constructor() {
    super("timing", 20);
  }

  async onBeforeRequest(request: Request): Promise<Request> {
    // 记录开始时间
    (request as any).startTime = Date.now();
    return request;
  }

  async onAfterResponse(response: Response): Promise<Response> {
    // 计算耗时
    const startTime = (response as any).startTime;
    if (startTime) {
      const duration = Date.now() - startTime;
      response.headers.set("X-Response-Time", `${duration}ms`);

      // 记录到日志
      console.log(`请求耗时: ${duration}ms`);
    }
    return response;
  }
}
```

## 自定义中间件配置

### 配置接口设计

```typescript
// 中间件配置接口
interface MiddlewareConfig {
  // 是否启用
  enabled: boolean;

  // 优先级
  priority?: number;

  // 自定义选项
  options?: Record<string, any>;

  // 条件配置
  conditions?: {
    // 路径匹配
    paths?: string[];
    // HTTP 方法
    methods?: string[];
    // 自定义条件函数
    custom?: (request: Request) => boolean;
  };
}

// 配置验证器
class ConfigValidator {
  static validate(config: MiddlewareConfig): boolean {
    // 验证必填字段
    if (typeof config.enabled !== "boolean") {
      throw new Error("config.enabled 必须是布尔值");
    }

    // 验证优先级范围
    if (config.priority !== undefined) {
      if (config.priority < 0 || config.priority > 1000) {
        throw new Error("priority 必须在 0-1000 之间");
      }
    }

    return true;
  }
}
```

### 带配置的中间件实现

```typescript
// 带配置的中间件基类
class ConfigurableMiddleware extends BaseMiddleware {
  protected config: MiddlewareConfig;

  constructor(
    name: string,
    config: MiddlewareConfig
  ) {
    super(name, config.priority || 200);
    this.config = config;
    ConfigValidator.validate(config);
  }

  shouldExecute(request: Request): boolean {
    // 检查是否启用
    if (!this.config.enabled) {
      return false;
    }

    // 检查路径条件
    if (this.config.conditions?.paths) {
      const url = new URL(request.url);
      const matchesPath = this.config.conditions.paths.some((pattern) => {
        const regex = new RegExp(pattern.replace("*", ".*"));
        return regex.test(url.pathname);
      });
      if (!matchesPath) return false;
    }

    // 检查方法条件
    if (this.config.conditions?.methods) {
      if (!this.config.conditions.methods.includes(request.method)) {
        return false;
      }
    }

    // 检查自定义条件
    if (this.config.conditions?.custom) {
      return this.config.conditions.custom(request);
    }

    return true;
  }

  // 更新配置
  updateConfig(newConfig: Partial<MiddlewareConfig>): void {
    this.config = { ...this.config, ...newConfig };
    ConfigValidator.validate(this.config);
  }

  // 获取当前配置
  getConfig(): MiddlewareConfig {
    return { ...this.config };
  }
}

// 使用示例
class RateLimitMiddleware extends ConfigurableMiddleware {
  private requestCounts: Map<string, number> = new Map();

  constructor(config: MiddlewareConfig) {
    super("rate-limit", config);
  }

  async onBeforeRequest(request: Request): Promise<Request> {
    const clientId = this.getClientId(request);
    const limit = this.config.options?.maxRequests || 100;
    const windowMs = this.config.options?.windowMs || 60000;

    // 获取当前计数
    const count = this.requestCounts.get(clientId) || 0;

    // 检查是否超限
    if (count >= limit) {
      throw new Error("请求频率超限");
    }

    // 增加计数
    this.requestCounts.set(clientId, count + 1);

    // 设置定时重置
    setTimeout(() => {
      this.requestCounts.delete(clientId);
    }, windowMs);

    return request;
  }

  private getClientId(request: Request): string {
    return (
      request.headers.get("X-API-Key") ||
      request.headers.get("X-Forwarded-For") ||
      "unknown"
    );
  }
}
```

### 配置管理器

```typescript
// 中间件配置管理器
class MiddlewareConfigManager {
  private configs: Map<string, MiddlewareConfig> = new Map();
  private middlewares: Map<string, Middleware> = new Map();

  // 注册中间件
  register(middleware: Middleware, config: MiddlewareConfig): void {
    this.configs.set(middleware.name, config);
    this.middlewares.set(middleware.name, middleware);
  }

  // 获取配置
  getConfig(name: string): MiddlewareConfig | undefined {
    return this.configs.get(name);
  }

  // 更新配置
  updateConfig(name: string, config: Partial<MiddlewareConfig>): void {
    const existing = this.configs.get(name);
    if (!existing) {
      throw new Error(`中间件 ${name} 未注册`);
    }

    const updated = { ...existing, ...config };
    this.configs.set(name, updated);

    // 更新中间件实例
    const middleware = this.middlewares.get(name);
    if (middleware && "updateConfig" in middleware) {
      (middleware as any).updateConfig(updated);
    }
  }

  // 批量更新配置
  updateConfigs(configs: Record<string, Partial<MiddlewareConfig>>): void {
    for (const [name, config] of Object.entries(configs)) {
      this.updateConfig(name, config);
    }
  }

  // 获取所有配置
  getAllConfigs(): Record<string, MiddlewareConfig> {
    const configs: Record<string, MiddlewareConfig> = {};
    for (const [name, config] of this.configs) {
      configs[name] = config;
    }
    return configs;
  }
}
```

### 测试框架

```typescript
// 中间件测试工具
class MiddlewareTester {
  // 创建模拟请求
  static createMockRequest(options: {
    method?: string;
    url?: string;
    headers?: Record<string, string>;
    body?: any;
  } = {}): Request {
    const {
      method = "GET",
      url = "http://localhost/test",
      headers = {},
      body = null,
    } = options;

    return new Request(url, {
      method,
      headers: new Headers(headers),
      body: body ? JSON.stringify(body) : null,
    });
  }

  // 创建模拟响应
  static createMockResponse(options: {
    status?: number;
    headers?: Record<string, string>;
    body?: any;
  } = {}): Response {
    const {
      status = 200,
      headers = {},
      body = { success: true },
    } = options;

    return new Response(JSON.stringify(body), {
      status,
      headers: new Headers(headers),
    });
  }

  // 测试中间件
  static async testMiddleware(
    middleware: Middleware,
    request: Request,
    expectedBehavior: {
      shouldModifyRequest?: boolean;
      shouldModifyResponse?: boolean;
      shouldThrowError?: boolean;
    }
  ): Promise<TestResult> {
    const result: TestResult = {
      passed: true,
      errors: [],
    };

    try {
      // 测试 shouldExecute
      if (middleware.shouldExecute) {
        const shouldExecute = middleware.shouldExecute(request);
        console.log(`shouldExecute: ${shouldExecute}`);
      }

      // 测试 onBeforeRequest
      if (middleware.onBeforeRequest) {
        const modifiedRequest = await middleware.onBeforeRequest(request);
        if (expectedBehavior.shouldModifyRequest) {
          // 验证请求是否被修改
          if (modifiedRequest === request) {
            result.errors.push("请求未被修改");
            result.passed = false;
          }
        }
      }

      // 测试 onAfterResponse
      if (middleware.onAfterResponse) {
        const response = MiddlewareTester.createMockResponse();
        const modifiedResponse = await middleware.onAfterResponse(response);
        if (expectedBehavior.shouldModifyResponse) {
          // 验证响应是否被修改
          if (modifiedResponse === response) {
            result.errors.push("响应未被修改");
            result.passed = false;
          }
        }
      }
    } catch (error) {
      if (expectedBehavior.shouldThrowError) {
        result.passed = true;
      } else {
        result.errors.push(`意外错误: ${error.message}`);
        result.passed = false;
      }
    }

    return result;
  }
}

// 测试结果接口
interface TestResult {
  passed: boolean;
  errors: string[];
}
```

// 使用 Jest 进行单元测试
import { describe, it, expect } from "@jest/globals";

describe("RequestIdMiddleware", () => {
  let middleware: RequestIdMiddleware;

  beforeEach(() => {
    middleware = new RequestIdMiddleware();
  });

  it("应该为请求添加 Request-ID", async () => {
    // 准备
    const request = MiddlewareTester.createMockRequest();

    // 执行
    const modifiedRequest = await middleware.onBeforeRequest(request);

    // 验证
    const requestId = modifiedRequest.headers.get("X-Request-ID");
    expect(requestId).toBeDefined();
    expect(requestId).toMatch(/^req_/);
  });

  it("应该保留已有的 Request-ID", async () => {
    // 准备
    const existingId = "existing-request-id";
    const request = MiddlewareTester.createMockRequest({
      headers: { "X-Request-ID": existingId },
    });

    // 执行
    const modifiedRequest = await middleware.onBeforeRequest(request);

    // 验证
    expect(modifiedRequest.headers.get("X-Request-ID")).toBe(existingId);
  });

  it("应该在响应中添加 Request-ID", async () => {
    // 准备
    const request = MiddlewareTester.createMockRequest();
    const modifiedRequest = await middleware.onBeforeRequest(request);
    const response = MiddlewareTester.createMockResponse();
    (response as any).requestId = modifiedRequest.headers.get("X-Request-ID");

    // 执行
    const modifiedResponse = await middleware.onAfterResponse(response);

    // 验证
    expect(modifiedResponse.headers.get("X-Request-ID")).toBeDefined();
  });
});

describe("RateLimitMiddleware", () => {
  let middleware: RateLimitMiddleware;

  beforeEach(() => {
    middleware = new RateLimitMiddleware({
      enabled: true,
      options: {
        maxRequests: 2,
        windowMs: 1000,
      },
    });
  });

  it("应该允许未超限的请求", async () => {
    // 准备
    const request = MiddlewareTester.createMockRequest();

    // 执行 & 验证
    await expect(
      middleware.onBeforeRequest(request)
    ).resolves.toBeDefined();
  });

  it("应该拒绝超限的请求", async () => {
    // 准备
    const request = MiddlewareTester.createMockRequest();

    // 执行：发送超过限制的请求
    await middleware.onBeforeRequest(request);
    await middleware.onBeforeRequest(request);

    // 验证：第三次请求应该被拒绝
    await expect(
      middleware.onBeforeRequest(request)

  it("应该支持不同客户端独立计数", async () => {
    // 准备
    const request1 = MiddlewareTester.createMockRequest({
      headers: { "X-API-Key": "client1" },
    });
    const request2 = MiddlewareTester.createMockRequest({
      headers: { "X-API-Key": "client2" },
    });

    // 执行 & 验证：两个客户端各自独立
    await middleware.onBeforeRequest(request1);
    await middleware.onBeforeRequest(request1);
    await expect(
      middleware.onBeforeRequest(request2)
    ).resolves.toBeDefined();
  });

describe("Middleware Chain Integration", () => {
