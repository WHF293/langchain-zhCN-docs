# 第7章 消息

> 来源: [LangChain 官方文档](https://python.langchain.com/)

---

消息是 LangChain 中对话的基本单元，用于在用户和 AI 之间传递信息。

## 消息类型

LangChain 定义了几种标准消息类型：

### 消息层次结构

```
BaseMessage（基础消息）
  ├── SystemMessage（系统消息）
  ├── HumanMessage（用户消息）
  ├── AIMessage（AI 消息）
  ├── ToolMessage（工具消息）
  └── FunctionMessage（函数消息，已弃用）
```

### SystemMessage（系统消息）

用于设置 AI 的行为和角色：

```typescript
import { SystemMessage } from "@langchain/core/messages";

const systemMsg = new SystemMessage({
  content: "你是一个专业的翻译员，只翻译用户输入的文本，不做其他解释。",
});

// 带名称的系统消息
const systemMsg2 = new SystemMessage({
  content: "你是一个有帮助的助手",
  name: "system",
});
```

### HumanMessage（用户消息）

代表用户的输入：

```typescript
import { HumanMessage } from "@langchain/core/messages";

// 文本消息
const humanMsg = new HumanMessage({
  content: "你好，请帮我翻译这段话。",
});

// 多模态消息（包含图片）
const multimodalMsg = new HumanMessage({
  content: [
    { type: "text", text: "描述这张图片" },
    {
      type: "image_url",
      image_url: { url: "https://example.com/image.jpg" },
    },
  ],
});
```

### AIMessage（AI 消息）

代表 AI 的响应：

```typescript
import { AIMessage } from "@langchain/core/messages";

const aiMsg = new AIMessage({
  content: "你好！我很乐意帮你翻译。",
});

// 带工具调用的 AI 消息
const aiMsgWithToolCall = new AIMessage({
  content: "",
  tool_calls: [
    {
      name: "get_weather",
      args: { city: "北京" },
      id: "call_123",
    },
  ],
});
```

### ToolMessage（工具消息）

工具调用的返回结果：

```typescript
import { ToolMessage } from "@langchain/core/messages";

const toolMsg = new ToolMessage({
  content: "北京今天晴天，25°C",
  tool_call_id: "call_123",
  name: "get_weather",
});
```

## 消息历史

消息历史用于维护多轮对话的上下文。

### 内存消息历史

```typescript
import { ChatMessageHistory } from "@langchain/core/stores/history/in_memory";
import { HumanMessage, AIMessage } from "@langchain/core/messages";

// 创建消息历史
const history = new ChatMessageHistory();

// 添加消息
await history.addMessage(new HumanMessage("你好"));
await history.addMessage(new AIMessage("你好！有什么可以帮助你的？"));
await history.addMessage(new HumanMessage("我叫小明"));
await history.addMessage(new AIMessage("很高兴认识你，小明！"));

// 获取所有消息
const messages = await history.getMessages();
console.log(messages);

// 清空历史
await history.clear();
```

### 使用消息历史

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { ChatPromptTemplate, MessagesPlaceholder } from "@langchain/core/prompts";
import { StringOutputParser } from "@langchain/core/output_parsers";
import { ChatMessageHistory } from "@langchain/core/stores/history/in_memory";

// 创建模型和提示
const model = new ChatOpenAI({ modelName: "gpt-4" });
const prompt = ChatPromptTemplate.fromMessages([
  ["system", "你是一个友好的助手"],
  new MessagesPlaceholder("history"),
  ["user", "{input}"],
]);

// 创建链
const chain = prompt.pipe(model).pipe(new StringOutputParser());

// 创建历史
const history = new ChatMessageHistory();

// 聊天函数
async function chat(input: string) {
  const messages = await history.getMessages();

  const result = await chain.invoke({
    history: messages,
    input,
  });

  // 保存到历史
  await history.addMessage({ role: "user", content: input });
  await history.addMessage({ role: "assistant", content: result });

  return result;
}

// 多轮对话
console.log(await chat("你好，我叫小明"));
console.log(await chat("我的名字是什么？")); // 会记得是"小明"
```

### RunnableWithMessageHistory

更便捷的方式管理消息历史：

```typescript
import { ChatOpenAI } from "@langchain/openai";
import { ChatPromptTemplate, MessagesPlaceholder } from "@langchain/core/prompts";
import { StringOutputParser } from "@langchain/core/output_parsers";
import { RunnableWithMessageHistory } from "@langchain/core/runnables";
import { ChatMessageHistory } from "@langchain/core/stores/history/in_memory";

// 创建模型和提示
const model = new ChatOpenAI({ modelName: "gpt-4" });
const prompt = ChatPromptTemplate.fromMessages([
  ["system", "你是一个友好的助手"],
  new MessagesPlaceholder("history"),
  ["user", "{input}"],
]);

const chain = prompt.pipe(model).pipe(new StringOutputParser());

// 消息历史存储
const store = new Map<string, ChatMessageHistory>();

// 创建带历史的链
const withHistory = new RunnableWithMessageHistory({
  runnable: chain,
  getMessageHistory: (sessionId) => {
    if (!store.has(sessionId)) {
      store.set(sessionId, new ChatMessageHistory());
    }
    return store.get(sessionId)!,
  },
  inputMessagesKey: "input",
  historyMessagesKey: "history",
});

// 使用
const result = await withHistory.invoke(
  { input: "你好" },
  { configurable: { sessionId: "session-1" } }
);
```

## 消息模板

消息模板用于动态构建消息列表。

### ChatPromptTemplate

```typescript
import { ChatPromptTemplate } from "@langchain/core/prompts";

// 基础模板
const prompt = ChatPromptTemplate.fromMessages([
  ["system", "你是一个{role}"],
  ["user", "{input}"],
]);

// 格式化
const messages = await prompt.invoke({
  role: "翻译员",
  input: "你好",
});
```

### MessagesPlaceholder

用于插入动态消息列表：

```typescript
import { ChatPromptTemplate, MessagesPlaceholder } from "@langchain/core/prompts";

const prompt = ChatPromptTemplate.fromMessages([
  ["system", "你是一个友好的助手"],
  new MessagesPlaceholder("history"),  // 历史消息
  new MessagesPlaceholder("context"),  // 上下文
  ["user", "{input}"],
]);

// 使用
const messages = await prompt.invoke({
  history: [
    { role: "user", content: "我叫小明" },
    { role: "assistant", content: "你好，小明！" },
  ],
  context: [
    { role: "system", content: "当前时间：2024年1月1日" },
  ],
  input: "今天天气怎么样？",
});
```

### Few-shot 模板

使用示例来指导模型：

```typescript
import {
  ChatPromptTemplate,
  FewShotChatMessagePromptTemplate,
} from "@langchain/core/prompts";

// 定义示例
const examples = [
  { input: "你好", output: "你好！有什么可以帮助你的吗？" },
  { input: "谢谢", output: "不客气！如果还有其他问题，随时问我。" },
  { input: "再见", output: "再见！祝你有美好的一天！" },
];

// 创建示例模板
const examplePrompt = ChatPromptTemplate.fromMessages([
  ["human", "{input}"],
  ["ai", "{output}"],
]);

// 创建 few-shot 模板
const fewShotPrompt = new FewShotChatMessagePromptTemplate({
  examplePrompt,
  examples,
  inputVariables: ["input"],
});

// 完整提示
const finalPrompt = ChatPromptTemplate.fromMessages([
  ["system", "你是一个友好的中文助手"],
  fewShotPrompt,
  ["human", "{input}"],
]);

// 使用
const messages = await finalPrompt.invoke({ input: "早上好" });
```

### 管道提示模板

组合多个提示模板：

```typescript
import {
  ChatPromptTemplate,
  PipelinePromptTemplate,
} from "@langchain/core/prompts";

// 定义子模板
const systemTemplate = ChatPromptTemplate.fromTemplate(
  "你是一个{role}，专注于{domain}领域"
);

  "请完成以下任务：{task}"
);

  pipelinePrompts: [
    { name: "system", prompt: systemTemplate },
    { name: "task", prompt: taskTemplate },
  ],
  finalPrompt: ChatPromptTemplate.fromMessages([
    ["system", "{system}"],
    ["user", "{task}"],
  ]),
});
```

### 消息截断

```typescript
import { HumanMessage, AIMessage } from "@langchain/core/messages";

function truncateMessages(messages, maxTokens) {
  let totalTokens = 0;
  const result = [];

  // 从最新的消息开始保留
  for (let i = messages.length - 1; i >= 0; i--) {
    const msg = messages[i];
    const tokens = estimateTokens(msg.content);

    if (totalTokens + tokens > maxTokens) {
      break;
    }

    totalTokens += tokens;
    result.unshift(msg);
  }

  return result;
}

function estimateTokens(text) {
  // 简单估算：4 字符 ≈ 1 token
  return Math.ceil(text.length / 4);
}
```

### 历史管理最佳实践

```typescript
import { ChatMessageHistory } from "@langchain/core/stores/history/in_memory";

// ✅ 好的做法：定期清理历史
async function manageHistory(history, maxMessages = 50) {
  const messages = await history.getMessages();
  if (messages.length > maxMessages) {
    // 保留系统消息和最近的消息
    const systemMessages = messages.filter(m => m._getType() === "system");
    const recentMessages = messages.slice(-maxMessages);

    await history.clear();
    for (const msg of [...systemMessages, ...recentMessages]) {
      await history.addMessage(msg);
    }
  }
}
```

## 常见问题

### 消息历史丢失

**问题**：应用重启后消息历史消失

**解决方案**：使用持久化存储

```typescript
import { RedisChatMessageHistory } from "@langchain/community/stores/message/ioredis";

const history = new RedisChatMessageHistory("session-123", {
  sessionId: "user-session",
});
```

### 内存溢出

**问题**：长时间对话导致内存占用过高

**解决方案**：限制历史消息数量

```typescript
const MAX_MESSAGES = 100;

async function trimHistory(history) {
  const messages = await history.getMessages();
  if (messages.length > MAX_MESSAGES) {
    await history.clear();
    const recent = messages.slice(-MAX_MESSAGES);
    for (const msg of recent) {
      await history.addMessage(msg);
    }
  }
}
```

### 消息格式错误

**问题**：模型返回意外的消息格式

**解决方案**：使用输出解析器

```typescript
import { StringOutputParser } from "@langchain/core/output_parsers";

const parser = new StringOutputParser();
const chain = prompt.pipe(model).pipe(parser);

// 确保输出是字符串
const result = await chain.invoke(input);
```
